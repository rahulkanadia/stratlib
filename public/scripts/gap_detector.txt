"""
Gap Detection and Mean Reversion Analysis Utilities

This module provides core gap identification and analysis functions for the mean reversion strategy.
It extracts OHLC data from CSV files and implements the gap detection criteria specific to the
SPY gap-fading strategy.

Key Functions:
- Extract OHLC data from CSV files into Python lists for analysis
- Validate gaps: $0.20 minimum, 15%-85% of previous day's range
- Calculate P&L outcomes: detect gap fills, measure profit/loss
- Analyze success rates: percentage of gaps that revert vs extend

This was adapted from Spy_Close.py and represents core strategy logic in computational form.
Used for both backtesting and real-time signal generation.

Author: Daniel Peer
"""

import pandas as pd

pd.options.display.show_dimensions = False

def data_frame_file_to_lists(file_name):
    df = pd.read_csv(file_name, usecols=['Close'])
    df2 = pd.read_csv(file_name, usecols=['Open'])
    df3 = pd.read_csv(file_name, usecols=['Date'])
    df4 = pd.read_csv(file_name, usecols=['High'])
    df5 = pd.read_csv(file_name, usecols=['Low'])
    close_list = df.values.tolist()
    open_list = df2.values.tolist()
    date_list = df3.values.tolist()
    high_list = df4.values.tolist()  # Corrected to use df4 for 'High'
    low_list = df5.values.tolist()
    return close_list, open_list, date_list, high_list, low_list


def data_frame_to_doubles(file_name):
    stock = pd.read_csv(file_name)
    list_of_dataframes = [stock.loc[i:i + 1].reset_index(drop=True) for i in range(len(stock), 0, -1)]
    return list_of_dataframes[2:]


def calc_gap(high, low):
    gap = high - low
    return gap > 0.2


def chek_two_days_abs(double_df):
    """
    Validates whether a gap meets trading criteria.

    Checks if gap size is within acceptable range: at least $0.20 and between 15%-85%
    of the previous day's trading range. Gaps outside this range are either noise or
    structural moves unlikely to mean-revert.

    Parameters:
        double_df (DataFrame): Two-row DataFrame with consecutive days [day1, day2]

    Returns:
        bool: True if gap meets all validation criteria, False otherwise
    """
    day1, day2 = double_df.loc[0], double_df.loc[1]
    spread1 = abs(day1.loc['High'] - day1.loc['Low'])  # Previous day's range
    spread2 = abs(day1.loc['Close'] - day2.loc['Open'])  # Gap size

    if (spread2 < 0.2) or (spread1 * 0.85 < spread2) or (spread1 * 0.15 > spread2):
        return False
    return True


def chek_two_days(double_df):
    day1, day2 = double_df.loc[0], double_df.loc[1]

    spread1 = abs(day1.loc['High'] - day1.loc['Low'])
    spread2 = abs(day1.loc['Close'] - day2.loc['Open'])
    if (spread2 < 0.2) or (spread1 * 0.85 < spread2) or (spread1 * 0.15 > spread2):
        return (0, 0)
    return spread2, day2.loc['Open']


def price_calc(buying_price, double_df):
    """
    Calculates P&L outcome for a gap trade.

    For a valid gap, determines:
    1. Profit magnitude: absolute difference between close and entry
    2. Win/Loss: 1 if profitable, 0 if loss
    3. Direction: 0 for short (gap up), 1 for long (gap down)
    4. Entry price: the open price where trade was entered

    Returns invalid tuple (2,2,2,2) if gap doesn't meet criteria.

    Parameters:
        buying_price (float): Unused, kept for interface compatibility
        double_df (DataFrame): Two consecutive days [day1, day2]

    Returns:
        tuple: (profit_magnitude, win, direction, entry_price)
               profit_magnitude: absolute P&L in dollars
               win: 1 if trade was profitable, 0 if loss
               direction: 0=short gap-up, 1=long gap-down
               entry_price: market open price (day2_open)
    """
    if not chek_two_days_abs(double_df):
        return 2, 2, 2, 2  # Invalid gap marker

    day1, day2 = double_df.loc[0], double_df.loc[1]
    day1_open, day1_close = day1.loc['Open'], day1.loc['Close']
    day2_open, day2_close = day2.loc['Open'], day2.loc['Close']
    day2_high, day2_low = day2.loc['High'], day2.loc['Low']
    spread1 = day1_open - day1_close

    if day2_open >= day1_close:  # Gap up (fade with short)
        if day2_low <= day1_close:  # Gap closed to previous close
            return abs(day2_open - day1_close), 1, 0, day2_open  # Profitable short
        if day2_close < day2_open:  # Red day (close below open)
            return abs(day2_close - day1_close), 1, 0, day2_open  # Profitable short
        return abs(day2_close - day1_close), 0, 0, day2_open  # Loss on short

    if day2_open <= day1_close:  # Gap down (fade with long)
        if day2_high >= day1_close:  # Gap closed to previous close
            return abs(day1_close - day2_open), 1, 1, day2_open  # Profitable long
        if day2_close > day2_open:  # Green day (close above open)
            return abs(day2_close - day1_close), 1, 1, day2_open  # Profitable long
        return abs(day2_close - day1_close), 0, 1, day2_open  # Loss on long

    return 999999999999999999999999999  # Data error


def num_of_getting_in(list_double_df):
    cnt = 0
    for i in range(len(list_double_df)):
        if chek_two_days_abs(list_double_df[i]):
            cnt += 1
    return cnt

def num_of_events(days_to_check, file_name):
    tpls_lst = []
    list_of_doubles = data_frame_to_doubles(file_name)
    tmp = list_of_doubles[:days_to_check]
    tmp = tmp[::-1]
    for i in range(days_to_check):
        price = price_calc(0, tmp[i])
        if price[0] != 2:
            tpls_lst.append(price)
    return tpls_lst


def turn_tuple_to_percentage(tpl):
    perc = 0
    if tpl[1] == 1:  # Gain
        if tpl[2] == 1:  # Long
            perc = (tpl[3] + tpl[0]) / tpl[3]
        if tpl[2] == 0:  # Short
            perc = (tpl[3] / (tpl[0] - tpl[3]))
    if tpl[1] == 0:  # Loss
        if tpl[2] == 1:
            perc = (tpl[3] - tpl[0]) / tpl[3]
        if tpl[2] == 0:  # Short
            perc = (tpl[3] / (tpl[0] + tpl[3]))
    return perc
