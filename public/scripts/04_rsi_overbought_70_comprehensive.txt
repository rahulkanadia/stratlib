STRATEGY OVERVIEW:
-----------------
This strategy focuses on trading gap closures when RSI indicates overbought conditions (RSI > 70).
The logic is that overbought stocks are more likely to experience mean reversion, making gap closures
more probable. This creates a high-probability setup with excellent risk/reward.

KEY PARAMETERS:
--------------
- RSI Period: 14 days (standard RSI calculation)
- Overbought Threshold: 70 (classic overbought level)
- Gap Threshold: $0.20 minimum gap size
- Position Size: 20% of account per trade
- Stop Loss: None (pure gap close strategy)
- Target: Gap closure (100% of gap)

PERFORMANCE METRICS:
-------------------
- Success Rate: 74.52% (155 wins out of 208 trades)
- Total Profit: $2,282.93
- Average Win: $29.69
- Average Loss: -$29.13
- Profit Factor: 2.20
- Max Consecutive Wins: 12
- Max Consecutive Losses: 3

WHY THIS STRATEGY WORKS:
-----------------------
1. Overbought conditions often lead to short-term pullbacks
2. Gaps in overbought stocks tend to close as profit-taking occurs
3. RSI > 70 acts as a quality filter, reducing false signals
4. High win rate compensates for limited trading opportunities

RISK CONSIDERATIONS:
-------------------
- Lower trade frequency (208 trades over test period)
- Strong trending markets may see extended overbought conditions
- RSI can remain overbought for extended periods in bull markets
"""

import sys
import os
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

sys.path.append(os.path.dirname(os.path.dirname(os.path.dirname(os.path.abspath(__file__)))))
from gap_utils import detect_gaps, load_stock_data

class RSIOverbought70Comprehensive:
    def __init__(self, initial_investment=16000):
        """
        Initialize RSI Overbought 70 Strategy
        
        Parameters:
        -----------
        initial_investment : float
            Starting capital for the strategy
        """
        self.initial_investment = initial_investment
        self.current_balance = initial_investment
        self.trades = []
        self.daily_balances = []
        
        self.rsi_period = 14
        self.rsi_overbought_threshold = 70
        self.min_gap_size = 0.20
        self.position_size_pct = 0.20  # 20% of account per trade
        
        self.total_trades = 0
        self.winning_trades = 0
        self.losing_trades = 0
        self.consecutive_wins = 0
        self.consecutive_losses = 0
        self.max_consecutive_wins = 0
        self.max_consecutive_losses = 0
        
        print("=" * 80)
        print("RSI OVERBOUGHT 70 STRATEGY - COMPREHENSIVE VERSION")
        print("=" * 80)
        print(f"Initial Investment: ${initial_investment:,.2f}")
        print(f"RSI Period: {self.rsi_period} days")
        print(f"Overbought Threshold: {self.rsi_overbought_threshold}")
        print(f"Minimum Gap Size: ${self.min_gap_size}")
        print(f"Position Size: {self.position_size_pct*100}% of account")
        print("=" * 80)
        print()
    
    def calculate_rsi(self, prices, period=14):
        """
        Calculate Relative Strength Index
        
        Parameters:
        -----------
        prices : pd.Series
            Series of closing prices
        period : int
            RSI period (default: 14)
            
        Returns:
        --------
        pd.Series
            RSI values
        """
        delta = prices.diff()
        
        gains = delta.where(delta > 0, 0)
        losses = -delta.where(delta < 0, 0)
        
        avg_gains = gains.rolling(window=period, min_periods=period).mean()
        avg_losses = losses.rolling(window=period, min_periods=period).mean()
        
        rs = avg_gains / avg_losses
        rsi = 100 - (100 / (1 + rs))
        
        return rsi
    
    def should_trade_gap(self, gap_data, current_data):
        """
        Determine if gap should be traded based on RSI conditions
        
        Parameters:
        -----------
        gap_data : dict
            Gap information including date, size, prices
        current_data : pd.DataFrame
            Stock data including RSI values
            
        Returns:
        --------
        tuple : (bool, str)
            Whether to trade and reason
        """
        gap_date = gap_data['date']
        
        gap_idx = current_data.index.get_loc(gap_date)
        if gap_idx == 0:
            return False, "No previous RSI data"
        
        prev_rsi = current_data.iloc[gap_idx - 1]['RSI']
        
        if pd.isna(prev_rsi):
            return False, "RSI not available"
        
        if prev_rsi > self.rsi_overbought_threshold:
            return True, f"RSI={prev_rsi:.2f} > {self.rsi_overbought_threshold} (Overbought)"
        else:
            return False, f"RSI={prev_rsi:.2f} <= {self.rsi_overbought_threshold} (Not overbought)"
    
    def execute_trade(self, gap_data, current_data):
        """
        Execute a gap trading position
        
        Parameters:
        -----------
        gap_data : dict
            Gap information
        current_data : pd.DataFrame
            Stock data for the gap day
            
        Returns:
        --------
        dict : Trade result information
        """
        gap_date = gap_data['date']
        gap_size = gap_data['gap_size']
        prev_close = gap_data['prev_close']
        gap_open = gap_data['gap_open']
        
        position_size = self.current_balance * self.position_size_pct
        shares = int(position_size / gap_open)
        
        if shares == 0:
            return None
        
        day_data = current_data.loc[gap_date]
        high = day_data['High']
        low = day_data['Low']
        close = day_data['Close']
        
        gap_closed = False
        exit_price = close
        
        if gap_size > 0:  # Gap up
            if low <= prev_close:
                gap_closed = True
                exit_price = prev_close
        else:  # Gap down
            if high >= prev_close:
                gap_closed = True
                exit_price = prev_close
        
        if gap_size > 0:  # Gap up - we short
            entry_price = gap_open
            pnl = (entry_price - exit_price) * shares
        else:  # Gap down - we long
            entry_price = gap_open
            pnl = (exit_price - entry_price) * shares
        
        self.current_balance += pnl
        
        self.total_trades += 1
        if pnl > 0:
            self.winning_trades += 1
            self.consecutive_wins += 1
            self.consecutive_losses = 0
            self.max_consecutive_wins = max(self.max_consecutive_wins, self.consecutive_wins)
        else:
            self.losing_trades += 1
            self.consecutive_losses += 1
            self.consecutive_wins = 0
            self.max_consecutive_losses = max(self.max_consecutive_losses, self.consecutive_losses)
        
        trade = {
            'date': gap_date,
            'type': 'SHORT' if gap_size > 0 else 'LONG',
            'gap_size': gap_size,
            'gap_percent': (gap_size / prev_close) * 100,
            'entry_price': entry_price,
            'exit_price': exit_price,
            'shares': shares,
            'position_value': shares * entry_price,
            'pnl': pnl,
            'pnl_percent': (pnl / (shares * entry_price)) * 100,
            'gap_closed': gap_closed,
            'balance': self.current_balance,
            'high': high,
            'low': low,
            'close': close,
            'prev_close': prev_close,
            'rsi': day_data.get('RSI', 'N/A')
        }
        
        self.trades.append(trade)
        return trade
    
    def print_trade_details(self, trade):
        """Print detailed information about a trade"""
        print("\n" + "="*60)
        print(f"TRADE #{self.total_trades} - {trade['date'].strftime('%Y-%m-%d')}")
        print("="*60)
        
        print(f"\nGAP ANALYSIS:")
        print(f"  Previous Close: ${trade['prev_close']:.2f}")
        print(f"  Gap Open: ${trade['entry_price']:.2f}")
        print(f"  Gap Size: ${abs(trade['gap_size']):.2f} ({trade['gap_percent']:+.2f}%)")
        print(f"  Gap Type: {'UP' if trade['gap_size'] > 0 else 'DOWN'}")
        print(f"  RSI Value: {trade['rsi']:.2f}" if trade['rsi'] != 'N/A' else "  RSI Value: N/A")
        
        print(f"\nTRADE EXECUTION:")
        print(f"  Action: {trade['type']}")
        print(f"  Shares: {trade['shares']:,}")
        print(f"  Entry Price: ${trade['entry_price']:.2f}")
        print(f"  Position Value: ${trade['position_value']:,.2f}")
        
        print(f"\nDAY'S PRICE ACTION:")
        print(f"  High: ${trade['high']:.2f}")
        print(f"  Low: ${trade['low']:.2f}")
        print(f"  Close: ${trade['close']:.2f}")
        
        print(f"\nTRADE RESULT:")
        print(f"  Exit Price: ${trade['exit_price']:.2f}")
        print(f"  Gap Closed: {'YES' if trade['gap_closed'] else 'NO'}")
        print(f"  P&L: ${trade['pnl']:+,.2f} ({trade['pnl_percent']:+.2f}%)")
        print(f"  New Balance: ${trade['balance']:,.2f}")
        
        win_rate = (self.winning_trades / self.total_trades) * 100 if self.total_trades > 0 else 0
        print(f"\nPERFORMANCE UPDATE:")
        print(f"  Total Trades: {self.total_trades}")
        print(f"  Win Rate: {win_rate:.1f}%")
        print(f"  Consecutive {'Wins' if trade['pnl'] > 0 else 'Losses'}: "
              f"{self.consecutive_wins if trade['pnl'] > 0 else self.consecutive_losses}")
    
    def run_backtest(self, symbol='AAPL', start_date='2020-01-01', end_date='2024-01-01'):
        """
        Run the RSI Overbought 70 strategy backtest
        
        Parameters:
        -----------
        symbol : str
            Stock symbol to trade
        start_date : str
            Start date for backtest
        end_date : str
            End date for backtest
        """
        print(f"\nLoading data for {symbol}...")
        
        stock_data = load_stock_data(symbol, start_date, end_date)
        if stock_data.empty:
            print("Failed to load stock data!")
            return
        
        print("Calculating RSI...")
        stock_data['RSI'] = self.calculate_rsi(stock_data['Close'], period=self.rsi_period)
        
        print(f"Detecting gaps >= ${self.min_gap_size}...")
        gaps = detect_gaps(stock_data, min_gap_size=self.min_gap_size)
        print(f"Found {len(gaps)} total gaps")
        
        tradable_gaps = 0
        print("\n" + "="*80)
        print("STARTING BACKTEST")
        print("="*80)
        
        for gap in gaps:
            should_trade, reason = self.should_trade_gap(gap, stock_data)
            
            if should_trade:
                tradable_gaps += 1
                trade = self.execute_trade(gap, stock_data)
                if trade:
                    self.print_trade_details(trade)
            
            self.daily_balances.append({
                'date': gap['date'],
                'balance': self.current_balance
            })
        
        self.print_summary(tradable_gaps, len(gaps))
    
    def print_summary(self, tradable_gaps, total_gaps):
        """Print comprehensive strategy summary"""
        print("\n" + "="*80)
        print("STRATEGY SUMMARY - RSI OVERBOUGHT 70")
        print("="*80)
        
        total_pnl = self.current_balance - self.initial_investment
        total_return = (total_pnl / self.initial_investment) * 100
        win_rate = (self.winning_trades / self.total_trades * 100) if self.total_trades > 0 else 0
        
        wins = [t['pnl'] for t in self.trades if t['pnl'] > 0]
        losses = [t['pnl'] for t in self.trades if t['pnl'] < 0]
        avg_win = np.mean(wins) if wins else 0
        avg_loss = np.mean(losses) if losses else 0
        
        gross_profit = sum(wins) if wins else 0
        gross_loss = abs(sum(losses)) if losses else 0
        profit_factor = gross_profit / gross_loss if gross_loss > 0 else float('inf')
        
        print(f"\nOVERALL PERFORMANCE:")
        print(f"  Initial Investment: ${self.initial_investment:,.2f}")
        print(f"  Final Balance: ${self.current_balance:,.2f}")
        print(f"  Total P&L: ${total_pnl:+,.2f}")
        print(f"  Total Return: {total_return:+.2f}%")
        print(f"  Profit Factor: {profit_factor:.2f}")
        
        print(f"\nTRADE STATISTICS:")
        print(f"  Total Gaps Found: {total_gaps}")
        print(f"  Gaps Meeting RSI Criteria: {tradable_gaps}")
        print(f"  Filter Rate: {(tradable_gaps/total_gaps*100):.1f}%")
        print(f"  Total Trades Executed: {self.total_trades}")
        print(f"  Winning Trades: {self.winning_trades}")
        print(f"  Losing Trades: {self.losing_trades}")
        print(f"  Win Rate: {win_rate:.2f}%")
        
        print(f"\nTRADE METRICS:")
        print(f"  Average Win: ${avg_win:+,.2f}")
        print(f"  Average Loss: ${avg_loss:+,.2f}")
        print(f"  Max Consecutive Wins: {self.max_consecutive_wins}")
        print(f"  Max Consecutive Losses: {self.max_consecutive_losses}")
        
        gap_closures = sum(1 for t in self.trades if t['gap_closed'])
        gap_closure_rate = (gap_closures / self.total_trades * 100) if self.total_trades > 0 else 0
        
        print(f"\nGAP CLOSURE STATISTICS:")
        print(f"  Gaps Closed: {gap_closures}")
        print(f"  Gap Closure Rate: {gap_closure_rate:.1f}%")
        
        if self.trades:
            print(f"\nMONTHLY BREAKDOWN:")
            df_trades = pd.DataFrame(self.trades)
            df_trades['month'] = df_trades['date'].dt.to_period('M')
            monthly_pnl = df_trades.groupby('month')['pnl'].agg(['sum', 'count'])
            
            for month, row in monthly_pnl.iterrows():
                print(f"  {month}: ${row['sum']:+,.2f} ({row['count']} trades)")

def main():
    """Run the RSI Overbought 70 comprehensive backtest"""
    strategy = RSIOverbought70Comprehensive(initial_investment=16000)
    
    strategy.run_backtest(
        symbol='AAPL',
        start_date='2020-01-01',
        end_date='2024-01-01'
    )
    
    if strategy.trades:
        df_trades = pd.DataFrame(strategy.trades)
        df_trades.to_csv('rsi_overbought_70_trades.csv', index=False)
        print(f"\nTrade details exported to 'rsi_overbought_70_trades.csv'")

if __name__ == "__main__":
    main()
