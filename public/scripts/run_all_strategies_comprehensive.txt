import pandas as pd

def run_single_strategy(file_path):
    """Run a single strategy file and extract results."""
    try:
        from io import StringIO
        import contextlib
        
        old_stdout = sys.stdout
        sys.stdout = buffer = StringIO()
        
        spec = importlib.util.spec_from_file_location("strategy", file_path)
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        
        sys.stdout = old_stdout
        output = buffer.getvalue()
        
        lines = output.split('\n')
        
        total_trades = 0
        success_rate = 0
        total_profit = 0
        
        for line in lines:
            if 'Total Trades:' in line:
                total_trades = int(line.split(':')[1].strip())
            elif 'Success Rate:' in line and '%' in line:
                success_rate = float(line.split(':')[1].replace('%', '').strip())
            elif 'Total Profit:' in line and '$' in line:
                profit_str = line.split('$')[1].replace(',', '').strip()
                if profit_str:
                    total_profit = float(profit_str)
        
        return {
            'total_trades': total_trades,
            'success_rate': success_rate,
            'total_profit': total_profit,
            'status': 'Success'
        }
        
    except Exception as e:
        return {
            'total_trades': 0,
            'success_rate': 0,
            'total_profit': 0,
            'status': f'Error: {str(e)}'
        }
