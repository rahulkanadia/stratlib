"""
Gap & Go Success_Rate #76 Strategy

Gap trading strategy optimized for success_rate with Uptrend Sma50 And Medium Gap

Strategy Type: Gap & Go (Success Rate Optimized)
Category: gap_and_go

This is a standalone, beginner-friendly implementation.
"""

import pandas as pd
import numpy as np
from pathlib import Path
import sys
import os
import warnings
warnings.filterwarnings('ignore')

current_file = Path(os.path.abspath(__file__))
project_root = current_file.parent.parent.parent
sys.path.insert(0, str(project_root))

from utils.base_strategy_fixed import BaseStrategy

class GapAndGoSuccess_Rate76Strategy(BaseStrategy):
    """
    Gap & Go Success_Rate #76 Implementation
    
    
    This strategy trades morning gaps when specific conditions are met:
    - Uptrend Sma50 And Medium Gap
    - Optimized for maximum success_rate
    
    """
    
    def __init__(self, data_path=None):
        super().__init__(
            name="Gap & Go Success_Rate #76",
            description="Trades gaps with Uptrend Sma50 And Medium Gap",
            required_data=['Date', 'Open', 'High', 'Low', 'Close', 'Volume']
        )
        
        self.min_gap_percent = 0.5  # Minimum gap size
        self.max_gap_percent = 5.0   # Maximum gap size
        self.ma_fast = 5
        self.ma_slow = 10
        
        if data_path:
            self.load_data(data_path)
    
    def calculate_indicators(self):
        """Calculate required indicators"""
        df = self.data.copy()
        
        df['Gap'] = df['Open'] - df['Close'].shift(1)
        df['Gap_Percent'] = (df['Gap'] / df['Close'].shift(1)) * 100
        
        df['MA_Fast'] = df['Close'].rolling(window=self.ma_fast).mean()
        df['MA_Slow'] = df['Close'].rolling(window=self.ma_slow).mean()
        
        return df
    
    def calculate_signals(self):
        """Generate trading signals based on strategy rules"""
        if self.data is None:
            raise ValueError("No data loaded. Please load data first.")
        
        df = self.calculate_indicators()
        signals = []
        
        for i in range(20, len(df)):  # Start after indicators are calculated
            curr_day = df.iloc[i]
            prev_day = df.iloc[i-1]
            
            gap_percent = curr_day['Gap_Percent']
            if pd.isna(gap_percent) or abs(gap_percent) < self.min_gap_percent:
                continue
            if abs(gap_percent) > self.max_gap_percent:
                continue
            
            condition_met = True
            
            if condition_met:
                if gap_percent > 0:  # Gap up - SHORT
                    direction = 'SHORT'
                    entry_price = curr_day['Open']
                    
                    if curr_day['Low'] <= prev_day['Close']:
                        exit_price = prev_day['Close']
                    else:
                        exit_price = curr_day['Close']
                else:  # Gap down - LONG
                    direction = 'LONG'
                    entry_price = curr_day['Open']
                    
                    if curr_day['High'] >= prev_day['Close']:
                        exit_price = prev_day['Close']
                    else:
                        exit_price = curr_day['Close']
                
                signal = {
                    'date': curr_day['Date'],
                    'direction': direction,
                    'entry_price': entry_price,
                    'exit_price': exit_price,
                    'gap_percent': gap_percent,
                    'signal_strength': min(abs(gap_percent) / 2.0, 1.0)
                }
                signals.append(signal)
        
        self.signals = pd.DataFrame(signals)
        return self.signals
    
    def run_backtest(self, initial_capital=10000, position_size=1.0):
        """Run backtest simulation"""
        return super().run_backtest(initial_capital, position_size)
    
    def get_summary_metrics(self):
        """Get performance metrics with strategy-specific additions"""
        if self.performance_metrics is None:
            return None
        
        metrics = self.performance_metrics.copy()
        
        
        if self.signals is not None and len(self.signals) > 0:
            metrics['avg_gap_size'] = self.signals.get('gap_percent', pd.Series()).mean()
            metrics['conditions_met'] = len(self.signals)
        
        
        return metrics

def main():
    """Run the strategy with example usage"""
    print("=" * 60)
    print("GAP & GO SUCCESS_RATE #76 STRATEGY")
    print("=" * 60)
    print("\nTrades gaps when Uptrend Sma50 And Medium Gap conditions are present")
    print("=" * 60)
    
    data_path = Path(__file__).parent.parent.parent / "data" / "7J0OK1VR.csv"
    
    strategy = GapAndGoSuccess_Rate76Strategy(data_path)
    
    print("\nSearching for trading opportunities...")
    signals = strategy.calculate_signals()
    print(f"Found {len(signals)} trading signals")
    
    if len(signals) > 0:
        print(f"\nFirst signal example:")
        first_signal = signals.iloc[0]
        print(f"  Date: {first_signal['date']}")
        print(f"  Direction: {first_signal.get('direction', 'N/A')}")
        print(f"  Entry: ${first_signal.get('entry_price', 0):.2f}")
    
    print("\nRunning backtest with $10,000 initial capital...")
    strategy.run_backtest(initial_capital=10000)
    
    metrics = strategy.get_summary_metrics()
    
    print("\n" + "=" * 60)
    print("STRATEGY RESULTS")
    print("=" * 60)
    if metrics:
        print(f"Total Trades: {metrics['total_trades']}")
        print(f"Win Rate: {metrics['win_rate']:.1f}%")
        print(f"Total Profit/Loss: ${metrics['total_profit']:.2f}")
        print(f"Sharpe Ratio: {metrics.get('sharpe_ratio', 0):.2f}")
        print(f"Max Drawdown: {metrics['max_drawdown']:.1f}%")
        print(f"Win/Loss Ratio: {metrics.get('win_loss_ratio', 0):.2f}")
    else:
        print("No trades executed. Check strategy parameters.")
    
    print("\n" + "=" * 60)
    print("💡 TIP: Adjust parameters in __init__ to optimize performance")

if __name__ == "__main__":
    main()
