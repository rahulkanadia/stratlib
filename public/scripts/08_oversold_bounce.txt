import pandas as pd
import numpy as np
from base_strategy import GapStrategy
import sys
import os
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from gap_utils import *

class OversoldBounceStrategy(GapStrategy):
    """
    Oversold Bounce Strategy
    Success Rate: 73.08% | Total Profit: $2,325.51 | Trades: 26
    
    This complex strategy combines multiple oversold indicators:
    RSI < 30, Price < Lower BB, and Stochastic < 20 for high-confidence setups.
    """
    
    def __init__(self):
        super().__init__(
            name="Oversold Bounce",
            description="Fade gaps when multiple oversold conditions align"
        )
        self.rsi_period = 14
        self.rsi_oversold = 30
        self.bb_period = 20
        self.bb_std = 2
        self.stoch_period = 14
        self.stoch_oversold = 20
    
    def calculate_indicators(self, df):
        """Calculate RSI, Bollinger Bands, and Stochastic"""
        df['RSI'] = calculate_rsi(df, period=self.rsi_period)
        
        upper_bb, middle_bb, lower_bb = calculate_bollinger_bands(
            df, period=self.bb_period, std_dev=self.bb_std
        )
        df['BB_Upper'] = upper_bb
        df['BB_Middle'] = middle_bb
        df['BB_Lower'] = lower_bb
        
        k_percent, d_percent = calculate_stochastic(df, period=self.stoch_period)
        df['Stoch_K'] = k_percent
        df['Stoch_D'] = d_percent
        
        return df
    
    def get_filter_condition(self, df):
        """All three oversold conditions must be met"""
        return (
            (df['RSI'] < self.rsi_oversold) & 
            (df['Close'].shift(1) < df['BB_Lower']) & 
            (df['Stoch_K'] < self.stoch_oversold)
        )

def main():
    """Run the strategy"""
    csv_files = [f for f in os.listdir('.') if f.endswith('.csv') and f not in ['gap_strategies_all_results.csv', 'gap_strategies_top10_profit.csv', 'gap_strategies_top10_success_rate.csv']]
    
    if csv_files:
        data_file = csv_files[0]
        print(f"Using data file: {data_file}")
    else:
        parent_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
        data_file = os.path.join(parent_dir, '..', '7J0OK1VR.csv')
        if not os.path.exists(data_file):
            print("Error: No CSV data file found. Please place a CSV file in the current directory.")
            return
    
    strategy = OversoldBounceStrategy()
    
    try:
        stats, trades_df = strategy.run_strategy(data_file)
        strategy.print_results(stats, trades_df)
        
        output_file = f"{strategy.name.replace(' ', '_')}_trades.csv"
        trades_df.to_csv(output_file, index=False)
        print(f"Trade details saved to: {output_file}")
        
    except Exception as e:
        print(f"Error running strategy: {e}")

if __name__ == "__main__":
    main()
