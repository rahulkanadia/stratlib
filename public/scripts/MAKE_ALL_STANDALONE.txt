def load_data(file_path):
    """Load trading data from CSV file"""
    try:
        df = pd.read_csv(file_path, parse_dates=['Date'])
        df.sort_values('Date', inplace=True)
        df.reset_index(drop=True, inplace=True)
        return df
    except FileNotFoundError:
        print(f"Warning: Could not find {{file_path}}")
        print("Using sample data instead...")
        return create_sample_data()
    except Exception as e:
        print(f"Error loading data: {{e}}")
        print("Using sample data instead...")
        return create_sample_data()

def create_sample_data():
    """Create sample SPY data for testing"""
    dates = pd.date_range(start='2024-01-01', end='2024-01-31', freq='B')
    np.random.seed(42)
    
    data = []
    price = 470.0
    
    for date in dates:
        daily_return = np.random.normal(0.0002, 0.01)  # 0.02% mean, 1% std
        open_price = price * (1 + np.random.normal(0, 0.002))
        high_price = open_price * (1 + abs(np.random.normal(0, 0.005)))
        low_price = open_price * (1 - abs(np.random.normal(0, 0.005)))
        close_price = open_price * (1 + daily_return)
        volume = int(np.random.normal(80000000, 20000000))
        
        data.append({{
            'Date': date,
            'Open': round(open_price, 2),
            'High': round(max(high_price, open_price, close_price), 2),
            'Low': round(min(low_price, open_price, close_price), 2),
            'Close': round(close_price, 2),
            'Volume': max(volume, 1000000)
        }})
        
        price = close_price
    
    return pd.DataFrame(data)

def calculate_returns(prices):
    """Calculate simple returns from price series"""
    return prices.pct_change().fillna(0)

def calculate_sharpe_ratio(returns, risk_free_rate=0.02, periods_per_year=252):
    """Calculate Sharpe Ratio"""
    if len(returns) < 2:
        return 0
    
    excess_returns = returns - risk_free_rate / periods_per_year
    if excess_returns.std() == 0:
        return 0
    
    return np.sqrt(periods_per_year) * excess_returns.mean() / excess_returns.std()

def calculate_max_drawdown(equity_curve):
    """Calculate maximum drawdown from equity curve"""
    if len(equity_curve) < 2:
        return 0
        
    rolling_max = equity_curve.expanding().max()
    drawdown = (equity_curve - rolling_max) / rolling_max
    return abs(drawdown.min() * 100)

def calculate_performance_metrics(trades, initial_capital=10000):
    """Calculate all performance metrics"""
    if not trades:
        return {{
            'total_trades': 0,
            'win_rate': 0,
            'total_profit': 0,
            'sharpe_ratio': 0,
            'max_drawdown': 0,
            'win_loss_ratio': 0
        }}
    
    df = pd.DataFrame(trades)
    
    total_trades = len(df)
    winning_trades = len(df[df['profit_loss'] > 0])
    losing_trades = len(df[df['profit_loss'] < 0])
    
    win_rate = (winning_trades / total_trades * 100) if total_trades > 0 else 0
    total_profit = df['profit_loss'].sum()
    
    avg_win = df[df['profit_loss'] > 0]['profit_loss'].mean() if winning_trades > 0 else 0
    avg_loss = abs(df[df['profit_loss'] < 0]['profit_loss'].mean()) if losing_trades > 0 else 0
    win_loss_ratio = (avg_win / avg_loss) if avg_loss > 0 else float('inf') if avg_win > 0 else 0
    
    equity_curve = initial_capital + df['profit_loss'].cumsum()
    returns = calculate_returns(equity_curve)
    
    sharpe_ratio = calculate_sharpe_ratio(returns)
    max_drawdown = calculate_max_drawdown(equity_curve)
    
    return {{
        'total_trades': total_trades,
        'win_rate': round(win_rate, 1),
        'total_profit': round(total_profit, 2),
        'sharpe_ratio': round(sharpe_ratio, 2),
        'max_drawdown': round(max_drawdown, 1),
        'win_loss_ratio': round(win_loss_ratio, 2)
    }}

def extract_strategy_info(file_path, content):
    """Extract strategy information from existing file"""
    info = {

        'class_name': 'UnknownStrategy',

        'strategy_name_upper': 'UNKNOWN STRATEGY'
    }
    
    filename = file_path.stem

        number = filename.split('_')[3]
        condition = ' '.join(filename.split('_')[4:]).replace('_', ' ')
        info['strategy_name'] = f"Gap & Go Profit #{number} - {condition.title()}"
        info['class_name'] = f"GapGoProfit{number}Strategy"
        info['strategy_explanation'] = f"Gap trading optimized for profit with {condition}"
    elif filename.startswith('gap_go_success_'):
        number = filename.split('_')[3]
        condition = ' '.join(filename.split('_')[4:]).replace('_', ' ')
        info['strategy_name'] = f"Gap & Go Success #{number} - {condition.title()}"
        info['class_name'] = f"GapGoSuccess{number}Strategy"
        info['strategy_explanation'] = f"Gap trading optimized for win rate with {condition}"
    elif 'spy' in filename:
        info['strategy_name'] = filename.replace('_', ' ').title()
        info['class_name'] = filename.replace('_', '').title() + 'Strategy'
    else:
        info['strategy_name'] = filename.replace('_', ' ').title()
        info['class_name'] = filename.replace('_', '').title() + 'Strategy'
    
    info['strategy_name_upper'] = info['strategy_name'].upper()
    
    if 'gap' in filename.lower():
        info['parameters'] = '''
        self.min_gap_percent = 0.5
        self.max_gap_percent = 5.0'''

                if curr_day['Low'] <= prev_day['Close']:
                    exit_price = prev_day['Close']
                else:
                    exit_price = curr_day['Close']
            else:  # Gap down - LONG

                entry_price = curr_day['Open']
                
                if curr_day['High'] >= prev_day['Close']:

                else:

    elif 'dip' in filename.lower():

        df['Volume_MA'] = df['Volume'].rolling(window=20).mean()
        df['Volume_Spike'] = df['Volume'] > (df['Volume_MA'] * 1.5)'''
    
    return info

def convert_strategy_to_standalone(source_path, target_path):
    """Convert a single strategy to standalone format"""
    try:
        with open(source_path, 'r', encoding='utf-8') as f:
            original_content = f.read()
        
        info = extract_strategy_info(source_path, original_content)
        
        template = get_standalone_template()
        
        standalone_content = template.format(**info)
        
        with open(target_path, 'w', encoding='utf-8') as f:
            f.write(standalone_content)
        
        return True
    except Exception as e:
        print(f"Error converting {source_path.name}: {e}")
        return False

def main():
    """Convert all strategies to standalone format"""
    print("=" * 60)
    print("MAKING ALL STRATEGIES STANDALONE")
    print("=" * 60)
    
    base_dir = Path(__file__).parent
    strategies_dir = base_dir / 'strategies'
    standalone_dir = base_dir / 'standalone_strategies'
    
    standalone_dir.mkdir(exist_ok=True)
    
    strategy_files = []
    for root, dirs, files in os.walk(strategies_dir):
        for file in files:
            if file.endswith('.py') and not file.startswith('__'):
                full_path = Path(root) / file
                strategy_files.append(full_path)
    
    print(f"Found {len(strategy_files)} strategy files to convert")
    
    converted = 0
    for source_path in strategy_files:
        relative_path = source_path.relative_to(strategies_dir)
        target_path = standalone_dir / relative_path
        target_path.parent.mkdir(parents=True, exist_ok=True)
        
        print(f"Converting: {relative_path}")
        
        if convert_strategy_to_standalone(source_path, target_path):
            converted += 1
    
    print(f"\n{'='*60}")
    print(f"Conversion Complete!")
    print(f"Converted {converted}/{len(strategy_files)} strategies")
    print(f"Standalone strategies saved in: {standalone_dir}")
    print(f"\nAll strategies can now be run independently with:")
    print(f"  python standalone_strategies/<category>/<strategy_name>.py")
    print(f"{'='*60}")

if __name__ == "__main__":
    main()
