    def _get_two_indicator_calculations(self, condition):
        """Get indicator calculations for two-indicator strategies"""
        indicators = """# Gap calculation
        df['Gap'] = df['Open'] - df['Close'].shift(1)
        df['Gap_Percent'] = (df['Gap'] / df['Close'].shift(1)) * 100"""
        
        condition_lower = condition.lower()
        
        if 'ma' in condition_lower:
            indicators += """
        
        df['MA_Fast'] = df['Close'].rolling(window=self.ma_fast).mean()
        df['MA_Slow'] = df['Close'].rolling(window=self.ma_slow).mean()
        df['MA_Cross'] = np.where(df['MA_Fast'] > df['MA_Slow'], 1, -1)
        df['MA_Cross_Signal'] = df['MA_Cross'].diff()"""
        
        if 'rsi' in condition_lower:
            indicators += """
        
        delta = df['Close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=self.rsi_period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=self.rsi_period).mean()
        rs = gain / loss
        df['RSI'] = 100 - (100 / (1 + rs))"""
        
        if 'atr' in condition_lower:
            indicators += """
        
        high_low = df['High'] - df['Low']
        high_close = np.abs(df['High'] - df['Close'].shift())
        low_close = np.abs(df['Low'] - df['Close'].shift())
        true_range = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
        df['ATR'] = true_range.rolling(window=self.atr_period).mean()
        df['ATR_Percent'] = (df['ATR'] / df['Close']) * 100"""
        
        if 'bb' in condition_lower:
            indicators += """
        
        df['BB_Middle'] = df['Close'].rolling(window=self.bb_period).mean()
        bb_std = df['Close'].rolling(window=self.bb_period).std()
        df['BB_Upper'] = df['BB_Middle'] + (bb_std * self.bb_std)
        df['BB_Lower'] = df['BB_Middle'] - (bb_std * self.bb_std)
        df['BB_Position'] = (df['Close'] - df['BB_Lower']) / (df['BB_Upper'] - df['BB_Lower'])"""
        
        if 'volume' in condition_lower:
            indicators += """
        
        df['Volume_MA'] = df['Volume'].rolling(window=20).mean()
        df['Volume_Ratio'] = df['Volume'] / df['Volume_MA']
        df['High_Volume'] = df['Volume_Ratio'] > self.volume_multiplier"""
        
        if 'macd' in condition_lower:
            indicators += """
        
        ema_fast = df['Close'].ewm(span=self.macd_fast, adjust=False).mean()
        ema_slow = df['Close'].ewm(span=self.macd_slow, adjust=False).mean()
        df['MACD'] = ema_fast - ema_slow
        df['MACD_Signal'] = df['MACD'].ewm(span=self.macd_signal, adjust=False).mean()
        df['MACD_Histogram'] = df['MACD'] - df['MACD_Signal']"""
        
        if 'stoch' in condition_lower:
            indicators += """
        
        low_min = df['Low'].rolling(window=self.stoch_period).min()
        high_max = df['High'].rolling(window=self.stoch_period).max()
        df['Stoch_K'] = 100 * ((df['Close'] - low_min) / (high_max - low_min))
        df['Stoch_D'] = df['Stoch_K'].rolling(window=self.stoch_smooth).mean()"""
        
        if 'red_days' in condition_lower:
            indicators += """
        
        df['IsRed'] = df['Close'] < df['Open']
        df['ConsecRed'] = df.groupby((df['IsRed'] != df['IsRed'].shift()).cumsum()).cumcount() + 1
        df['ConsecRed'] = np.where(df['IsRed'], df['ConsecRed'], 0)"""
        
        if 'engulfing' in condition_lower:
            indicators += """
        
        df['Bullish_Engulfing'] = ((df['Close'] > df['Open']) & 
                                   (df['Open'].shift(1) > df['Close'].shift(1)) &
                                   (df['Close'] > df['Open'].shift(1)) &
                                   (df['Open'] < df['Close'].shift(1)))"""
        
        if 'mean_reversion' in condition_lower:
            indicators += """
        
        df['Price_MA20'] = df['Close'].rolling(window=20).mean()
        df['Price_StdDev'] = df['Close'].rolling(window=20).std()
        df['Z_Score'] = (df['Close'] - df['Price_MA20']) / df['Price_StdDev']"""
        
        if 'vix' in condition_lower:
            indicators += """
        
        
        return indicators
    
    def _get_two_indicator_signals(self, condition):
        """Get signal logic for two-indicator strategies"""
        condition_lower = condition.lower()
        
        signal_logic = """# Two-indicator signal generation
        for i in range(30, len(df)):  # Ensure indicators are calculated
            curr_day = df.iloc[i]
            prev_day = df.iloc[i-1]
            
            gap = curr_day['Gap']
            if pd.isna(gap) or abs(gap) < self.min_gap_size:
                continue
            
            condition1_met = False
            condition2_met = False"""
        
        conditions = self._parse_two_conditions(condition)
        
        if conditions['condition1']:
            signal_logic += f"""
            
            signal_logic += self._get_condition_check(conditions['condition1'], 'condition1_met')
        
        if conditions['condition2']:
            signal_logic += f"""
            
            signal_logic += self._get_condition_check(conditions['condition2'], 'condition2_met')
        
        signal_logic += """
            
            if condition1_met and condition2_met:
                """
        
        if 'ma_5_10_cross' in condition_lower:
            signal_logic += """if pd.notna(curr_day.get('MA_Cross_Signal', None)) and curr_day['MA_Cross_Signal'] > 0:
                    direction = 'LONG'
                elif pd.notna(curr_day.get('MA_Cross_Signal', None)) and curr_day['MA_Cross_Signal'] < 0:
                    direction = 'SHORT'
                else:
                    direction = 'SHORT' if gap > 0 else 'LONG'  # Fade gap"""
        
        elif 'mean_reversion' in condition_lower:
            signal_logic += """direction = 'SHORT' if gap > 0 else 'LONG'  # Always fade"""
        
        elif 'trend' in condition_lower or 'momentum' in condition_lower:
            signal_logic += """direction = 'LONG' if gap > 0 else 'SHORT'  # Follow trend"""
        
        else:
            signal_logic += """direction = 'SHORT' if gap > 0 else 'LONG'  # Default fade"""
        
        signal_logic += """
                
                entry_price = curr_day['Open']
                
                if direction == 'LONG':
                    target = prev_day['Close'] if gap < 0 else entry_price + abs(gap) * 0.5
                    if curr_day['High'] >= target:
                        exit_price = target
                    else:
                        exit_price = curr_day['Close']
                else:  # SHORT
                    target = prev_day['Close'] if gap > 0 else entry_price - abs(gap) * 0.5

    def _parse_two_conditions(self, condition_text):
        """Parse two conditions from the strategy name"""
        condition_lower = condition_text.lower()

            parts = condition_lower.split(' and ')
            return {
                'condition1': parts[0].strip(),
                'condition2': parts[1].strip() if len(parts) > 1 else ''
            }
        
        patterns = {
            'ma_5_10_cross': 'MA 5/10 crossover',

    def _get_hf_signals(self, condition):
        """Signal logic for high frequency strategies"""
        signal_logic = """# High frequency signal generation
        daily_trades = {}  # Track trades per day
        
        for i in range(20, len(df)):
            curr_day = df.iloc[i]
            curr_date = curr_day['Date'].date()
            
            if curr_date in daily_trades and daily_trades[curr_date] >= self.max_trades_per_day:
                continue
            
            if abs(curr_day['Gap_Percent']) < self.min_gap_percent:
                continue
            
            signal = False"""
        
        condition_lower = condition.lower()
        
        if 'ma_5_10_cross' in condition_lower:
            signal_logic += """
            
            if pd.notna(curr_day.get('MA_Cross', 0)) and curr_day['MA_Cross'] != 0:
                signal = True
                direction = 'LONG' if curr_day['MA_Cross'] > 0 else 'SHORT'"""
        
        elif 'rsi_overbought' in condition_lower:
            signal_logic += """
            
            if pd.notna(curr_day.get('RSI', 0)) and curr_day['RSI'] > self.rsi_overbought:
                signal = True
                direction = 'SHORT'  # Fade overbought"""
        
        elif 'tuesday' in condition_lower:
            signal_logic += """
            
            if curr_day.get('Tuesday', False):
                signal = True
                direction = 'SHORT' if curr_day['Gap_Percent'] > 0 else 'LONG'"""
        
        elif 'volume_price_confirmation' in condition_lower:
            signal_logic += """
            
            if (pd.notna(curr_day.get('Volume_Spike', False)) and curr_day['Volume_Spike'] and
                pd.notna(curr_day.get('Momentum', 0)) and abs(curr_day['Momentum']) > 0.5):
                signal = True
                direction = 'LONG' if curr_day['Momentum'] > 0 else 'SHORT'"""
        
        elif 'mean_reversion' in condition_lower:
            signal_logic += """
            
            if pd.notna(curr_day.get('Price_Deviation', 0)) and abs(curr_day['Price_Deviation']) > 1:
                signal = True
                direction = 'SHORT' if curr_day['Price_Deviation'] > 0 else 'LONG'"""
        
        elif 'stoch_k_above_d' in condition_lower:
            signal_logic += """
            
            if (pd.notna(curr_day.get('Stoch_K', 0)) and pd.notna(curr_day.get('Stoch_D', 0)) and
                curr_day['Stoch_K'] > curr_day['Stoch_D']):
                signal = True
                direction = 'LONG'"""
        
        else:
            signal_logic += """
            
            signal = True
            direction = 'SHORT' if curr_day['Gap_Percent'] > 0 else 'LONG'"""
        
        signal_logic += """
            
            if signal:
                entry_price = curr_day['Open']
                
                gap_size = abs(curr_day['Gap_Percent'])
                
                if direction == 'LONG':
                    target = entry_price * (1 + gap_size * 0.005)  # 50% of gap
                    stop = entry_price * (1 - gap_size * 0.003)    # 30% stop
                    
                    if curr_day['High'] >= target:
                        exit_price = target
                    elif curr_day['Low'] <= stop:
                        exit_price = stop
                    else:
                        exit_price = curr_day['Close']
                else:  # SHORT
                    target = entry_price * (1 - gap_size * 0.005)  # 50% of gap
                    stop = entry_price * (1 + gap_size * 0.003)    # 30% stop
                    
                    if curr_day['Low'] <= target:
                        exit_price = target
                    elif curr_day['High'] >= stop:
                        exit_price = stop
                    else:
                        exit_price = curr_day['Close']
                
                trade_signal = {
                    'date': curr_day['Date'],
                    'direction': direction,
                    'entry_price': entry_price,
                    'exit_price': exit_price,
                    'gap_percent': curr_day['Gap_Percent'],
                    'signal_strength': 1.0
                }
                signals.append(trade_signal)
                
                if curr_date not in daily_trades:
                    daily_trades[curr_date] = 0
                daily_trades[curr_date] += 1"""
        
        return signal_logic
    
    def convert_file(self, source_path, target_folder):
        """Convert a single strategy file"""
        try:
            if 'gap_go_profit' in source_path.name or 'gap_go_success' in source_path.name:
                strategy_info = self.parse_gap_and_go_strategy(source_path)
                target_subfolder = 'gap_and_go'
            elif source_path.parent.name == 'TOP_STRATEGIES':
                strategy_info = self.parse_top_strategy(source_path)
                target_subfolder = 'gap_close_optimized'
            elif 'profit_' in source_path.name or 'success_' in source_path.name:
                strategy_info = self.parse_two_indicator_strategy(source_path)
                target_subfolder = 'two_indicator'
            elif source_path.name.startswith('hf_'):
                strategy_info = self.parse_high_frequency_strategy(source_path)
                target_subfolder = 'high_frequency'
            else:
                return False
            
            target_dir = self.target_base / target_folder / target_subfolder
            target_dir.mkdir(parents=True, exist_ok=True)
            
            clean_filename = f"{strategy_info['class_name'].lower()}.py"
            target_path = target_dir / clean_filename
            
            content = self.create_strategy_template(strategy_info)
            
            with open(target_path, 'w') as f:
                f.write(content)
            
            self.converted_count += 1
            print(f"✓ Converted: {source_path.name} -> {target_subfolder}/{clean_filename}")
            return True
            
        except Exception as e:
            error_msg = f"Error converting {source_path.name}: {str(e)}"
            self.errors.append(error_msg)
            print(f"✗ {error_msg}")
            return False
    
    def convert_all(self):
        """Convert all strategies in all folders"""
        print("\n" + "="*80)
        print("COMPREHENSIVE STRATEGY CONVERSION")
        print("="*80)
        
        conversions = [
            {
                'source': self.source_base / 'GAP_AND_GO_1000_STRATEGIES',
                'target': 'strategies',
                'pattern': 'gap_go_*.py'
            },
            {
                'source': self.source_base / 'GAP_CLOSE_OPTIMIZATIONS' / 'TOP_STRATEGIES',
                'target': 'strategies',
                'pattern': '*.py',
                'exclude': ['base_strategy.py', 'run_all_top_strategies.py', 'validate_data.py']
            },
            {
                'source': self.source_base / 'GAP_CLOSE_OPTIMIZATIONS' / 'TWO_INDICATOR_STRATEGIES',
                'target': 'strategies',
                'pattern': 'profit_*.py'
            },
            {
                'source': self.source_base / 'GAP_CLOSE_OPTIMIZATIONS' / 'TWO_INDICATOR_STRATEGIES',
                'target': 'strategies',
                'pattern': 'success_*.py'
            },
            {
                'source': self.source_base / 'GAP_CLOSE_OPTIMIZATIONS' / 'HIGH_FREQUENCY_TWO_INDICATORS',
                'target': 'strategies',
                'pattern': 'hf_*.py'
            }
        ]
        
        for conv in conversions:
            source_dir = conv['source']
            if not source_dir.exists():
                print(f"\n⚠ Source directory not found: {source_dir}")
                continue
            
            print(f"\nProcessing: {source_dir.relative_to(self.source_base)}")
            print("-" * 60)
            
            pattern = conv['pattern']
            exclude = conv.get('exclude', [])
            
            files = list(source_dir.glob(pattern))
            files = [f for f in files if f.name not in exclude]
            
            print(f"Found {len(files)} strategy files")
            
            for file_path in sorted(files):
                self.convert_file(file_path, conv['target'])
        
        print("\n" + "="*80)
        print("CONVERSION SUMMARY")
        print("="*80)
        print(f"Total files converted: {self.converted_count}")
        if self.errors:
            print(f"Errors encountered: {len(self.errors)}")
            for error in self.errors[:5]:  # Show first 5 errors
                print(f"  - {error}")
            if len(self.errors) > 5:
                print(f"  ... and {len(self.errors) - 5} more errors")
        else:
            print("✓ All conversions successful!")
        
        return self.converted_count
