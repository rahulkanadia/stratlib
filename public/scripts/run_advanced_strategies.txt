import pandas as pd
import numpy as np
from datetime import datetime

def load_data(file_name):
    """Loads the CSV file into a DataFrame."""
    df = pd.read_csv(file_name, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def calculate_rolling_average_drop(df, window=21):
    """Calculate rolling average of drops on red days - NO LOOK-AHEAD BIAS."""
    df['IsRedDay'] = df['Close'] < df['Open']
    df['DailyDrop'] = np.where(df['IsRedDay'], df['Open'] - df['Close'], 0.0)
    df['RollingAvgDrop'] = df['DailyDrop'].rolling(window=window).mean().shift(1)
    df['RollingAvgDrop'].fillna(0.0, inplace=True)
    return df

def identify_candle_patterns(df):
    """Identify candle patterns using ONLY PREVIOUS DAY'S data - NO LOOK-AHEAD BIAS."""
    df['Body'] = abs(df['Close'] - df['Open'])
    df['UpperWick'] = df['High'] - df[['Open', 'Close']].max(axis=1)
    df['LowerWick'] = df[['Open', 'Close']].min(axis=1) - df['Low']
    df['Range'] = df['High'] - df['Low']
    
    df['BodyRatio'] = np.where(df['Range'] > 0, df['Body'] / df['Range'], 0)
    df['LowerWickRatio'] = np.where(df['Range'] > 0, df['LowerWick'] / df['Range'], 0)
    
    df['IsBullish'] = df['Close'] > df['Open']
    
    df['Hammer_Prev'] = (
        (df['Body'].shift(1) < df['Range'].shift(1) * 0.3) &
        (df['LowerWick'].shift(1) > df['Body'].shift(1) * 2) &
        (df['UpperWick'].shift(1) < df['Body'].shift(1))
    )
    
    df['InvertedHammer_Prev'] = (
        (df['Body'].shift(1) < df['Range'].shift(1) * 0.3) &
        (df['UpperWick'].shift(1) > df['Body'].shift(1) * 2) &
        (df['LowerWick'].shift(1) < df['Body'].shift(1))
    )
    
    df['Doji_Prev'] = df['Body'].shift(1) < df['Range'].shift(1) * 0.1
    
    df['BullishEngulfing_Prev'] = (
        df['IsBullish'].shift(1) &  # Yesterday was bullish
        (~df['IsBullish'].shift(2).fillna(True)) &  # Day before yesterday was bearish
        (df['Open'].shift(1) < df['Close'].shift(2)) &  # Yesterday opened below day before close
        (df['Close'].shift(1) > df['Open'].shift(2))  # Yesterday closed above day before open
    )
    
    df['BullishPattern_Prev'] = (
        df['Hammer_Prev'] | 
        df['InvertedHammer_Prev'] | 
        df['Doji_Prev'] | 
        df['BullishEngulfing_Prev']
    )
    
    df['BullishPattern_Prev'].fillna(False, inplace=True)
    
    return df

def calculate_moving_averages(df):
    """Calculate various moving averages and trend indicators WITHOUT look-ahead bias."""
    df['MA10'] = df['Close'].rolling(window=10).mean().shift(1)
    df['MA20'] = df['Close'].rolling(window=20).mean().shift(1)
    df['MA50'] = df['Close'].rolling(window=50).mean().shift(1)
    
    df['OpenAboveMA10'] = df['Open'] > df['MA10']
    df['OpenAboveMA20'] = df['Open'] > df['MA20']
    df['OpenAboveMA50'] = df['Open'] > df['MA50']
    
    return df

def calculate_profit_loss_long(entry_price, exit_price, position_size):
    """Calculate profit/loss for long position."""
    if entry_price <= 0:
        return 0
    shares = position_size / entry_price
    return shares * (exit_price - entry_price)

def run_candle_pattern_strategy(df, initial_investment=16000):
    """Candle pattern DIP strategy - NO LOOK-AHEAD BIAS."""
    df = df.copy()
    df = calculate_rolling_average_drop(df)
    df = identify_candle_patterns(df)
    
    trades = []
    
    for i in range(len(df)):
        if i < 21:  # Need history for calculations
            continue
        
        current_day = df.iloc[i]
        
        avg_drop = current_day['RollingAvgDrop']  # Already shifted, uses past data
        has_pattern_prev = current_day['BullishPattern_Prev']  # Previous day's pattern
        
        if avg_drop <= 0 or pd.isna(has_pattern_prev) or not has_pattern_prev:
            continue
        
        open_price = current_day['Open']  # Known at market open
        desired_entry = open_price - avg_drop
        
        low_price = current_day['Low']
        if low_price <= desired_entry:
            entry_price = desired_entry
            exit_price = current_day['Close']  # Exit at close
            profit_loss = calculate_profit_loss_long(entry_price, exit_price, initial_investment)
            
            trades.append({
                'Date': current_day['Date'],
                'Entry': entry_price,
                'Exit': exit_price,
                'Profit/Loss': profit_loss
            })
    
    return trades

def run_ma_filter_strategy(df, initial_investment=16000, ma_period=20, position='above'):
    """MA filter DIP strategy - NO LOOK-AHEAD BIAS."""
    df = df.copy()
    df = calculate_rolling_average_drop(df)
    df = calculate_moving_averages(df)
    
    trades = []
    above_column = f'OpenAboveMA{ma_period}'
    
    for i in range(len(df)):
        if i < max(21, ma_period):  # Need history for calculations
            continue
        
        current_day = df.iloc[i]
        avg_drop = current_day['RollingAvgDrop']  # Already shifted, so it's yesterday's average
        
        if avg_drop <= 0:
            continue
        
        open_above_ma = current_day[above_column]
        if position == 'above' and not open_above_ma:
            continue
        elif position == 'below' and open_above_ma:
            continue
        
        open_price = current_day['Open']
        low_price = current_day['Low']
        close_price = current_day['Close']
        
        desired_entry = open_price - avg_drop
        
        if low_price <= desired_entry:
            entry_price = desired_entry
            exit_price = close_price
            profit_loss = calculate_profit_loss_long(entry_price, exit_price, initial_investment)
            
            trades.append({
                'Date': current_day['Date'],
                'Entry': entry_price,
                'Exit': exit_price,
                'Profit/Loss': profit_loss
            })
    
    return trades

def calculate_strategy_stats(trades):
    """Calculate strategy statistics."""
    if not trades:
        return {
            'total_trades': 0,
            'success_rate': 0,
            'total_profit': 0,
            'avg_profit': 0
        }
    
    df_trades = pd.DataFrame(trades)
    total_trades = len(trades)
    winning_trades = len(df_trades[df_trades['Profit/Loss'] > 0])
    success_rate = winning_trades / total_trades * 100
    total_profit = df_trades['Profit/Loss'].sum()
    avg_profit = df_trades['Profit/Loss'].mean()
    
    return {
        'total_trades': total_trades,
        'success_rate': success_rate,
        'total_profit': total_profit,
        'avg_profit': avg_profit
    }
