def load_data(file_name):
    """
    Loads the daily CSV file into a DataFrame, sorts it by date, and resets the index.
    """
    df = pd.read_csv(file_name, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def load_vix_data(vix_file):
    """
    Loads the VIX data from a CSV file into a DataFrame.
    """
    vix_df = pd.read_csv(vix_file, parse_dates=['DATE'], dayfirst=True)
    vix_df.rename(columns={'DATE': 'Date'}, inplace=True)
    vix_df['Date'] = pd.to_datetime(vix_df['Date'], dayfirst=True, errors='coerce')
    vix_df.dropna(subset=['Date'], inplace=True)
    vix_df.sort_values('Date', inplace=True)
    vix_df.reset_index(drop=True, inplace=True)
    return vix_df

def load_30min_data(file_name_30min):
    """
    Loads the 30-minute data CSV, parses Time as datetime, and creates a Date_only column.
    """
    df_30 = pd.read_csv(file_name_30min)
    df_30['Time'] = pd.to_datetime(df_30['Time'], format='%m/%d/%Y %H:%M', errors='coerce')
    df_30.dropna(subset=['Time'], inplace=True)
    df_30.sort_values('Time', inplace=True)
    df_30.reset_index(drop=True, inplace=True)
    df_30['Date_only'] = df_30['Time'].dt.date
    return df_30

def calculate_day_range(day):
    """
    Calculates the trading range of a day (High - Low).
    """
    return day['High'] - day['Low']

def calculate_gap(prev_close, curr_open):
    """
    Calculates the gap between the previous day's close and the current day's open.
    """
    return curr_open - prev_close

def is_gap_valid_inverse(gap, prev_range):
    """
    Inverse condition of a 'Gap & Go' extremes:
    - Gap must be at least $0.20
    - Gap must be BETWEEN 15% and 85% of the previous day's range
    """
    gap_abs = abs(gap)
    if gap_abs < 0.20 or prev_range == 0:
        return False
    gap_percentage = gap_abs / prev_range
    return 0.15 > gap_percentage or gap_percentage > 0.85

def determine_trade_direction(gap):
    """
    Determines the trade direction based on the gap:
    - Gap up: Short position
    - Gap down: Long position
    """
    return 'Short' if gap > 0 else 'Long'

def check_gap_closure(trade_direction, day_data, prev_close):
    """
    Checks if the gap is closed during the trading day and returns the exit price.
    day_data should have 'High', 'Low', 'Close' columns.
    """
    if trade_direction == 'Short':
        if day_data['Low'] <= prev_close:
            exit_price = prev_close
            gap_closed = True
        else:
            exit_price = day_data['Close']
            gap_closed = False
    else:  # Long position
        if day_data['High'] >= prev_close:
            exit_price = prev_close
            gap_closed = True
        else:
            exit_price = day_data['Close']
            gap_closed = False
    return exit_price, gap_closed

def calculate_profit_loss(trade_direction, entry_price, exit_price, position_size):
    """
    Calculates the profit or loss of a trade.
    """
    number_of_shares = position_size / entry_price
    if trade_direction == 'Long':
        profit_loss = number_of_shares * (exit_price - entry_price)
    else:  # Short position
        profit_loss = number_of_shares * (entry_price - exit_price)
    return profit_loss

def precompute_trades(df, vix_df, df_30, initial_investment):
    """
    Identifies potential trades (based on the 'inverse' gap condition) and then sets the entry price 
    to the close of the first 30-min candle. Also fetches the 30-min volume.
    """
    total_days = len(df)

    vix_open_dict = {}
    vix_dates = set(vix_df['Date'])
    min_vix_date = vix_df['Date'].min()
    
    for current_date in df['Date']:
        prev_date = current_date - pd.Timedelta(days=1)
        while prev_date not in vix_dates and prev_date > min_vix_date:
            prev_date -= pd.Timedelta(days=1)
        if prev_date in vix_dates:
            vix_open = vix_df.loc[vix_df['Date'] == prev_date, 'OPEN'].values[0]
            vix_open_dict[current_date] = vix_open
        else:
            vix_open_dict[current_date] = np.nan

    potential_trades = []
    for i_day in range(1, total_days):
        prev_day = df.iloc[i_day - 1]
        curr_day = df.iloc[i_day]
        current_date = curr_day['Date']
        vix_open = vix_open_dict.get(current_date, np.nan)

        if pd.isna(vix_open):
            continue

        prev_range = calculate_day_range(prev_day)
        if prev_range == 0:
            continue

        gap = calculate_gap(prev_day['Close'], curr_day['Open'])
        if not is_gap_valid_inverse(gap, prev_range):
            continue

        trade_direction = determine_trade_direction(gap)
        potential_trades.append({
            'Date': current_date,
            'Trade Direction': trade_direction,
            'Previous Close': prev_day['Close'],
            'High': curr_day['High'],
            'Low': curr_day['Low'],
            'Close': curr_day['Close'],
            'VIX_Open': vix_open
        })

    if not potential_trades:
        return pd.DataFrame()

    potential_trades_df = pd.DataFrame(potential_trades)
    potential_trades_df['Date_only'] = potential_trades_df['Date'].dt.date

    df_30_first_candle = df_30.groupby('Date_only').first().reset_index()

    trades_merged = potential_trades_df.merge(
        df_30_first_candle[['Date_only', 'Last', 'Volume']], 
        on='Date_only', how='left', suffixes=('', '_30min')
    )

    finalized_trades = []
    for _, row in trades_merged.iterrows():
        trade_direction = row['Trade Direction']
        entry_price = row['Last']    # from the 30-min candle
        exit_price, gap_closed = check_gap_closure(trade_direction, row, row['Previous Close'])
        profit_loss = calculate_profit_loss(trade_direction, entry_price, exit_price, initial_investment)
        finalized_trades.append({
            'Date': row['Date'],
            'Trade Direction': trade_direction,
            'Entry Price': entry_price,
            'Exit Price': exit_price,
            'Gap Closed': gap_closed,
            'VIX_Open': row['VIX_Open'],
            'Profit/Loss': profit_loss,
            'Volume_30min': row['Volume']
        })

    return pd.DataFrame(finalized_trades)

def find_best_volume_threshold(trades_df):
    """
    Finds the best single volume threshold for the first 30-minute candle, 
    now treating the threshold as a *lower bound* (we keep trades above that volume).
    """
    if trades_df.empty:
        print("No trades to evaluate.")
        return None, None

    volumes = trades_df['Volume_30min'].dropna()
    if volumes.empty:
        print("No 30-min volume data available.")
        return None, None

    percentiles = np.linspace(0, 1, 100)  # 100 candidate thresholds
    candidate_thresholds = volumes.quantile(percentiles)

    results = []
    for vol_threshold in candidate_thresholds:
        filtered = trades_df[trades_df['Volume_30min'] >= vol_threshold]
        total_profit = filtered['Profit/Loss'].sum() if not filtered.empty else 0
        results.append((vol_threshold, total_profit))

    results_df = pd.DataFrame(results, columns=['Volume_Threshold', 'Total_Profit'])
    results_df.sort_values('Total_Profit', ascending=False, inplace=True)
    results_df.reset_index(drop=True, inplace=True)

    best_threshold = results_df.iloc[0]['Volume_Threshold']
    best_profit = results_df.iloc[0]['Total_Profit']

    print("\n=== Volume Threshold Optimization (Inverse Gap / Lower Bound) ===")
    print(f"Best Volume Threshold: {best_threshold:.0f} with Total Profit: ${best_profit:,.2f}\n")

    return best_threshold, results_df
