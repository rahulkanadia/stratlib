def run_strategy_and_capture_results(script_path, strategy_name):
    """Run a strategy script and capture its key results."""
    try:
        original_dir = os.getcwd()
        script_dir = os.path.dirname(script_path)
        if script_dir:
            os.chdir(script_dir)
        
        spec = {}
        exec(open(script_path).read(), spec)
        
        os.chdir(original_dir)
        
        return f"{strategy_name}: Successfully executed"
    except Exception as e:
        os.chdir(original_dir)
        return f"{strategy_name}: Error - {str(e)}"
