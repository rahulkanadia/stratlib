import pandas as pd
import numpy as np
from pathlib import Path
import sys
import os
import warnings
warnings.filterwarnings('ignore')

project_root = current_file.parent.parent.parent

from utils.base_strategy_fixed import BaseStrategy

class TwoIndicatorSuccess04Strategy(BaseStrategy):
    """
    Two Indicator Success #04 Implementation
    
    Uses combination of Rsi Overbought 70 And Macd Bearish for better signal confirmation
    """
    
    def __init__(self, data_path=None):
        super().__init__(
            name="Two Indicator Success #04",
            description="Two indicators - Rsi Overbought 70 And Macd Bearish",
            required_data=['Date', 'Open', 'High', 'Low', 'Close', 'Volume']
        )
        
        self.min_gap_size = 0.20
        self.rsi_period = 14
        self.rsi_overbought = 70
        self.macd_fast = 12
        self.macd_slow = 26
        self.macd_signal = 9
        
        if data_path:
            self.load_data(data_path)
    
    def calculate_indicators(self):
        """Calculate required indicators"""
        df = self.data.copy()
        
        df['Gap'] = df['Open'] - df['Close'].shift(1)
        df['Gap_Percent'] = (df['Gap'] / df['Close'].shift(1)) * 100
        
        df['MA_Fast'] = df['Close'].rolling(window=self.ma_fast).mean()
        df['MA_Slow'] = df['Close'].rolling(window=self.ma_slow).mean()
        df['MA_Cross'] = np.where(df['MA_Fast'] > df['MA_Slow'], 1, -1)
        df['MA_Cross_Signal'] = df['MA_Cross'].diff()
        
        delta = df['Close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=self.rsi_period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=self.rsi_period).mean()
        rs = gain / loss
        df['RSI'] = 100 - (100 / (1 + rs))
        
        ema_fast = df['Close'].ewm(span=self.macd_fast, adjust=False).mean()
        ema_slow = df['Close'].ewm(span=self.macd_slow, adjust=False).mean()
        df['MACD'] = ema_fast - ema_slow
        df['MACD_Signal'] = df['MACD'].ewm(span=self.macd_signal, adjust=False).mean()
        df['MACD_Histogram'] = df['MACD'] - df['MACD_Signal']
        
        return df
    
    def calculate_signals(self):
        """Generate trading signals based on strategy rules"""
        if self.data is None:
            raise ValueError("No data loaded. Please load data first.")
        
        df = self.calculate_indicators()
        signals = []
        
        for i in range(30, len(df)):  # Ensure indicators are calculated
            curr_day = df.iloc[i]
            prev_day = df.iloc[i-1]
            
            gap = curr_day['Gap']
            if pd.isna(gap) or abs(gap) < self.min_gap_size:
                continue
            
            condition1_met = False
            condition2_met = False
            
            if pd.notna(curr_day.get('RSI', 0)) and curr_day['RSI'] > 70:
                condition1_met = True
            
            if pd.notna(curr_day.get('MACD_Histogram', 0)) and curr_day['MACD_Histogram'] < 0:
                condition2_met = True
            
            if condition1_met and condition2_met:
                direction = 'SHORT' if gap > 0 else 'LONG'  # Default fade
                
                entry_price = curr_day['Open']
                
                if direction == 'LONG':
                    target = prev_day['Close'] if gap < 0 else entry_price + abs(gap) * 0.5
                    if curr_day['High'] >= target:
                        exit_price = target
                    else:
                        exit_price = curr_day['Close']
                else:  # SHORT
                    target = prev_day['Close'] if gap > 0 else entry_price - abs(gap) * 0.5
                    if curr_day['Low'] <= target:
                        exit_price = target
                    else:
                        exit_price = curr_day['Close']
                
                signal = {
                    'date': curr_day['Date'],
                    'direction': direction,
                    'entry_price': entry_price,
                    'exit_price': exit_price,
                    'gap_size': gap,
                    'signal_strength': 1.0
                }
                signals.append(signal)
        
        self.signals = pd.DataFrame(signals)
        return self.signals
    
    def run_backtest(self, initial_capital=10000, position_size=1.0):
        """Run backtest simulation"""
        return super().run_backtest(initial_capital, position_size)
    
    def get_summary_metrics(self):
        """Get performance metrics with strategy-specific additions"""
        if self.performance_metrics is None:
            return None
        
        metrics = self.performance_metrics.copy()
        
        
        
        return metrics
