import pandas as pd
import numpy as np
import os
import sys

sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from gap_utils import *

class GapStrategy:
    """Base class for gap trading strategies"""
    
    def __init__(self, name, description):
        self.name = name
        self.description = description
        self.initial_investment = 16000
        
    def calculate_indicators(self, df):
        """Override this method in child classes to calculate specific indicators"""
        raise NotImplementedError("Must implement calculate_indicators method")
        
    def get_filter_condition(self, df):
        """Override this method in child classes to define entry conditions"""
        raise NotImplementedError("Must implement get_filter_condition method")
        
    def run_strategy(self, data_file):
        """Run the complete strategy and return statistics"""
        df = load_data(data_file)
        
        df = self.calculate_indicators(df)
        
        filter_condition = self.get_filter_condition(df)
        
        trades = self.analyze_trades(df, filter_condition)
        
        trades_df = pd.DataFrame(trades)
        stats = calculate_statistics(trades_df, self.name, self.initial_investment)
        
        return stats, trades_df
    
    def analyze_trades(self, df, filter_condition):
        """Analyze trades based on filter condition"""
        trades = []
        total_days = len(df)
        
        for i in range(1, total_days):
            prev_day = df.iloc[i - 1]
            curr_day = df.iloc[i]
            
            if not filter_condition.iloc[i]:
                continue
            
            gap = calculate_gap(prev_day['Close'], curr_day['Open'])
            if not is_gap_valid(gap):
                continue
            
            trade_direction = determine_trade_direction(gap)
            entry_price = curr_day['Open']
            exit_price, gap_closed = check_gap_closure(trade_direction, curr_day, prev_day['Close'])
            
            profit_loss = calculate_profit_loss(
                trade_direction, entry_price, exit_price, self.initial_investment
            )
            
            trades.append({
                'Date': curr_day['Date'],
                'Trade Direction': trade_direction,
                'Entry Price': entry_price,
                'Exit Price': exit_price,
                'Gap': gap,
                'Gap Closed': gap_closed,
                'Profit/Loss': profit_loss
            })
        
        return trades
    
    def print_results(self, stats, trades_df):
        """Print formatted results"""
        print(f"\n{'='*60}")
        print(f"{self.name} - Gap Trading Strategy")
        print(f"{'='*60}")
        print(f"Description: {self.description}")
        print(f"\nStrategy Performance:")
        print(f"Total Trades: {stats['Total Trades']}")
        print(f"Winning Trades: {stats['Winning Trades']}")
        print(f"Losing Trades: {stats['Losing Trades']}")
        print(f"Success Rate: {stats['Success Rate']:.2f}%")
        print(f"Total Profit: ${stats['Total Profit']:,.2f}")
        print(f"Average Profit per Trade: ${stats['Average Profit']:,.2f}")
        print(f"ROI: {stats['ROI']:.2f}%")
        print(f"Final Equity: ${stats['Final Equity']:,.2f}")
        
        if not trades_df.empty:
            print(f"\nBest Trade: ${trades_df['Profit/Loss'].max():,.2f}")
            print(f"Worst Trade: ${trades_df['Profit/Loss'].min():,.2f}")
            print(f"Average Win: ${stats['Average Win']:,.2f}")
            print(f"Average Loss: ${stats['Average Loss']:,.2f}")
        
        print(f"{'='*60}\n")
