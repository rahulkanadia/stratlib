"""
Calculate Sharpe Ratios for all Top Gap Trading Strategies
"""
import pandas as pd
import numpy as np
import glob
import os
import sys

def calculate_sharpe_ratio(returns, risk_free_rate=0.0):
    """
    Calculate Sharpe Ratio

    Args:
        returns: Series or array of returns
        risk_free_rate: Annual risk-free rate (default 0 for simplicity)

    Returns:
        Sharpe ratio (annualized)
    """
    if len(returns) == 0 or returns.std() == 0:
        return 0.0

    returns = np.array(returns)

    excess_returns = returns - risk_free_rate

    sharpe = excess_returns.mean() / returns.std()

    return sharpe


def process_strategy_file(file_path):
    """Process a single strategy file to calculate Sharpe ratio"""
    try:
        file_name = os.path.basename(file_path)
        strategy_name = file_name.replace('.py', '')

        import importlib.util
        spec = importlib.util.spec_from_file_location(strategy_name, file_path)
        module = importlib.util.module_from_spec(spec)

        from io import StringIO
        old_stdout = sys.stdout
        sys.stdout = StringIO()

        try:
            spec.loader.exec_module(module)

            if hasattr(module, 'results') and module.results is not None:
                trades = module.results

                returns = trades['profit'].values
                sharpe = calculate_sharpe_ratio(returns)

                return sharpe
            else:
                return None

        finally:
            sys.stdout = old_stdout

    except Exception as e:
        print(f"Error processing {file_path}: {str(e)}")
        return None


def calculate_sharpe_from_summary(csv_file):
    """Calculate approximate Sharpe ratios from summary statistics"""
    df = pd.read_csv(csv_file)

    sharpe_ratios = []

    for idx, row in df.iterrows():

        total_trades = row['Total Trades']
        success_rate = row['Success Rate'] / 100
        avg_win = row['Average Win']
        avg_loss = row['Average Loss']

        expected_return = (success_rate * avg_win) + ((1 - success_rate) * avg_loss)

        win_variance = success_rate * (avg_win - expected_return) ** 2
        loss_variance = (1 - success_rate) * (avg_loss - expected_return) ** 2
        std_dev = np.sqrt(win_variance + loss_variance)

        if std_dev > 0:
            sharpe = expected_return / std_dev
        else:
            sharpe = 0.0

        sharpe_ratios.append(round(sharpe, 3))

    df['Sharpe Ratio'] = sharpe_ratios

    return df


if __name__ == "__main__":
    print("Calculating Sharpe Ratios for all strategies...")
