"""
Moving Average Crossover Strategy

Classic trend-following strategy using two moving averages:
- LONG when fast MA crosses above slow MA (uptrend)
- SHORT when fast MA crosses below slow MA (downtrend)
- Exit on opposite signal

Perfect introduction to trend-following strategies!
"""

import pandas as pd
import numpy as np
from pathlib import Path
import sys
import os

current_file = Path(os.path.abspath(__file__))
project_root = current_file.parent.parent.parent
sys.path.insert(0, str(project_root))

from utils.base_strategy_fixed import BaseStrategy

class MovingAverageCrossStrategy(BaseStrategy):
    """
    Simple Moving Average Crossover Strategy
    """
    
    def __init__(self, data_path=None, fast_period=10, slow_period=30):
        super().__init__(
            name="MA Crossover",
            description="Trades based on moving average crossovers",
            required_data=['Date', 'Open', 'High', 'Low', 'Close']
        )
        
        self.fast_period = fast_period
        self.slow_period = slow_period
        
        if data_path:
            self.load_data(data_path)
    
    def calculate_indicators(self):
        """Calculate moving averages"""
        df = self.data.copy()
        
        df['MA_Fast'] = df['Close'].rolling(window=self.fast_period).mean()
        df['MA_Slow'] = df['Close'].rolling(window=self.slow_period).mean()
        
        df['MA_Diff'] = df['MA_Fast'] - df['MA_Slow']
        df['MA_Cross'] = np.sign(df['MA_Diff']).diff()
        
        return df
    
    def calculate_signals(self):
        """Find MA crossover signals"""
        if self.data is None:
            raise ValueError("No data loaded. Please load data first.")
        
        df = self.calculate_indicators()
        signals = []
        
        position = None  # Track current position
        
        for i in range(self.slow_period + 1, len(df)):
            curr_day = df.iloc[i]
            
            if pd.isna(curr_day['MA_Cross']):
                continue
            
            if curr_day['MA_Cross'] > 0:  # Bullish crossover
                if position == 'SHORT':
                    signal = {
                        'date': curr_day['Date'],
                        'direction': 'CLOSE_SHORT',
                        'entry_price': curr_day['Open'],
                        'exit_price': curr_day['Open'],  # Exit at open
                        'signal_type': 'exit',
                        'signal_strength': 0.5
                    }
                    signals.append(signal)
                
                signal = {
                    'date': curr_day['Date'],
                    'direction': 'LONG',
                    'entry_price': curr_day['Open'],
                    'exit_price': None,  # Will be set on exit
                    'ma_fast': curr_day['MA_Fast'],
                    'ma_slow': curr_day['MA_Slow'],
                    'signal_type': 'entry',
                    'signal_strength': 1.0
                }
                signals.append(signal)
                position = 'LONG'
                
            elif curr_day['MA_Cross'] < 0:  # Bearish crossover
                if position == 'LONG':
                    signal = {
                        'date': curr_day['Date'],
                        'direction': 'CLOSE_LONG',
                        'entry_price': curr_day['Open'],
                        'exit_price': curr_day['Open'],  # Exit at open
                        'signal_type': 'exit',
                        'signal_strength': 0.5
                    }
                    signals.append(signal)
                
                signal = {
                    'date': curr_day['Date'],
                    'direction': 'SHORT',
                    'entry_price': curr_day['Open'],
                    'exit_price': None,  # Will be set on exit
                    'ma_fast': curr_day['MA_Fast'],
                    'ma_slow': curr_day['MA_Slow'],
                    'signal_type': 'entry',
                    'signal_strength': 1.0
                }
                signals.append(signal)
                position = 'SHORT'
        
        processed_signals = []
        open_position = None
        
        for signal in signals:
            if signal['signal_type'] == 'entry' and open_position is None:
                open_position = signal
            elif signal['signal_type'] == 'exit' and open_position is not None:
                trade_signal = {
                    'date': open_position['date'],
                    'direction': open_position['direction'],
                    'entry_price': open_position['entry_price'],
                    'exit_price': signal['exit_price'],
                    'exit_date': signal['date'],
                    'signal_strength': 1.0
                }
                processed_signals.append(trade_signal)
                open_position = None
        
        self.signals = pd.DataFrame(processed_signals)
        return self.signals

def main():
    """Example usage"""
    print("=" * 60)
    print("MOVING AVERAGE CROSSOVER STRATEGY")
    print("=" * 60)
    print(f"\nTrades when {10}-day MA crosses {30}-day MA")
    print("Classic trend-following approach")
    print("=" * 60)
    
    data_path = Path(__file__).parent.parent.parent / "data" / "7J0OK1VR.csv"
    
    strategy = MovingAverageCrossStrategy(data_path, fast_period=10, slow_period=30)
    
    print("\nCalculating moving averages...")
    signals = strategy.calculate_signals()
    print(f"Found {len(signals)} crossover trades")
    
    print("\nRunning backtest...")
    strategy.run_backtest(initial_capital=10000)
    
    metrics = strategy.get_summary_metrics()
    
    print("\n" + "=" * 60)
    print("RESULTS")
    print("=" * 60)
    print(f"Total Trades: {metrics['total_trades']}")
    print(f"Win Rate: {metrics['win_rate']:.1f}%")
    print(f"Total Profit: ${metrics['total_profit']:.2f}")
    print(f"Max Drawdown: {metrics['max_drawdown']:.1f}%")

if __name__ == "__main__":
    main()
