import pandas as pd
import numpy as np

def load_data(file_name):
    """Loads the CSV file into a DataFrame."""
    df = pd.read_csv(file_name, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def calculate_rolling_average_drop(df, window=21):
    """Calculate rolling average of drops on red days."""
    df['IsRedDay'] = df['Close'] < df['Open']
    df['DailyDrop'] = np.where(df['IsRedDay'], df['Open'] - df['Close'], 0.0)
    df['RollingAvgDrop'] = df['DailyDrop'].rolling(window=window).mean().shift(1)
    df['RollingAvgDrop'].fillna(0.0, inplace=True)
    return df

def calculate_profit_loss_long(entry_price, exit_price, position_size):
    """Calculate profit/loss for long position."""
    if entry_price <= 0:
        return 0
    shares = position_size / entry_price
    return shares * (exit_price - entry_price)

def analyze_dip_strategy_with_multiplier(df, initial_investment=16000, rolling_window=21, multiplier=1.0):
    """Implement the DIP strategy with a multiplier for the average drop."""
    df = df.copy()
    df = calculate_rolling_average_drop(df, window=rolling_window)
    
    trades = []
    
    for i in range(len(df)):
        if i == 0:
            continue
        
        current_day = df.iloc[i]
        avg_drop = current_day['RollingAvgDrop']
        
        if avg_drop <= 0:
            continue
        
        open_price = current_day['Open']
        low_price = current_day['Low']
        close_price = current_day['Close']
        
        desired_entry = open_price - (avg_drop * multiplier)
        
        if low_price <= desired_entry:
            entry_price = desired_entry
            exit_price = close_price
            profit_loss = calculate_profit_loss_long(entry_price, exit_price, initial_investment)
            
            trades.append({
                'Date': current_day['Date'],
                'Entry Price': entry_price,
                'Exit Price': exit_price,
                'Profit/Loss': profit_loss
            })
    
    return pd.DataFrame(trades)

def optimize_multiplier(file_name, initial_investment=16000):
    """Optimize the multiplier for best success rate."""
    df = load_data(file_name)
    
    best_multiplier = None
    best_success_rate = 0
    results = []
    
    for multiplier in np.arange(0.5, 2.1, 0.1):
        trades_df = analyze_dip_strategy_with_multiplier(df, initial_investment, 21, multiplier)
        
        if len(trades_df) >= 100:  # Only consider if we have at least 100 trades
            total_trades = len(trades_df)
            winning_trades = len(trades_df[trades_df['Profit/Loss'] > 0])
            success_rate = winning_trades / total_trades if total_trades > 0 else 0
            total_profit = trades_df['Profit/Loss'].sum()
            
            results.append({
                'Multiplier': round(multiplier, 1),
                'Total Trades': total_trades,
                'Success Rate': success_rate,
                'Total Profit': total_profit
            })
            
            if success_rate > best_success_rate:
                best_success_rate = success_rate
                best_multiplier = multiplier
    
    results_df = pd.DataFrame(results)
    print("=== Multiplier Optimization Results ===")
    print(results_df.sort_values('Success Rate', ascending=False).head(10))
    
    print(f"\nBest Multiplier: {best_multiplier:.1f}")
    print(f"Best Success Rate: {best_success_rate:.2%}")
    
    if best_multiplier:
        final_trades = analyze_dip_strategy_with_multiplier(df, initial_investment, 21, best_multiplier)
        print(f"\nFinal Results with Multiplier={best_multiplier:.1f}:")
        print(f"Total Trades: {len(final_trades)}")
        print(f"Total Profit: ${final_trades['Profit/Loss'].sum():,.2f}")
    
    return best_multiplier, best_success_rate
