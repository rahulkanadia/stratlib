"""
Dip Buying Strategy Analysis Engine
====================================
Comprehensive backtesting framework for intraday dip-buying strategies on SPY.

Core Thesis:
    On days when SPY drops intraday, the price tends to recover toward the close.
    By buying at a calculated dip level (Open - HistoricalAvgDrop), we can capture
    the intraday recovery for a same-day profit.

Strategy Variants Tested:
    1. Basic Dip (baseline) — buy every dip, no filters
    2. VIX-Filtered — only trade in elevated volatility (VIX > threshold)
    3. RSI-Filtered — only trade when RSI signals oversold
    4. Bollinger Band-Filtered — only trade near lower Bollinger Band
    5. Consecutive Red Days — only trade after N+ consecutive down days
    6. Day-of-Week — isolate which weekday performs best
    7. Combined Best Filters — stack the strongest signals together

Key Results (2009-2024, $16K capital):
    - Basic Strategy: 2,830 trades | 56.0% win rate | $24,809 profit | 155% ROI
    - 2+ Consec Red Days: 559 trades | 58.1% win rate | $11,586 profit | Sharpe 2.39
    - Monday Only: 500 trades | 57.4% win rate | $5,360 profit
    - Combined (Mon+VIX>15+RedPrior): 173 trades | 59.0% win rate | Sharpe 2.31
    - Basic strategy profitable in 14 out of 16 years

Look-Ahead Bias: ELIMINATED
    All rolling averages use .shift(1) to ensure only historical data informs
    entry decisions. No future data leaks into the signal.

Data Requirements:
    - SPY daily OHLCV CSV: columns [Date, Open, High, Low, Close, Volume]
    - VIX daily CSV: columns [DATE, OPEN, HIGH, LOW, CLOSE] (optional, for VIX strategies)

Usage:
    python dip_analysis_engine.py --spy <path_to_spy.csv> [--vix <path_to_vix.csv>]

Author: Daniel Peer
"""

import pandas as pd
import numpy as np
import argparse
import sys
import os



def load_spy_data(file_path):
    """Load SPY OHLCV data from CSV.

    Expected columns: Date, Open, High, Low, Close, Volume
    Date format: YYYY-MM-DD
    """
    df = pd.read_csv(file_path, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df


def load_vix_data(file_path):
    """Load VIX daily data from CSV.

    Handles the DD/MM/YYYY date format commonly found in VIX history exports.
    Returns only Date and VIX_Open columns for merging with SPY data.
    """
    vix = pd.read_csv(file_path)
    vix.rename(columns={'DATE': 'Date', 'OPEN': 'VIX_Open', 'CLOSE': 'VIX_Close'}, inplace=True)
    vix['Date'] = pd.to_datetime(vix['Date'], dayfirst=True, errors='coerce')
    vix.dropna(subset=['Date'], inplace=True)
    vix.sort_values('Date', inplace=True)
    vix.reset_index(drop=True, inplace=True)
    return vix[['Date', 'VIX_Open']]



def add_rolling_avg_drop(df, window=21):
    """
    Calculate the rolling average of intraday drops on red days.

    A "red day" is when Close < Open. The drop magnitude is (Open - Close).
    The rolling average is shifted by 1 day to prevent look-ahead bias,
    meaning today's trading decision only uses data up to yesterday.

    Args:
        df: DataFrame with OHLCV columns
        window: Number of trading days for the rolling window (default 21 ~ 1 month)

    Returns:
        DataFrame with added columns: is_red_day, daily_drop, avg_drop
    """
    df = df.copy()
    df['is_red_day'] = df['Close'] < df['Open']
    df['daily_drop'] = np.where(df['is_red_day'], df['Open'] - df['Close'], 0.0)
    df['avg_drop'] = df['daily_drop'].rolling(window=window).mean().shift(1)
    df['avg_drop'] = df['avg_drop'].fillna(0.0)
    return df


def add_rsi(df, period=14):
    """Calculate Relative Strength Index (RSI).

    Standard Wilder smoothing method. RSI < 30 is typically considered
    oversold, > 70 overbought.

    Args:
        df: DataFrame with Close column
        period: RSI lookback period (default 14)

    Returns:
        DataFrame with added 'rsi' column
    """
    df = df.copy()
    delta = df['Close'].diff()
    gain = np.where(delta > 0, delta, 0.0)
    loss = np.where(delta < 0, -delta, 0.0)
    avg_gain = pd.Series(gain, index=df.index).rolling(window=period).mean()
    avg_loss = pd.Series(loss, index=df.index).rolling(window=period).mean()
    rs = avg_gain / avg_loss.replace(0, np.nan)
    df['rsi'] = 100 - (100 / (1 + rs))
    return df


def add_bollinger_bands(df, period=20, std_dev=2):
    """Calculate Bollinger Bands and %B position indicator.

    %B measures where the close price sits within the bands:
        0.0 = at the lower band
        0.5 = at the middle band
        1.0 = at the upper band
    Values below 0.2 suggest the price is near the lower band (potential dip).

    Args:
        df: DataFrame with Close column
        period: Moving average period (default 20)
        std_dev: Number of standard deviations for band width (default 2)

    Returns:
        DataFrame with added columns: bb_mid, bb_lower, bb_upper, bb_pct
    """
    df = df.copy()
    df['bb_mid'] = df['Close'].rolling(window=period).mean()
    bb_std = df['Close'].rolling(window=period).std()
    df['bb_upper'] = df['bb_mid'] + (std_dev * bb_std)
    df['bb_lower'] = df['bb_mid'] - (std_dev * bb_std)
    band_width = df['bb_upper'] - df['bb_lower']
    df['bb_pct'] = np.where(band_width > 0,
                            (df['Close'] - df['bb_lower']) / band_width,
                            0.5)
    return df


def add_consecutive_red_days(df):
    """Count consecutive red days preceding the current day.

    Used to identify multi-day selloffs where a dip-buy has higher
    probability of success (mean reversion after sustained selling).

    Returns:
        DataFrame with 'prev_consec_red' column (shifted by 1 to prevent look-ahead)
    """
    df = df.copy()
    df['is_red_day'] = df['Close'] < df['Open']
    consecutive = []
    count = 0
    for red in df['is_red_day']:
        if red:
            count += 1
        else:
            count = 0
        consecutive.append(count)
    df['consec_red'] = consecutive
    df['prev_consec_red'] = df['consec_red'].shift(1).fillna(0).astype(int)
    return df



def run_backtest(df, filter_func=None, initial_capital=16000, label="Strategy"):
    """
    Run dip-buying backtest on prepared DataFrame.

    Entry Logic:
        - Calculate desired entry = Open - avg_drop (historical average drop)
        - If the day's Low reaches this level, we assume a fill at desired_entry
        - This models a limit buy order placed at market open

    Exit Logic:
        - Sell at market Close the same day (intraday round-trip)

    Args:
        df: DataFrame with OHLCV + avg_drop + indicator columns
        filter_func: Optional callable(row) -> bool. Only trade when True.
        initial_capital: Dollar amount per trade position
        label: Human-readable name for this strategy

    Returns:
        dict with:
            - Summary metrics (trades, win_rate, total_pnl, sharpe, etc.)
            - trades_df: DataFrame with every individual trade
    """
    trades = []

    for i in range(1, len(df)):
        row = df.iloc[i]
        avg_drop = row['avg_drop']

        if avg_drop <= 0:
            continue

        if filter_func and not filter_func(row):
            continue

        desired_entry = row['Open'] - avg_drop

        if row['Low'] <= desired_entry:
            entry_price = desired_entry
            exit_price = row['Close']
            shares = initial_capital / entry_price
            pnl = shares * (exit_price - entry_price)

            trades.append({
                'date': row['Date'],
                'entry': round(entry_price, 2),
                'exit': round(exit_price, 2),
                'pnl': round(pnl, 2),
                'pct_return': round((exit_price - entry_price) / entry_price * 100, 3)
            })

    trades_df = pd.DataFrame(trades)

    if len(trades_df) == 0:
        return {
            'label': label, 'trades': 0, 'wins': 0, 'win_rate': 0,
            'total_pnl': 0, 'avg_pnl': 0, 'max_win': 0, 'max_loss': 0,
            'roi_pct': 0, 'sharpe': 0, 'trades_df': trades_df
        }

    wins = len(trades_df[trades_df['pnl'] > 0])
    total = len(trades_df)
    total_pnl = trades_df['pnl'].sum()
    avg_pnl = trades_df['pnl'].mean()
    std_pnl = trades_df['pnl'].std()
    sharpe = (avg_pnl / std_pnl * np.sqrt(252)) if std_pnl > 0 else 0

    return {
        'label': label,
        'trades': total,
        'wins': wins,
        'win_rate': round(wins / total * 100, 2),
        'total_pnl': round(total_pnl, 2),
        'avg_pnl': round(avg_pnl, 2),
        'max_win': round(trades_df['pnl'].max(), 2),
        'max_loss': round(trades_df['pnl'].min(), 2),
        'roi_pct': round(total_pnl / initial_capital * 100, 2),
        'sharpe': round(sharpe, 2),
        'trades_df': trades_df
    }



def strategy_basic(df, capital=16000):
    """Baseline: buy every dip with no additional filter."""
    return run_backtest(df, filter_func=None, initial_capital=capital,
                        label="Basic Dip (No Filter)")


def strategy_vix_above(df, capital=16000, threshold=20):
    """Only buy dips when VIX is above threshold (elevated market fear)."""
    return run_backtest(df,
                        filter_func=lambda r: r.get('VIX_Open', 0) > threshold,
                        initial_capital=capital,
                        label=f"VIX > {threshold}")


def strategy_rsi_below(df, capital=16000, threshold=40):
    """Only buy dips when RSI indicates oversold conditions."""
    return run_backtest(df,
                        filter_func=lambda r: r.get('rsi', 50) < threshold,
                        initial_capital=capital,
                        label=f"RSI < {threshold}")


def strategy_bb_lower(df, capital=16000, threshold=0.2):
    """Only buy dips when price is near the lower Bollinger Band."""
    return run_backtest(df,
                        filter_func=lambda r: r.get('bb_pct', 0.5) < threshold,
                        initial_capital=capital,
                        label=f"BB %B < {threshold}")


def strategy_consecutive_red(df, capital=16000, min_days=2):
    """Only buy dips after N+ consecutive red (down) days."""
    return run_backtest(df,
                        filter_func=lambda r: r.get('prev_consec_red', 0) >= min_days,
                        initial_capital=capital,
                        label=f"{min_days}+ Consec Red Days")


def strategy_day_of_week(df, capital=16000, day=0):
    """Only trade on a specific weekday (0=Monday through 4=Friday)."""
    day_names = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday']
    return run_backtest(df,
                        filter_func=lambda r: r['Date'].dayofweek == day,
                        initial_capital=capital,
                        label=f"{day_names[day]} Only")


def strategy_combined_best(df, capital=16000):
    """Stack the strongest filters: Monday + VIX>15 + at least 1 prior red day."""
    def combo_filter(r):
        is_monday = r['Date'].dayofweek == 0
        vix_ok = r.get('VIX_Open', 0) > 15
        red_prior = r.get('prev_consec_red', 0) >= 1
        return is_monday and vix_ok and red_prior
    return run_backtest(df, filter_func=combo_filter, initial_capital=capital,
                        label="Combined: Mon + VIX>15 + RedPrior")



def yearly_breakdown(trades_df):
    """Generate year-by-year performance breakdown."""
    if len(trades_df) == 0:
        return pd.DataFrame()

    t = trades_df.copy()
    t['year'] = t['date'].dt.year

    yearly = t.groupby('year').agg(
        trades=('pnl', 'count'),
        wins=('pnl', lambda x: (x > 0).sum()),
        total_pnl=('pnl', 'sum'),
        avg_pnl=('pnl', 'mean'),
        max_win=('pnl', 'max'),
        max_loss=('pnl', 'min'),
    ).round(2)

    yearly['win_rate'] = (yearly['wins'] / yearly['trades'] * 100).round(1)
    return yearly[['trades', 'wins', 'win_rate', 'total_pnl', 'avg_pnl', 'max_win', 'max_loss']]


def print_results_table(results):
    """Print formatted results summary table."""
    print(f"\n{'Strategy':<35} {'Trades':>7} {'Win%':>7} {'Total P&L':>12} "
          f"{'Avg P&L':>10} {'ROI%':>8} {'Sharpe':>7}")
    print("-" * 90)

    for r in sorted(results, key=lambda x: x['win_rate'], reverse=True):
        if r['trades'] >= 20:
            print(f"{r['label']:<35} {r['trades']:>7} {r['win_rate']:>6.1f}% "
                  f"${r['total_pnl']:>10,.2f} ${r['avg_pnl']:>8,.2f} "
                  f"{r['roi_pct']:>7.1f}% {r['sharpe']:>6.2f}")


def print_risk_analysis(basic_result):
    """Print detailed risk metrics for the basic strategy."""
    bt = basic_result['trades_df']
    if len(bt) == 0:
        return

    print(f"\n  Total Trades:     {len(bt)}")
    print(f"  Win Rate:         {basic_result['win_rate']}%")
    print(f"  Total Profit:     ${basic_result['total_pnl']:,.2f}")
    print(f"  Best Single Day:  ${bt['pnl'].max():,.2f}")
    print(f"  Worst Single Day: ${bt['pnl'].min():,.2f}")
    print(f"  Avg Win:          ${bt[bt['pnl']>0]['pnl'].mean():,.2f}")
    print(f"  Avg Loss:         ${bt[bt['pnl']<0]['pnl'].mean():,.2f}")

    losses = (bt['pnl'] < 0).astype(int)
    max_streak = 0
    current = 0
    for val in losses:
        current = current + 1 if val else 0
        max_streak = max(max_streak, current)
    print(f"  Max Loss Streak:  {max_streak} consecutive losses")

    cum_pnl = bt['pnl'].cumsum()
    max_dd = (cum_pnl - cum_pnl.cummax()).min()
    print(f"  Max Cum. Drawdown: ${max_dd:,.2f}")



def main():
    parser = argparse.ArgumentParser(description='Dip Buying Strategy Analysis Engine')
    parser.add_argument('--spy', type=str, help='Path to SPY daily OHLCV CSV')
    parser.add_argument('--vix', type=str, help='Path to VIX daily CSV (optional)')
    parser.add_argument('--capital', type=float, default=16000, help='Capital per trade (default: $16,000)')
    args = parser.parse_args()

    print("=" * 80)
    print("  DIP BUYING STRATEGY ANALYSIS ENGINE")
    print("  SPY Intraday Dip Recovery — Comprehensive Backtest")
    print("=" * 80)

    script_dir = os.path.dirname(os.path.abspath(__file__))
    spy_file = args.spy or os.path.join(script_dir, '../../data/sample/spy_daily.csv')
    vix_file = args.vix

    if not os.path.exists(spy_file):
        print(f"\n[ERROR] SPY data file not found: {spy_file}")
        print("  Usage: python dip_analysis_engine.py --spy <path_to_spy.csv> [--vix <path_to_vix.csv>]")
        sys.exit(1)

    print(f"\n[DATA] Loading SPY data from {spy_file}...")
    df = load_spy_data(spy_file)
    print(f"  Loaded {len(df)} trading days: {df['Date'].min().date()} to {df['Date'].max().date()}")

    has_vix = False
    if vix_file and os.path.exists(vix_file):
        vix = load_vix_data(vix_file)
        df = df.merge(vix, on='Date', how='left')
        df['VIX_Open'] = df['VIX_Open'].ffill()
        has_vix = True
        print(f"  VIX data merged ({len(vix)} records)")

    print("\n[INDICATORS] Computing technical indicators...")
    df = add_rolling_avg_drop(df, window=21)
    df = add_rsi(df, period=14)
    df = add_bollinger_bands(df, period=20, std_dev=2)
    df = add_consecutive_red_days(df)
    print("  Rolling Avg Drop (21-day), RSI(14), Bollinger Bands(20,2), Consecutive Red Days")

    capital = args.capital

    print("\n" + "=" * 80)
    print("  STRATEGY RESULTS")
    print("=" * 80)

    results = []

    results.append(strategy_basic(df, capital))

    if has_vix:
        for thresh in [15, 20, 25, 30]:
            results.append(strategy_vix_above(df, capital, thresh))

    for thresh in [30, 35, 40, 45]:
        results.append(strategy_rsi_below(df, capital, thresh))

    for thresh in [0.1, 0.2, 0.3]:
        results.append(strategy_bb_lower(df, capital, thresh))

    for n in [1, 2, 3, 4]:
        results.append(strategy_consecutive_red(df, capital, n))

    for day in range(5):
        results.append(strategy_day_of_week(df, capital, day))

    if has_vix:
        results.append(strategy_combined_best(df, capital))

    print_results_table(results)

    print("\n" + "=" * 80)
    print("  TOP STRATEGIES — YEAR-BY-YEAR BREAKDOWN")
    print("=" * 80)

    top3 = sorted([r for r in results if r['trades'] >= 50],
                  key=lambda x: x['win_rate'], reverse=True)[:3]

    for r in top3:
        print(f"\n--- {r['label']} ---")
        print(f"  Overall: {r['trades']} trades | {r['win_rate']}% win rate | "
              f"${r['total_pnl']:,.2f} profit | Sharpe {r['sharpe']}")
        yearly = yearly_breakdown(r['trades_df'])
        if len(yearly) > 0:
            print(yearly.to_string())

    print("\n" + "=" * 80)
    print("  CONSISTENCY CHECK")
    print("=" * 80)

    basic = results[0]
    basic_yearly = yearly_breakdown(basic['trades_df'])
    if len(basic_yearly) > 0:
        profitable = (basic_yearly['total_pnl'] > 0).sum()
        total_yrs = len(basic_yearly)
        print(f"\n  Basic Strategy: Profitable in {profitable}/{total_yrs} years")
        losing = basic_yearly[basic_yearly['total_pnl'] <= 0]
        if len(losing) > 0:
            print(f"  Losing years: {list(losing.index)}")
        else:
            print("  NO LOSING YEARS — profitable every single year!")

    print("\n" + "=" * 80)
    print("  RISK ANALYSIS — BASIC STRATEGY")
    print("=" * 80)
    print_risk_analysis(basic)

    print("\n" + "=" * 80)
    print("  ANALYSIS COMPLETE")
    print("=" * 80)

    return results


if __name__ == "__main__":
    main()
