class VolumePriceConfirmationAndMeanReversionHighFrequencyStrategy:
    """
    High-Frequency Two-Indicator Strategy: Volume_Price_Confirmation + Mean_Reversion
    Success Rate: 77.24% | Total Profit: $1,199.01 | Trades: 123
    
    This HIGH-FREQUENCY strategy combines:
    1. Volume_Price_Confirmation
    2. Mean_Reversion
    
    Minimum Trade Requirement: 100+ trades (ACHIEVED: 123 trades)
    """
    
    def __init__(self):
        self.name = "Volume_Price_Confirmation + Mean_Reversion (High-Frequency)"
        self.indicator1 = "Volume_Price_Confirmation"
        self.indicator2 = "Mean_Reversion"
        self.initial_investment = 16000
        self.min_trades_required = 100
        
    def calculate_indicators(self, df):
        """Calculate required indicators"""
        
        df['Volume_SMA'] = df['Volume'].rolling(window=20).mean().shift(1)
        df['Volume_Ratio'] = (df['Volume'] / df['Volume_SMA']).shift(1)
        df['OBV'] = calculate_obv(df)
        
        return df
    
    def get_filter_conditions(self, df):
        """Get the two indicator conditions"""
        condition1 = ((df['Close'].shift(1) > df['Close'].shift(2)) & (df['Volume'].shift(1) > df['Volume'].shift(2)))
        
        condition2 = (df['Close'].shift(1) < df['SMA_20'] * 0.98)
        
        return condition1 & condition2
    
    def run_strategy(self, data_file):
        """Run the complete high-frequency strategy"""
        df = load_data(data_file)
        
        df = self.calculate_indicators(df)
        
        filter_condition = self.get_filter_conditions(df)
        
        trades = []
        for i in range(1, len(df)):
            if not filter_condition.iloc[i]:
                continue
                
            prev_day = df.iloc[i - 1]
            curr_day = df.iloc[i]
            
            gap = calculate_gap(prev_day['Close'], curr_day['Open'])
            if not is_gap_valid(gap):
                continue
            
            trade_direction = determine_trade_direction(gap)
            entry_price = curr_day['Open']
            exit_price, gap_closed = check_gap_closure(trade_direction, curr_day, prev_day['Close'])
            
            profit_loss = calculate_profit_loss(
                trade_direction, entry_price, exit_price, self.initial_investment
            )
            
            trades.append({
                'Date': curr_day['Date'],
                'Trade Direction': trade_direction,
                'Entry Price': entry_price,
                'Exit Price': exit_price,
                'Gap': gap,
                'Gap Closed': gap_closed,
                'Profit/Loss': profit_loss
            })
        
        trades_df = pd.DataFrame(trades)
        
        if len(trades_df) < self.min_trades_required:
            print(f"WARNING: Strategy only generated {len(trades_df)} trades, below high-frequency threshold of {self.min_trades_required}")
        
        stats = calculate_statistics(trades_df, self.name, self.initial_investment)
        
        return stats, trades_df
    
    def print_results(self, stats, trades_df):
        """Print formatted results"""
        print(f"\\n{'='*70}")
        print(f"{self.name} - High-Frequency Gap Trading Strategy")
        print(f"{'='*70}")
        print(f"Indicators: {self.indicator1} + {self.indicator2}")
        print(f"High-Frequency Classification: {len(trades_df)} trades (requirement: {self.min_trades_required}+)")
        print(f"\\nStrategy Performance:")
        print(f"Total Trades: {stats['Total Trades']}")
        print(f"Success Rate: {stats['Success Rate']:.2f}%")
        print(f"Total Profit: ${stats['Total Profit']:,.2f}")
        print(f"Average Profit per Trade: ${stats['Total Profit']/max(1, stats['Total Trades']):.2f}")
        print(f"ROI: {stats['ROI']:.2f}%")
        print(f"Final Equity: ${stats['Final Equity']:,.2f}")
        
        if len(trades_df) > 0:
            profitable_trades = len(trades_df[trades_df['Profit/Loss'] > 0])
            avg_win = trades_df[trades_df['Profit/Loss'] > 0]['Profit/Loss'].mean() if profitable_trades > 0 else 0
            avg_loss = trades_df[trades_df['Profit/Loss'] < 0]['Profit/Loss'].mean() if len(trades_df) - profitable_trades > 0 else 0
            max_win = trades_df['Profit/Loss'].max()
            max_loss = trades_df['Profit/Loss'].min()
            
            print(f"\\nHigh-Frequency Analytics:")
            print(f"Trades per Year: {len(trades_df) / 5:.1f}")  # 5-year period
            print(f"Average Win: ${avg_win:.2f}")
            print(f"Average Loss: ${avg_loss:.2f}")
            print(f"Largest Win: ${max_win:.2f}")
            print(f"Largest Loss: ${max_loss:.2f}")
            print(f"Win/Loss Ratio: {abs(avg_win/avg_loss):.2f if avg_loss != 0 else 'N/A'}")
        
        print(f"{'='*70}\\n")

def main():
    """Run the high-frequency strategy"""
    csv_files = [f for f in os.listdir('.') if f.endswith('.csv') 
                and not f.startswith('high_frequency')]
    
    if csv_files:
        data_file = csv_files[0]
    else:
        data_file = '/mnt/c/Users/danie/OneDrive/Desktop/claude/2020 to 2024/7J0OK1VR.csv'
        if not os.path.exists(data_file):
            print("Error: No CSV data file found.")
            return
    
    print(f"Using data file: {data_file}")
    
    strategy = VolumePriceConfirmationAndMeanReversionHighFrequencyStrategy()
    
    try:
        stats, trades_df = strategy.run_strategy(data_file)
        strategy.print_results(stats, trades_df)
        
        output_file = f"{strategy.name.replace(' ', '_').replace('(', '').replace(')', '')}_trades.csv"
        trades_df.to_csv(output_file, index=False)
        print(f"Trade details saved to: {output_file}")
        
    except Exception as e:
        print(f"Error running high-frequency strategy: {e}")

if __name__ == "__main__":
    main()
