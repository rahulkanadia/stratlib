    def calculate_bollinger_bands(self, prices, period=20, std_dev=2):
        """
        Calculate Bollinger Bands
        
        Parameters:
        -----------
        prices : pd.Series
            Series of closing prices
        period : int
            Moving average period (default: 20)
        std_dev : float
            Number of standard deviations (default: 2)
            
        Returns:
        --------
        tuple : (middle_band, upper_band, lower_band)
            Bollinger Band values as pandas Series
        """
        middle_band = prices.rolling(window=period).mean()
        
        std = prices.rolling(window=period).std()
        
        upper_band = middle_band + (std_dev * std)
        lower_band = middle_band - (std_dev * std)
        
        return middle_band, upper_band, lower_band
    
    def should_trade_gap(self, gap_data, current_data):
        """
        Determine if gap should be traded based on Bollinger Band conditions
        
        Parameters:
        -----------
        gap_data : dict
            Gap information including date, size, prices
        current_data : pd.DataFrame
            Stock data including Bollinger Band values
            
        Returns:
        --------
        tuple : (bool, str)
            Whether to trade and reason
        """
        gap_date = gap_data['date']
        
        gap_idx = current_data.index.get_loc(gap_date)
        if gap_idx == 0:
            return False, "No previous data"
        
        prev_data = current_data.iloc[gap_idx - 1]
        prev_close = prev_data['Close']
        prev_bb_upper = prev_data['BB_Upper']
        
        if pd.isna(prev_bb_upper):
            return False, "Bollinger Bands not available"
        
        if prev_close > prev_bb_upper:
            distance = prev_close - prev_bb_upper
            pct_above = (distance / prev_close) * 100
            return True, f"Close ${prev_close:.2f} > BB Upper ${prev_bb_upper:.2f} (${distance:.2f} / {pct_above:.2f}% above)"
        else:
            distance = prev_bb_upper - prev_close
            pct_below = (distance / prev_close) * 100
            return False, f"Close ${prev_close:.2f} <= BB Upper ${prev_bb_upper:.2f} (${distance:.2f} / {pct_below:.2f}% below)"
    
    def execute_trade(self, gap_data, current_data):
        """
        Execute a gap trading position
        
        Parameters:
        -----------
        gap_data : dict
            Gap information
        current_data : pd.DataFrame
            Stock data for the gap day
            
        Returns:
        --------
        dict : Trade result information
        """
        gap_date = gap_data['date']
        gap_size = gap_data['gap_size']
        prev_close = gap_data['prev_close']
        gap_open = gap_data['gap_open']
        
        position_size = self.current_balance * self.position_size_pct
        shares = int(position_size / gap_open)
        
        if shares == 0:
            return None
        
        day_data = current_data.loc[gap_date]
        high = day_data['High']
        low = day_data['Low']
        close = day_data['Close']
        
        gap_idx = current_data.index.get_loc(gap_date)
        prev_data = current_data.iloc[gap_idx - 1]
        
        gap_closed = False
        exit_price = close
        
        if gap_size > 0:  # Gap up
            if low <= prev_close:
                gap_closed = True
                exit_price = prev_close
        else:  # Gap down
            if high >= prev_close:
                gap_closed = True
                exit_price = prev_close
        
        if gap_size > 0:  # Gap up - we short
            entry_price = gap_open
            pnl = (entry_price - exit_price) * shares
        else:  # Gap down - we long
            entry_price = gap_open
            pnl = (exit_price - entry_price) * shares
        
        self.current_balance += pnl
        
        self.total_trades += 1
        if pnl > 0:
            self.winning_trades += 1
            self.consecutive_wins += 1
            self.consecutive_losses = 0
            self.max_consecutive_wins = max(self.max_consecutive_wins, self.consecutive_wins)
        else:
            self.losing_trades += 1
            self.consecutive_losses += 1
            self.consecutive_wins = 0
            self.max_consecutive_losses = max(self.max_consecutive_losses, self.consecutive_losses)
        
        trade = {
            'date': gap_date,
            'type': 'SHORT' if gap_size > 0 else 'LONG',
            'gap_size': gap_size,
            'gap_percent': (gap_size / prev_close) * 100,
            'entry_price': entry_price,
            'exit_price': exit_price,
            'shares': shares,
            'position_value': shares * entry_price,
            'pnl': pnl,
            'pnl_percent': (pnl / (shares * entry_price)) * 100,
            'gap_closed': gap_closed,
            'balance': self.current_balance,
            'high': high,
            'low': low,
            'close': close,
            'prev_close': prev_close,
            'prev_bb_upper': prev_data['BB_Upper'],
            'prev_bb_middle': prev_data['BB_Middle'],
            'prev_bb_lower': prev_data['BB_Lower'],
            'bb_position': (prev_close - prev_data['BB_Lower']) / (prev_data['BB_Upper'] - prev_data['BB_Lower'])
        }
        
        self.trades.append(trade)
        return trade

    def run_backtest(self, symbol='AAPL', start_date='2020-01-01', end_date='2024-01-01'):
        """
        Run the Price Above BB Upper strategy backtest
        
        Parameters:
        -----------
        symbol : str
            Stock symbol to trade
        start_date : str
            Start date for backtest
        end_date : str
            End date for backtest
        """
        print(f"\nLoading data for {symbol}...")
        
        stock_data = load_stock_data(symbol, start_date, end_date)
        if stock_data.empty:
            print("Failed to load stock data!")
            return
        
        print("Calculating Bollinger Bands...")
        bb_middle, bb_upper, bb_lower = self.calculate_bollinger_bands(
            stock_data['Close'], 
            period=self.bb_period, 
            std_dev=self.bb_std
        )
        
        stock_data['BB_Middle'] = bb_middle
        stock_data['BB_Upper'] = bb_upper
        stock_data['BB_Lower'] = bb_lower
        
        print(f"Detecting gaps >= ${self.min_gap_size}...")
        gaps = detect_gaps(stock_data, min_gap_size=self.min_gap_size)
        print(f"Found {len(gaps)} total gaps")
        
        tradable_gaps = 0
        print("\n" + "="*80)
        print("STARTING ELITE BACKTEST - HUNTING FOR 2-SIGMA BREAKOUTS")
        print("="*80)
        
        for gap in gaps:
            should_trade, reason = self.should_trade_gap(gap, stock_data)
            
            if should_trade:
                tradable_gaps += 1
                trade = self.execute_trade(gap, stock_data)
                if trade:
                    self.print_trade_details(trade)
            
            self.daily_balances.append({
                'date': gap['date'],
                'balance': self.current_balance
            })
        
        self.print_summary(tradable_gaps, len(gaps))
