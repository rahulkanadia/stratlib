def get_indicator_calculations(indicator1, indicator2):
    """Generate calculation code based on indicators"""
    calculations = []
    indicators = [indicator1, indicator2]
    
    needs_sma = any('MA' in ind and 'EMA' not in ind for ind in indicators)
    needs_ema = any('EMA' in ind for ind in indicators)
    needs_rsi = any('RSI' in ind for ind in indicators)
    needs_bb = any('BB' in ind or 'Bollinger' in ind for ind in indicators)
    needs_atr = any('ATR' in ind or 'Range' in ind for ind in indicators)
    needs_macd = any('MACD' in ind for ind in indicators)
    needs_stoch = any('Stoch' in ind for ind in indicators)
    needs_volume = any('Volume' in ind or 'OBV' in ind for ind in indicators)
    needs_patterns = any('Red' in ind or 'Green' in ind or 'Engulfing' in ind for ind in indicators)
    
    if needs_sma:
        calculations.append("""
        for period in [5, 10, 20, 50, 100, 200]:
            df[f'SMA_{period}'] = calculate_sma(df, period)""")
    
    if needs_ema:
        calculations.append("""
        for period in [5, 10, 20, 50]:
            df[f'EMA_{period}'] = calculate_ema(df, period)""")
    
    if needs_rsi:
        calculations.append("""
        df['RSI'] = calculate_rsi(df, 14)
        df['RSI_5'] = calculate_rsi(df, 5)
        df['RSI_21'] = calculate_rsi(df, 21)""")
    
    if needs_bb:
        calculations.append("""
        upper_bb, middle_bb, lower_bb = calculate_bollinger_bands(df)
        df['BB_Upper'] = upper_bb
        df['BB_Middle'] = middle_bb
        df['BB_Lower'] = lower_bb""")
    
    if needs_atr:
        calculations.append("""
        df['ATR'] = calculate_atr(df, 14)
        df['High_Low_Range'] = (df['High'] - df['Low']).shift(1)""")
    
    if needs_macd:
        calculations.append("""
        macd_line, signal_line = calculate_macd(df)
        df['MACD'] = macd_line
        df['MACD_Signal'] = signal_line""")
    
    if needs_stoch:
        calculations.append("""
        k_percent, d_percent = calculate_stochastic(df)
        df['Stoch_K'] = k_percent
        df['Stoch_D'] = d_percent""")
    
    if needs_volume:
        calculations.append("""
        df['Volume_SMA'] = df['Volume'].rolling(window=20).mean().shift(1)
        df['Volume_Ratio'] = (df['Volume'] / df['Volume_SMA']).shift(1)
        df['OBV'] = calculate_obv(df)""")
    
    if needs_patterns:
        calculations.append("""
        df['PrevDayRed'] = (df['Close'] < df['Open']).shift(1).fillna(False)
        df['Prev2DaysRed'] = (df['PrevDayRed'] & df['PrevDayRed'].shift(1)).fillna(False)
        df['Prev3DaysRed'] = (df['Prev2DaysRed'] & df['PrevDayRed'].shift(2)).fillna(False)""")
    
    if any(day in ind for day in ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday'] for ind in indicators):
        calculations.append("""
        df['DayOfWeek'] = pd.to_datetime(df['Date']).dt.dayofweek""")
    
    return '\\n'.join(calculations) if calculations else "# No additional calculations needed"

def get_condition_code(indicator):
    """Get the condition code for a specific indicator"""
    conditions_map = {
        "MA5_Above": "df['Close'].shift(1) > df['SMA_5']",
        "MA10_Above": "df['Close'].shift(1) > df['SMA_10']",
        "MA20_Above": "df['Close'].shift(1) > df['SMA_20']",
        "MA50_Above": "df['Close'].shift(1) > df['SMA_50']",
        "MA100_Above": "df['Close'].shift(1) > df['SMA_100']",
        "MA200_Above": "df['Close'].shift(1) > df['SMA_200']",
        "MA_5_10_Cross": "df['SMA_5'] > df['SMA_10']",
        "MA_10_20_Cross": "df['SMA_10'] > df['SMA_20']",
        "MA_20_50_Cross": "df['SMA_20'] > df['SMA_50']",
        "MA_50_200_Cross": "df['SMA_50'] > df['SMA_200']",
        "EMA5_Above": "df['Close'].shift(1) > df['EMA_5']",
        "EMA10_Above": "df['Close'].shift(1) > df['EMA_10']",
        "EMA20_Above": "df['Close'].shift(1) > df['EMA_20']",
        "EMA50_Above": "df['Close'].shift(1) > df['EMA_50']",
        "RSI_Oversold_30": "df['RSI'] < 30",
        "RSI_Oversold_35": "df['RSI'] < 35",
        "RSI_Overbought_70": "df['RSI'] > 70",
        "RSI_Overbought_65": "df['RSI'] > 65",
        "Price_Below_BB_Lower": "df['Close'].shift(1) < df['BB_Lower']",
        "Price_Above_BB_Upper": "df['Close'].shift(1) > df['BB_Upper']",
        "High_Volume": "df['Volume_Ratio'] > 1.5",
        "Volume_Spike": "df['Volume_Ratio'] > 2.0",
        "Monday": "df['DayOfWeek'] == 0",
        "Tuesday": "df['DayOfWeek'] == 1",
        "Wednesday": "df['DayOfWeek'] == 2",
        "Thursday": "df['DayOfWeek'] == 3",
        "Friday": "df['DayOfWeek'] == 4",
        "After_Red_Day": "df['PrevDayRed']",
        "After_2_Red_Days": "df['Prev2DaysRed']",
        "After_3_Red_Days": "df['Prev3DaysRed']",
        "MACD_Cross_Down": "(df['MACD'] < df['MACD_Signal']) & (df['MACD'].shift(1) >= df['MACD_Signal'].shift(1))",
        "Stoch_Oversold": "df['Stoch_K'] < 20",
        "Stoch_Overbought": "df['Stoch_K'] > 80",
    }
    
    return conditions_map.get(indicator, "pd.Series([True]*len(df))")
