def load_stock_data(file_name):
    """
    Loads the stock CSV file into a DataFrame, parses the 'Date' column, sorts it, and resets the index.
    """
    df = pd.read_csv(file_name, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def load_vix_data(vix_file):
    """
    Loads the VIX data from a CSV file into a DataFrame, parses dates, sorts it, and resets the index.
    """
    vix_df = pd.read_csv(vix_file)
    
    if 'DATE' in vix_df.columns:
        vix_df.rename(columns={'DATE': 'Date'}, inplace=True)
    elif 'Date' not in vix_df.columns:
        raise ValueError("VIX CSV must contain a 'Date' or 'DATE' column.")
    
    vix_df['Date'] = pd.to_datetime(vix_df['Date'], dayfirst=True, errors='coerce')
    
    vix_df.dropna(subset=['Date'], inplace=True)
    
    vix_df.sort_values('Date', inplace=True)
    
    vix_df.reset_index(drop=True, inplace=True)
    
    return vix_df

def load_cpci_data(cpci_file_name):
    """
    Loads the CPCI CSV file into a DataFrame, parses the 'Date' column, sorts it, and resets the index.
    """
    cpci_df = pd.read_csv(cpci_file_name)
    if 'foll' in cpci_df.columns:
        cpci_df.rename(columns={'foll': 'Date'}, inplace=True)  # Adjust column name as needed
    elif 'Date' not in cpci_df.columns:
        raise ValueError("CPCI CSV must contain a 'foll' or 'Date' column.")
    
    cpci_df['Date'] = pd.to_datetime(cpci_df['Date'], format='%m/%d/%Y', errors='coerce')  # Adjust format as needed
    cpci_df.dropna(subset=['Date'], inplace=True)
    cpci_df.sort_values('Date', inplace=True)
    cpci_df.reset_index(drop=True, inplace=True)
    return cpci_df

def calculate_day_range(day):
    """
    Calculates the trading range of a day (High - Low).
    """
    return day['High'] - day['Low']

def calculate_gap(prev_close, curr_open):
    """
    Calculates the gap between the previous day's close and the current day's open.
    """
    return curr_open - prev_close

def is_gap_valid(gap, prev_range):
    """
    Checks if the gap meets the specified conditions:
    - At least $0.20
    - Between 15% and 85% of the previous day's range
    """
    gap_abs = abs(gap)
    if gap_abs < 0.20 or prev_range == 0:
        return False
    gap_percentage = gap_abs / prev_range
    return 0.15 <= gap_percentage <= 0.85

def determine_trade_direction(gap, reverse=False):
    """
    Determines the trade direction based on the gap and whether to reverse the standard interpretation.
    - Standard Interpretation (reverse=False):
        - Gap up: Short position
        - Gap down: Long position
    - Reversed Interpretation (reverse=True):
        - Gap up: Long position
        - Gap down: Short position
    """
    if reverse:
        return 'Long' if gap > 0 else 'Short'
    else:
        return 'Short' if gap > 0 else 'Long'

def check_gap_closure(trade_direction, curr_day, prev_close):
    """
    Checks if the gap is closed during the trading day and returns the exit price.
    """
    if trade_direction == 'Short':
        if curr_day['Low'] <= prev_close:
            exit_price = prev_close
            gap_closed = True
        else:
            exit_price = curr_day['Close']
            gap_closed = False
    else:  # Long position
        if curr_day['High'] >= prev_close:
            exit_price = prev_close
            gap_closed = True
        else:
            exit_price = curr_day['Close']
            gap_closed = False
    return exit_price, gap_closed

def calculate_profit_loss(trade_direction, entry_price, exit_price, position_size):
    """
    Calculates the profit or loss of a trade.
    """
    number_of_shares = position_size / entry_price

    if trade_direction == 'Long':
        profit_loss = number_of_shares * (exit_price - entry_price)
    else:  # Short position
        profit_loss = number_of_shares * (entry_price - exit_price)

    return profit_loss

def precompute_trades(df, vix_df, cpci_df, initial_investment):
    """
    Precomputes all valid trades based on gap criteria, VIX, and CPCI conditions.
    Returns a DataFrame containing all trades.
    """
    trades = []
    total_days = len(df)

    df = df.merge(cpci_df[['Date', 'Last']], on='Date', how='left', suffixes=('', '_CPCI'))
    df.rename(columns={'Last': 'CPCI'}, inplace=True)

    vix_open_dict = {}
    vix_dates = set(vix_df['Date'])
    min_vix_date = vix_df['Date'].min()

    for current_date in df['Date']:
        prev_date = current_date - pd.Timedelta(days=1)
        while prev_date not in vix_dates and prev_date > min_vix_date:
            prev_date -= pd.Timedelta(days=1)
        if prev_date in vix_dates:
            vix_open = vix_df.loc[vix_df['Date'] == prev_date, 'OPEN'].values[0]
            vix_open_dict[current_date] = vix_open
        else:
            vix_open_dict[current_date] = np.nan  # No VIX data for the previous day

    for i_day in range(1, total_days):
        prev_day = df.iloc[i_day - 1]
        curr_day = df.iloc[i_day]
        current_date = curr_day['Date']
        vix_open = vix_open_dict.get(current_date, np.nan)
        cpci_value = curr_day['CPCI']

        if pd.isna(vix_open) or pd.isna(cpci_value):
            continue  # Skip if no VIX or CPCI data

        prev_range = calculate_day_range(prev_day)
        if prev_range == 0:
            continue  # Avoid division by zero

        gap = calculate_gap(prev_day['Close'], curr_day['Open'])
        if not is_gap_valid(gap, prev_range):
            continue

        trade_direction = determine_trade_direction(gap)
        entry_price = curr_day['Open']
        exit_price, gap_closed = check_gap_closure(trade_direction, curr_day, prev_day['Close'])
        profit_loss = calculate_profit_loss(trade_direction, entry_price, exit_price, initial_investment)

        trades.append({
            'Date': current_date,
            'Trade Direction': trade_direction,
            'Entry Price': entry_price,
            'Exit Price': exit_price,
            'Gap Closed': gap_closed,
            'VIX_Open': vix_open,
            'CPCI': cpci_value,
            'Profit/Loss': profit_loss
        })

    trades_df = pd.DataFrame(trades)
    return trades_df


def find_best_vix_conditions_optimized(trades_df, i_min=10.0, i_max=30.0, j_min=10.0, j_max=30.0, step=0.1):
    """
    Optimized function to find the best (i, j) VIX conditions that maximize profit.
    Utilizes vectorized operations for efficiency.
    Returns the best (i, j) pair and a DataFrame of all results.
    """
    i_values = np.round(np.arange(i_min, j_max + step, step), 1)
    j_values = np.round(np.arange(j_min, j_max + step, step), 1)

    i_grid, j_grid = np.meshgrid(i_values, j_values)
    i_flat = i_grid.flatten()
    j_flat = j_grid.flatten()
    valid_pairs_mask = i_flat < j_flat
    i_valid = i_flat[valid_pairs_mask]
    j_valid = j_flat[valid_pairs_mask]

    total_profits = np.zeros(len(i_valid))

    vix_open_array = trades_df['VIX_Open'].values
    profit_loss_array = trades_df['Profit/Loss'].values

    sorted_indices = np.argsort(vix_open_array)
    sorted_vix = vix_open_array[sorted_indices]
    sorted_profits = profit_loss_array[sorted_indices]

    cumulative_profits = np.cumsum(sorted_profits)

    def compute_profit(i, j):
        left = np.searchsorted(sorted_vix, i, side='right')
        right = np.searchsorted(sorted_vix, j, side='left')
        if right > 0:
            return cumulative_profits[right - 1] - (cumulative_profits[left - 1] if left > 0 else 0)
        else:
            return 0.0

    vectorized_compute_profit = np.vectorize(compute_profit)
    total_profits = vectorized_compute_profit(i_valid, j_valid)

    results_df = pd.DataFrame({
        'i': i_valid,
        'j': j_valid,
        'Total Profit': total_profits
    })

    best_idx = results_df['Total Profit'].idxmax()
    best_i = results_df.loc[best_idx, 'i']
    best_j = results_df.loc[best_idx, 'j']
    best_profit = results_df.loc[best_idx, 'Total Profit']

    results_df_sorted = results_df.sort_values(by='Total Profit', ascending=False).reset_index(drop=True)

    print("\n=== VIX Optimization Complete ===")
    print(f"Best VIX Conditions: i={best_i}, j={best_j} with Total Profit: ${best_profit:,.2f}\n")

    print("Top 10 VIX Conditions by Total Profit:")
    print(results_df_sorted.head(10).to_string(index=False))

    return best_i, best_j, results_df_sorted


def calculate_atr(df, n):
    """
    Calculates the Average True Range (ATR) over 'n' consecutive days.
    Ensures that 'n' is an integer.
    """
    n = int(n)  # Ensure n is an integer
    df['Previous Close'] = df['Close'].shift(1)
    df['High-Low'] = df['High'] - df['Low']
    df['High-PrevClose'] = abs(df['High'] - df['Previous Close'])
    df['Low-PrevClose'] = abs(df['Low'] - df['Previous Close'])
    df['TR'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=n).mean()
    return df

def compute_atr_total_profit(args):
    """
    Computes the total profit for a given combination of n and T.
    This function must be defined at the top level to be picklable by multiprocessing.
    """
    trades_df, df, n, T, initial_investment = args
    df_with_atr = calculate_atr(df.copy(), n)
    trades_with_atr = trades_df.merge(df_with_atr[['Date', 'ATR']], on='Date', how='left')
    trades_with_atr = trades_with_atr.dropna(subset=['ATR'])
    trades_filtered = trades_with_atr[trades_with_atr['ATR'] <= T]
    total_profit = trades_filtered['Profit/Loss'].sum() if not trades_filtered.empty else 0
    return (n, T, total_profit)

def optimize_atr_parameters(trades_df, df, initial_investment, n_values, T_values):
    """
    Optimizes both ATR period 'n' and ATR threshold 'T' to maximize total profit.
    """
    param_combinations = list(product(n_values, T_values))

    args_list = [(trades_df.copy(), df.copy(), n, T, initial_investment) for n, T in param_combinations]

    with Pool(processes=cpu_count()) as pool:
        results = pool.map(compute_atr_total_profit, args_list)

    results_df = pd.DataFrame(results, columns=['n', 'T', 'Total Profit'])

    if not results_df.empty:
        best_result = results_df.loc[results_df['Total Profit'].idxmax()]
        best_n = int(best_result['n'])  # Ensure n is an integer
        best_T = best_result['T']
        best_total_profit_atr = best_result['Total Profit']
    else:
        best_n, best_T, best_total_profit_atr = None, None, 0

    print("\n=== ATR Optimization Complete ===")
    if best_n is not None:
        print(f"Best ATR Parameters: n={best_n}, T={best_T} with Total Profit: ${best_total_profit_atr:,.2f}\n")
    else:
        print("No valid ATR parameter combinations found.\n")

    top_results = results_df.sort_values(by='Total Profit', ascending=False).reset_index(drop=True)
    print("Top 10 ATR Parameter Combinations by Total Profit:")
    if not top_results.empty:
        print(top_results.head(10).to_string(index=False))
    else:
        print("No ATR parameter combinations yielded any profit.")

    return best_n, best_T, top_results


def find_best_cpci_reject_intervals(trades_df, max_intervals=5, step=0.01):
    """
    Finds multiple CPCI intervals to reject for both Long and Short trades to maximize total profit.
    Uses a greedy approach to iteratively find and reject the best CPCI intervals for each trade direction.
    
    Parameters:
    - trades_df: DataFrame containing trades with 'CPCI', 'Trade Direction', and 'Profit/Loss' columns.
    - max_intervals: Maximum number of CPCI intervals to reject per trade direction.
    - step: Step size for CPCI range discretization.
    
    Returns:
    - reject_intervals: Dictionary with 'Long' and 'Short' as keys and lists of (CPCI_Min, CPCI_Max) tuples as values.
    - total_profit: Total profit after rejecting the specified intervals.
    - all_reject_results: Dictionary with 'Long' and 'Short' as keys and lists of DataFrames containing optimization results at each step.
    """
    reject_intervals = {'Long': [], 'Short': []}
    all_reject_results = {'Long': [], 'Short': []}
    remaining_trades = trades_df.copy()
    total_profit = remaining_trades['Profit/Loss'].sum()

    for direction in ['Long', 'Short']:
        for interval_num in range(max_intervals):
            direction_trades = remaining_trades[remaining_trades['Trade Direction'] == direction]
            if direction_trades.empty:
                print(f"No more {direction} trades to optimize.")
                break

            cpci_min_value = direction_trades['CPCI'].min()
            cpci_max_value = direction_trades['CPCI'].max()

            if cpci_min_value == cpci_max_value:
                print(f"No further variation in CPCI for {direction} trades at interval {interval_num + 1}.")
                break

            cpci_values = np.round(np.arange(cpci_min_value, cpci_max_value + step, step), 2)
            cpci_pairs = [(i, j) for i in cpci_values for j in cpci_values if i < j]

            cpci_array = direction_trades['CPCI'].values
            profit_loss = direction_trades['Profit/Loss'].values

            results = []

            cpci_array_broadcasted = cpci_array[:, np.newaxis]
            chunk_size = 100000  # Adjust based on available memory
            num_pairs = len(cpci_pairs)
            for start in range(0, num_pairs, chunk_size):
                end = start + chunk_size
                chunk = cpci_pairs[start:end]
                i_vals, j_vals = zip(*chunk)
                i_vals = np.array(i_vals)
                j_vals = np.array(j_vals)
                mask = (cpci_array_broadcasted > i_vals) & (cpci_array_broadcasted < j_vals)
                total_profits = (mask * profit_loss[:, np.newaxis]).sum(axis=0)
                chunk_results = pd.DataFrame({
                    'CPCI_Min': i_vals,
                    'CPCI_Max': j_vals,
                    'Total Profit': total_profits
                })
                results.append(chunk_results)

            results_df = pd.concat(results, ignore_index=True)

            best_idx = results_df['Total Profit'].idxmax()
            best_cpci_min = results_df.loc[best_idx, 'CPCI_Min']
            best_cpci_max = results_df.loc[best_idx, 'CPCI_Max']
            best_profit = results_df.loc[best_idx, 'Total Profit']

            current_profit = direction_trades['Profit/Loss'].sum()
            profit_after_rejection = current_profit - best_profit

            if best_profit >= 0:
                print(f"No profitable CPCI interval to reject for {direction} trades at interval {interval_num + 1}. Stopping.")
                break

            reject_intervals[direction].append((best_cpci_min, best_cpci_max))
            all_reject_results[direction].append(results_df)

            condition = (
                (remaining_trades['Trade Direction'] == direction) &
                (remaining_trades['CPCI'] > best_cpci_min) &
                (remaining_trades['CPCI'] < best_cpci_max)
            )
            rejected_trades = remaining_trades[condition]
            remaining_trades = remaining_trades[~condition]

            total_profit = remaining_trades['Profit/Loss'].sum()

            print(f"Interval {interval_num + 1}: Rejected {direction} trades with CPCI between {best_cpci_min} and {best_cpci_max}")
            print(f"Profit removed by rejection: ${best_profit:,.2f}")
            print(f"Total Profit after rejection: ${total_profit:,.2f}\n")

            if remaining_trades.empty:
                print("All trades have been rejected. Optimization complete.")
                break

    return reject_intervals, total_profit, all_reject_results


def optimize_vix_atr_cpci_with_multiple_intervals(trades_df, df, initial_investment, 
                                                  vix_i_min=10.0, vix_i_max=30.0, 
                                                  vix_j_min=10.0, vix_j_max=30.0, vix_step=0.1,
                                                  atr_n_min=2, atr_n_max=20, atr_T_min=10.0, atr_T_max=20.0, atr_T_step=0.1,
                                                  cpci_max_intervals=3, cpci_step=0.01):
    """
    Optimizes VIX conditions first, followed by ATR parameters, and finally CPCI conditions with multiple rejection intervals for both Long and Short trades to maximize total profit.
    """
    best_i, best_j, all_vix_results = find_best_vix_conditions_optimized(
        trades_df=trades_df,
        i_min=vix_i_min,
        i_max=vix_i_max,
        j_min=vix_j_min,
        j_max=vix_j_max,
        step=vix_step
    )

    trades_best_vix = trades_df[
        (trades_df['VIX_Open'] > best_i) & 
        (trades_df['VIX_Open'] < best_j)
    ]

    if trades_best_vix.empty:
        print("No trades found with the best VIX conditions.")
        return None, None, None, None, None

    n_values = range(atr_n_min, atr_n_max + 1)          # ATR periods from atr_n_min to atr_n_max
    T_values = np.round(np.arange(atr_T_min, atr_T_max + atr_T_step, atr_T_step), 1)  # ATR thresholds

    best_n, best_T, all_atr_results = optimize_atr_parameters(
        trades_df=trades_best_vix,
        df=df,
        initial_investment=initial_investment,
        n_values=n_values,
        T_values=T_values
    )

    if best_n is None:
        print("ATR optimization did not find any valid parameter combinations.")
        return best_i, best_j, None, None, None

    df_with_atr = calculate_atr(df.copy(), best_n)
    trades_best_vix_atr = trades_best_vix.merge(df_with_atr[['Date', 'ATR']], on='Date', how='left')
    trades_best_vix_atr = trades_best_vix_atr[trades_best_vix_atr['ATR'] <= best_T]

    if trades_best_vix_atr.empty:
        print("No trades found after applying ATR conditions.")
        return best_i, best_j, best_n, best_T, 0

    reject_intervals, total_profit, all_reject_results = find_best_cpci_reject_intervals(
        trades_df=trades_best_vix_atr,
        max_intervals=cpci_max_intervals,
        step=cpci_step
    )

    print(f"\nAnalyzing trades with optimal VIX, ATR, and CPCI rejection intervals:")
    for direction, intervals in reject_intervals.items():
        for idx, (min_cpci, max_cpci) in enumerate(intervals, start=1):
            print(f"  Rejected {direction} Interval {idx}: {min_cpci} < CPCI < {max_cpci}")

    print(f"\nTotal Profit after rejecting {len(reject_intervals['Long']) + len(reject_intervals['Short'])} CPCI intervals: ${total_profit:,.2f}")

    if all_vix_results is not None:
        all_vix_results.to_csv('optimized_vix_conditions_results.csv', index=False)
    if all_atr_results is not None:
        all_atr_results.to_csv('optimized_atr_parameters_results.csv', index=False)
    if all_reject_results:
        for direction, reject_dfs in all_reject_results.items():
            for idx, reject_df in enumerate(reject_dfs, start=1):
                reject_df.to_csv(f'optimized_cpci_reject_interval_{direction}_{idx}_results.csv', index=False)

    num_winning_trades = trades_best_vix_atr[trades_best_vix_atr['Profit/Loss'] > 0].shape[0]
    num_losing_trades = trades_best_vix_atr[trades_best_vix_atr['Profit/Loss'] < 0].shape[0]

    print(f"\nNumber of Winning Trades: {num_winning_trades}")
    print(f"Number of Losing Trades: {num_losing_trades}")

    print("\nTrade Details:")
    if not trades_best_vix_atr.empty:
        print(trades_best_vix_atr.to_string(index=False))
    else:
        print("No trades meet all VIX and ATR conditions.")

    print("\n=== Combined Optimization Summary ===")
    print(f"Optimal VIX Conditions: i={best_i}, j={best_j}")
    print(f"Optimal ATR Parameters: n={best_n}, T={best_T}")
    print(f"Rejected CPCI Intervals:")
    for direction, intervals in reject_intervals.items():
        for idx, (min_cpci, max_cpci) in enumerate(intervals, start=1):
            print(f"  {direction} Interval {idx}: {min_cpci} < CPCI < {max_cpci}")
    print(f"Total Profit: ${total_profit:,.2f}")

    return {
        'VIX_i': best_i,
        'VIX_j': best_j,
        'ATR_n': best_n,
        'ATR_T': best_T,
        'CPCI_Reject_Intervals': reject_intervals,
        'Total_Profit': total_profit
    }


def count_winning_trades(trades_df, trade_direction):
    """
    Returns the number of trades with Profit/Loss greater than zero for the given trade direction.
    """
    return trades_df[trades_df['Profit/Loss'] > 0].shape[0]

def count_losing_trades(trades_df, trade_direction):
    """
    Returns the number of trades with Profit/Loss less than zero for the given trade direction.
    """
    return trades_df[trades_df['Profit/Loss'] < 0].shape[0]


def analyze_trading_strategy_combined(main_file, vix_file, cpci_file, initial_investment, 
                                      vix_i_min=10.0, vix_i_max=30.0, 
                                      vix_j_min=10.0, vix_j_max=30.0, vix_step=0.1,
                                      atr_n_min=2, atr_n_max=20, atr_T_min=10.0, atr_T_max=20.0, atr_T_step=0.1,
                                      cpci_max_intervals=3, cpci_step=0.01):
    """
    Main function that orchestrates the combined VIX, ATR, and CPCI optimization with multiple CPCI rejection intervals for both Long and Short trades.
    The optimization sequence is now VIX → ATR → CPCI (with multiple intervals).
    """
    print("=== Combined VIX, ATR, and CPCI Optimization with Multiple CPCI Intervals ===\n")


    print("Loading stock data...")
    df = load_stock_data(main_file)
    print("Loading VIX data...")
    vix_df = load_vix_data(vix_file)
    print("Loading CPCI data...")
    cpci_df = load_cpci_data(cpci_file)

    print("Precomputing trades based on gap, VIX, and CPCI conditions...")
    trades_df = precompute_trades(df, vix_df, cpci_df, initial_investment)

    if trades_df.empty:
        print("No valid trades found based on initial criteria.")
        return None

    print(f"Total Valid Trades: {len(trades_df)}\n")

    optimization_results = optimize_vix_atr_cpci_with_multiple_intervals(
        trades_df=trades_df,
        df=df,
        initial_investment=initial_investment,
        vix_i_min=vix_i_min,
        vix_i_max=vix_i_max,
        vix_j_min=vix_j_min,
        vix_j_max=vix_j_max,
        vix_step=vix_step,
        atr_n_min=atr_n_min,
        atr_n_max=atr_n_max,
