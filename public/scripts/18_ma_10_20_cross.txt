import pandas as pd
import numpy as np
from base_strategy import GapStrategy
import sys
import os
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from gap_utils import *

class MA1020CrossStrategy(GapStrategy):
    """
    MA 10/20 Cross Strategy
    Success Rate: 70.24% | Total Profit: $6,008.71 | Trades: 742
    
    This strategy trades gaps when the 10-day MA is above the 20-day MA,
    providing a dual moving average trend filter.
    """
    
    def __init__(self):
        super().__init__(
            name="MA 10/20 Cross",
            description="Fade gaps when 10-day MA is above 20-day MA"
        )
        self.fast_ma = 10
        self.slow_ma = 20
    
    def calculate_indicators(self, df):
        """Calculate moving averages"""
        df['SMA_10'] = calculate_sma(df, period=self.fast_ma)
        df['SMA_20'] = calculate_sma(df, period=self.slow_ma)
        
        return df
    
    def get_filter_condition(self, df):
        """10-day MA must be above 20-day MA"""
        return df['SMA_10'] > df['SMA_20']

def main():
    """Run the strategy"""
    csv_files = [f for f in os.listdir('.') if f.endswith('.csv') and f not in ['gap_strategies_all_results.csv', 'gap_strategies_top10_profit.csv', 'gap_strategies_top10_success_rate.csv']]
    
    if csv_files:
        data_file = csv_files[0]
        print(f"Using data file: {data_file}")
    else:
        parent_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
        data_file = os.path.join(parent_dir, '..', '7J0OK1VR.csv')
        if not os.path.exists(data_file):
            print("Error: No CSV data file found. Please place a CSV file in the current directory.")
            return
    
    strategy = MA1020CrossStrategy()
    
    try:
        stats, trades_df = strategy.run_strategy(data_file)
        strategy.print_results(stats, trades_df)
        
        output_file = f"{strategy.name.replace(' ', '_')}_trades.csv"
        trades_df.to_csv(output_file, index=False)
        print(f"Trade details saved to: {output_file}")
        
    except Exception as e:
        print(f"Error running strategy: {e}")

if __name__ == "__main__":
    main()
