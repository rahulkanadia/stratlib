class RSIOverbought70AndMACDBearishStrategy:
    """
    Two-Indicator Strategy: RSI_Overbought_70 + MACD_Bearish
    Success Rate: 94.44% | Total Profit: $736.91 | Trades: 18
    
    This strategy combines:
    1. RSI_Overbought_70
    2. MACD_Bearish
    """
    
    def __init__(self):
        self.name = "RSI_Overbought_70 + MACD_Bearish"
        self.indicator1 = "RSI_Overbought_70"
        self.indicator2 = "MACD_Bearish"
        self.initial_investment = 16000
        
    def calculate_indicators(self, df):
        """Calculate required indicators"""
        
        for period in [5, 10, 20, 50, 100, 200]:
            df[f'SMA_{period}'] = calculate_sma(df, period)

        df['RSI'] = calculate_rsi(df, 14)
        df['RSI_5'] = calculate_rsi(df, 5)
        df['RSI_21'] = calculate_rsi(df, 21)

        macd_line, signal_line = calculate_macd(df)
        df['MACD'] = macd_line
        df['MACD_Signal'] = signal_line
        
        return df
    
    def get_filter_conditions(self, df):
        """Get the two indicator conditions"""
        condition1 = df['RSI'] > 70
        
        condition2 = pd.Series([True]*len(df))
        
        return condition1 & condition2
    
    def run_strategy(self, data_file):
        """Run the complete strategy"""
        df = load_data(data_file)
        
        df = self.calculate_indicators(df)
        
        filter_condition = self.get_filter_conditions(df)
        
        trades = []
        for i in range(1, len(df)):
            if not filter_condition.iloc[i]:
                continue
                
            prev_day = df.iloc[i - 1]
            curr_day = df.iloc[i]
            
            gap = calculate_gap(prev_day['Close'], curr_day['Open'])
            if not is_gap_valid(gap):
                continue
            
            trade_direction = determine_trade_direction(gap)
            entry_price = curr_day['Open']
            exit_price, gap_closed = check_gap_closure(trade_direction, curr_day, prev_day['Close'])
            
            profit_loss = calculate_profit_loss(
                trade_direction, entry_price, exit_price, self.initial_investment
            )
            
            trades.append({
                'Date': curr_day['Date'],
                'Trade Direction': trade_direction,
                'Entry Price': entry_price,
                'Exit Price': exit_price,
                'Gap': gap,
                'Gap Closed': gap_closed,
                'Profit/Loss': profit_loss
            })
        
        trades_df = pd.DataFrame(trades)
        stats = calculate_statistics(trades_df, self.name, self.initial_investment)
        
        return stats, trades_df
    
    def print_results(self, stats, trades_df):
        """Print formatted results"""
        print(f"\n{'='*60}")
        print(f"{self.name} - Two-Indicator Gap Trading Strategy")
        print(f"{'='*60}")
        print(f"Indicators: {self.indicator1} + {self.indicator2}")
        print(f"\nStrategy Performance:")
        print(f"Total Trades: {stats['Total Trades']}")
        print(f"Success Rate: {stats['Success Rate']:.2f}%")
        print(f"Total Profit: ${stats['Total Profit']:,.2f}")
        print(f"ROI: {stats['ROI']:.2f}%")
        print(f"Final Equity: ${stats['Final Equity']:,.2f}")
        print(f"{'='*60}\n")

def main():
    """Run the strategy"""
    csv_files = [f for f in os.listdir('.') if f.endswith('.csv')

    print(f"Using data file: {data_file}")
    
    strategy = RSIOverbought70AndMACDBearishStrategy()
    
    try:
        stats, trades_df = strategy.run_strategy(data_file)
        strategy.print_results(stats, trades_df)

    except Exception as e:
        print(f"Error running strategy: {e}")

    main()
