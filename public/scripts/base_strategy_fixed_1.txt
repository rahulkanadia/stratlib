"""
Base Strategy Class - Fixed Version
-----------------------------------
Base class for trading strategies that matches the expected interface.
"""

import pandas as pd
import numpy as np
from datetime import datetime
from pathlib import Path
import os
import sys

sys.path.append(os.path.dirname(os.path.abspath(__file__)))
from trading_utils import *


class BaseStrategy:
    """
    Base class for trading strategies.
    
    This is a concrete base class that provides common functionality
    without requiring abstract methods to be implemented.
    """
    
    def __init__(self, name="Strategy", description="", required_data=None):
        """
        Initialize the strategy.
        
        Parameters:
        -----------
        name : str
            Name of the strategy
        description : str
            Description of the strategy
        required_data : list
            List of required data columns
        """
        self.name = name
        self.description = description
        self.required_data = required_data or ['Date', 'Open', 'High', 'Low', 'Close']
        self.data = None
        self.signals = None
        self.trades = []
        self.performance_metrics = None
        
    def load_data(self, data_path):
        """
        Load data from CSV file.
        
        Parameters:
        -----------
        data_path : str or Path
            Path to the data file
        """
        if isinstance(data_path, Path):
            data_path = str(data_path)
            
        self.data = load_data(data_path)
        if self.data is not None:
            if not validate_data(self.data, self.required_data):
                self.data = None
                raise ValueError(f"Data missing required columns: {self.required_data}")
    
    def calculate_signals(self):
        """
        Calculate trading signals.
        Override this method in derived classes.
        
        Returns:
        --------
        pd.DataFrame
            DataFrame with trading signals
        """
        raise NotImplementedError("Subclasses must implement calculate_signals()")
    
    def run_backtest(self, initial_capital=10000, position_size=1.0):
        """
        Run backtest on the signals.
        
        Parameters:
        -----------
        initial_capital : float
            Starting capital
        position_size : float
            Fraction of capital to use per trade (default: 1.0 = 100%)
            
        Returns:
        --------
        dict
            Performance metrics
        """
        if self.signals is None or len(self.signals) == 0:
            print("No signals to backtest")
            return None
        
        self.trades = []
        capital = initial_capital
        
        for _, signal in self.signals.iterrows():
            if capital <= 0:
                break
                
            trade_capital = capital * position_size
            shares = int(trade_capital / signal['entry_price'])
            
            if shares == 0:
                continue
            
            if signal['direction'] == 'LONG':
                profit_loss = (signal['exit_price'] - signal['entry_price']) * shares
            else:  # SHORT
                profit_loss = (signal['entry_price'] - signal['exit_price']) * shares
            
            capital += profit_loss
            
            trade = {
                'entry_date': signal['date'],
                'entry_price': signal['entry_price'],
                'exit_date': signal.get('exit_date', signal['date']),
                'exit_price': signal['exit_price'],
                'direction': signal['direction'],
                'shares': shares,
                'profit_loss': profit_loss,
                'capital_after': capital,
                'signal_strength': signal.get('signal_strength', 1.0)
            }
            self.trades.append(trade)
        
        self.performance_metrics = self._calculate_performance_metrics(initial_capital)
        return self.performance_metrics
    
    def _calculate_performance_metrics(self, initial_capital):
        """Calculate performance metrics from trades."""
        if len(self.trades) == 0:
            return {
                'total_trades': 0,
                'win_rate': 0,
                'total_profit': 0,
                'roi': 0,
                'sharpe_ratio': 0,
                'max_drawdown': 0,
                'profit_factor': 0
            }
        
        trades_df = pd.DataFrame(self.trades)
        
        total_trades = len(trades_df)
        profitable_trades = len(trades_df[trades_df['profit_loss'] > 0])
        win_rate = (profitable_trades / total_trades * 100) if total_trades > 0 else 0
        
        total_profit = trades_df['profit_loss'].sum()
        roi = (total_profit / initial_capital * 100)
        
        if len(trades_df) > 1:
            returns = trades_df['profit_loss'] / initial_capital
            sharpe_ratio = calculate_sharpe_ratio(returns)
        else:
            sharpe_ratio = 0
        
        equity_curve = initial_capital + trades_df['profit_loss'].cumsum()
        max_drawdown = calculate_max_drawdown(equity_curve)
        
        gross_profit = trades_df[trades_df['profit_loss'] > 0]['profit_loss'].sum()
        gross_loss = abs(trades_df[trades_df['profit_loss'] < 0]['profit_loss'].sum())
        profit_factor = (gross_profit / gross_loss) if gross_loss > 0 else float('inf')
        
        return {
            'total_trades': total_trades,
            'win_rate': round(win_rate, 2),
            'total_profit': round(total_profit, 2),
            'roi': round(roi, 2),
            'sharpe_ratio': round(sharpe_ratio, 3),
            'max_drawdown': round(max_drawdown, 2),
            'profit_factor': round(profit_factor, 2)
        }
    
    def get_summary_metrics(self):
        """Get performance summary metrics."""
        return self.performance_metrics
    
    def print_results(self):
        """Print strategy results."""
        if self.performance_metrics:
            print_statistics(self.performance_metrics, self.name)
        else:
            print(f"No results available for {self.name}")
