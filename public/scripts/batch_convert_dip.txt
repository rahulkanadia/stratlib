def extract_dip_strategy_info(file_path):
    """Extract key information from a DIP strategy file."""
    info = {
        'file_name': os.path.basename(file_path),
        'optimization_type': 'unknown',
        'description': '',
        'parameters': {},
        'filters': []
    }
    
    filename = os.path.basename(file_path)
    patterns = {
        'window': r'window|basic',
        'multiplier': r'multiplier',
        'vix': r'vix',
        'atr': r'atr',
        'volume': r'volume',
        'day_of_week': r'day.*week',
        'time_of_day': r'time.*day',
        'rsi': r'rsi',
        'move': r'move',
        'consecutive': r'consecutive',
        'gap': r'gap',
        'range': r'range',
        'price_level': r'price.*level',
        'month': r'month',
        'volatility': r'volatility',
        'momentum': r'momentum',
        'bollinger': r'bollinger|bb',
        'candle': r'candle'
    }
    
    for opt_type, pattern in patterns.items():
        if re.search(pattern, filename.lower()):
            info['optimization_type'] = opt_type
            break
    
    try:
        with open(file_path, 'r') as f:
            content = f.read()
        
        docstring_match = re.search(r'"""(.*?)"""', content, re.DOTALL)
        if docstring_match:
            info['description'] = docstring_match.group(1).strip()
        
        window_match = re.search(r'window\s*=\s*(\d+)', content, re.IGNORECASE)
        if window_match:
            info['parameters']['window'] = int(window_match.group(1))
        
        mult_match = re.search(r'multiplier\s*=\s*([\d.]+)', content, re.IGNORECASE)
        if mult_match:
            info['parameters']['multiplier'] = float(mult_match.group(1))
        
        vix_matches = re.findall(r'vix.*?([<>]=?)\s*([\d.]+)', content, re.IGNORECASE)
        if vix_matches:
            info['parameters']['vix_conditions'] = vix_matches
        
        filter_keywords = [
            'vix', 'atr', 'volume', 'rsi', 'macd', 'bollinger', 
            'momentum', 'volatility', 'gap', 'range'
        ]
        for keyword in filter_keywords:
            if keyword in content.lower():
                info['filters'].append(keyword.upper())
                
    except Exception as e:
        print(f"Error reading {file_path}: {e}")
    
    return info

def batch_convert_dip_strategies():
    """Convert all DIP strategies to the new format."""
    dip_dir = '/mnt/c/Users/danie/OneDrive/Desktop/claude/2020 to 2024/DIP_Optimizations'
    
    output_dir = '/mnt/c/Users/danie/OneDrive/Desktop/claude/2020 to 2024/clean_strategies/strategies/dip_strategies'
    
    dip_files = [f for f in os.listdir(dip_dir) if f.endswith('.py') and f.startswith('DIP_')]
    
    print(f"Found {len(dip_files)} DIP strategy files")
    print("="*70)
    
    summary = {
        'total': len(dip_files),
        'converted': 0,
        'skipped': 0,
        'errors': 0,
        'by_type': {}
    }
    
    for i, filename in enumerate(dip_files, 1):
        file_path = os.path.join(dip_dir, filename)
        print(f"\n[{i}/{len(dip_files)}] Processing: {filename}")
        
        try:
            info = extract_dip_strategy_info(file_path)
            
            opt_type = info['optimization_type']
            summary['by_type'][opt_type] = summary['by_type'].get(opt_type, 0) + 1
            
            new_filename = f"dip_{opt_type}.py"
            
            output_path = os.path.join(output_dir, new_filename)
            if os.path.exists(output_path):
                print(f"  → Skipped (already exists): {new_filename}")
                summary['skipped'] += 1
                continue
            
            strategy_code = generate_dip_strategy(info)
            
            with open(output_path, 'w') as f:
                f.write(strategy_code)
            
            print(f"  ✓ Converted to: {new_filename}")
            print(f"    Type: {opt_type}")
            print(f"    Filters: {', '.join(info['filters'])}")
            
            summary['converted'] += 1
            
        except Exception as e:
            print(f"  ✗ Error: {e}")
            summary['errors'] += 1
    
    print("\n" + "="*70)
    print("CONVERSION SUMMARY")
    print("="*70)
    print(f"Total files:     {summary['total']}")
    print(f"Converted:       {summary['converted']}")
    print(f"Skipped:         {summary['skipped']}")
    print(f"Errors:          {summary['errors']}")
    
    print("\nBy optimization type:")
    for opt_type, count in sorted(summary['by_type'].items()):
        print(f"  {opt_type:20} {count:3} files")
    
    print("\n" + "="*70)
    print("NEXT STEPS:")
    print("="*70)
    print("1. Review generated strategies in strategies/dip_strategies/")
    print("2. Fill in specific filter logic marked with comments")
    print("3. Test each strategy with run_all.py")
    print("4. Fine-tune parameters as needed")
    
    return summary
