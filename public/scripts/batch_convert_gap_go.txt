def extract_indicator_from_filename(filename):
    """Extract the two indicators from a Gap & Go strategy filename."""
    name = filename.replace('.py', '')
    
    if 'profit' in name:
        ranking_type = 'profit'
        pattern = r'gap_go_profit_(\d+)_(.*)'
    else:
        ranking_type = 'success'
        pattern = r'gap_go_success_(\d+)_(.*)'
    
    match = re.match(pattern, name)
    if not match:
        return None, None, None, None
    
    rank = int(match.group(1))
    indicator_part = match.group(2)
    
    if '_and_' not in indicator_part:
        return ranking_type, rank, None, None
    
    indicators = indicator_part.split('_and_')
    if len(indicators) != 2:
        return ranking_type, rank, None, None
    
    indicator1 = indicators[0].replace('_', ' ').title()
    indicator2 = indicators[1].replace('_', ' ').title()
    
    return ranking_type, rank, indicator1, indicator2

def analyze_gap_go_strategies():
    """Analyze all Gap & Go strategy files and extract patterns."""
    gap_go_dir = '/mnt/c/Users/danie/OneDrive/Desktop/claude/2020 to 2024/GAP_AND_GO_1000_STRATEGIES'
    
    strategy_files = [f for f in os.listdir(gap_go_dir) 
                     if f.startswith('gap_go_') and f.endswith('.py')]
    
    print(f"Found {len(strategy_files)} Gap & Go strategy files")
    
    profit_strategies = []
    success_strategies = []
    indicator_frequency = defaultdict(int)
    
    for filename in strategy_files:
        ranking_type, rank, ind1, ind2 = extract_indicator_from_filename(filename)
        
        if ranking_type and ind1 and ind2:
            strategy_info = {
                'filename': filename,
                'rank': rank,
                'indicator1': ind1,
                'indicator2': ind2,
                'normalized_ind1': normalize_indicator_name(ind1),
                'normalized_ind2': normalize_indicator_name(ind2)
            }
            
            if ranking_type == 'profit':
                profit_strategies.append(strategy_info)
            else:
                success_strategies.append(strategy_info)
            
            indicator_frequency[normalize_indicator_name(ind1)] += 1
            indicator_frequency[normalize_indicator_name(ind2)] += 1
    
    profit_strategies.sort(key=lambda x: x['rank'])
    success_strategies.sort(key=lambda x: x['rank'])
    
    return profit_strategies, success_strategies, indicator_frequency


def create_top_strategies_only(profit_strategies, success_strategies, top_n=10):
    """Create only the top N strategies by each ranking to avoid duplication."""
    strategies_to_create = []
    
    for strategy in profit_strategies[:top_n]:
        strategies_to_create.append(('profit', strategy))
    
    for strategy in success_strategies[:top_n]:
        strategies_to_create.append(('success', strategy))
    
    return strategies_to_create
