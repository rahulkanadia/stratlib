import pandas as pd

def load_data(file_name):
    """Loads the CSV file into a DataFrame."""
    df = pd.read_csv(file_name, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def load_vix_data(vix_file):
    """Loads VIX data."""
    try:
        vix_df = pd.read_csv(vix_file, parse_dates=['DATE'], dayfirst=True)
        vix_df.rename(columns={'DATE': 'Date', 'OPEN': 'VIX_Open', 'CLOSE': 'VIX_Close'}, inplace=True)
        vix_df = vix_df[['Date', 'VIX_Open', 'VIX_Close']]
        return vix_df
    except:
        return None

def calculate_all_indicators(df, vix_df=None):
    """Calculate ALL indicators with NO LOOK-AHEAD BIAS - everything shifted by 1."""
    
    df['IsRedDay'] = df['Close'] < df['Open']
    df['DailyDrop'] = np.where(df['IsRedDay'], df['Open'] - df['Close'], 0.0)
    df['RollingAvgDrop'] = df['DailyDrop'].rolling(window=21).mean().shift(1)
    df['RollingAvgDrop'].fillna(0.0, inplace=True)
    
    df['DayOfWeek'] = df['Date'].dt.dayofweek
    df['Month'] = df['Date'].dt.month
    
    df['RedDayGroup'] = (df['IsRedDay'] != df['IsRedDay'].shift()).cumsum()
    df['ConsecutiveRed'] = df.groupby(df['RedDayGroup']).cumcount() + 1
    df['ConsecutiveRed'] = np.where(df['IsRedDay'], df['ConsecutiveRed'], 0)
    df['PrevConsecutiveRed'] = df['ConsecutiveRed'].shift(1).fillna(0)
    
    for period in [5, 10, 20, 50, 100, 200]:
        df[f'MA{period}'] = df['Close'].rolling(window=period).mean().shift(1)
        df[f'OpenAboveMA{period}'] = df['Open'] > df[f'MA{period}']
    
    df['AvgVolume20'] = df['Volume'].rolling(window=20).mean().shift(1)
    df['PrevVolumeHigh'] = df['Volume'].shift(1) > df['AvgVolume20']
    df['VolumeRatio'] = df['Volume'].shift(1) / df['AvgVolume20']
    
    def calculate_rsi(data, period=14):
        delta = data.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi.shift(1)  # SHIFT for no look-ahead
    
    df['RSI'] = calculate_rsi(df['Close'])
    df['PrevRSIOversold'] = df['RSI'] < 30
    df['PrevRSINeutral'] = (df['RSI'] >= 30) & (df['RSI'] <= 70)
    
    df['BB_Middle'] = df['Close'].rolling(window=20).mean().shift(1)
    df['BB_Std'] = df['Close'].rolling(window=20).std().shift(1)
    df['BB_Upper'] = df['BB_Middle'] + (2 * df['BB_Std'])
    df['BB_Lower'] = df['BB_Middle'] - (2 * df['BB_Std'])
    df['OpenBelowBBLower'] = df['Open'] < df['BB_Lower']
    df['OpenAboveBBMiddle'] = df['Open'] > df['BB_Middle']
    
    df['PrevClose'] = df['Close'].shift(1)
    df['GapPercent'] = ((df['Open'] - df['PrevClose']) / df['PrevClose'] * 100)
    df['GapUp'] = df['GapPercent'] > 0.5
    df['GapDown'] = df['GapPercent'] < -0.5
    
    df['PrevDayReturn'] = ((df['Close'] - df['Open']) / df['Open'] * 100).shift(1)
    df['PrevDayUp'] = df['PrevDayReturn'] > 0
    df['PrevDayDown'] = df['PrevDayReturn'] < 0
    
    df['HL'] = df['High'] - df['Low']
    df['HC'] = abs(df['High'] - df['Close'].shift(1))
    df['LC'] = abs(df['Low'] - df['Close'].shift(1))
    df['TR'] = df[['HL', 'HC', 'LC']].max(axis=1)
    df['ATR14'] = df['TR'].rolling(window=14).mean().shift(1)
    df['ATRHigh'] = df['ATR14'] > df['ATR14'].rolling(window=50).mean().shift(1)
    
    df['PrevRange'] = (df['High'] - df['Low']).shift(1)
    df['AvgRange20'] = df['PrevRange'].rolling(window=20).mean()
    df['WideRange'] = df['PrevRange'] > df['AvgRange20']
    
    df['High20'] = df['High'].rolling(window=20).max().shift(1)
    df['DistanceFromHigh'] = ((df['Open'] - df['High20']) / df['High20'] * 100)
    df['NearHigh'] = df['DistanceFromHigh'] > -2
    df['FarFromHigh'] = df['DistanceFromHigh'] < -5
    
    df['Body'] = abs(df['Close'] - df['Open'])
    df['Range'] = df['High'] - df['Low']
    df['LowerWick'] = df[['Open', 'Close']].min(axis=1) - df['Low']
    df['Hammer_Prev'] = (
        (df['Body'].shift(1) < df['Range'].shift(1) * 0.3) &
        (df['LowerWick'].shift(1) > df['Body'].shift(1) * 2)
    ).fillna(False)
    df['Doji_Prev'] = (df['Body'].shift(1) < df['Range'].shift(1) * 0.1).fillna(False)
    
    df['BullMarket'] = df['Open'] > df['MA200']
    df['BearMarket'] = df['Open'] < df['MA200']
    
    df['Returns'] = df['Close'].pct_change()
    df['RollingVol'] = df['Returns'].rolling(window=20).std().shift(1)
    df['VolPercentile'] = df['RollingVol'].rolling(window=252).rank(pct=True).shift(1)
    df['HighVol'] = df['VolPercentile'] > 0.8
    df['LowVol'] = df['VolPercentile'] < 0.2
    
    if vix_df is not None:
        df = df.merge(vix_df, on='Date', how='left')
        df['VIX_Open'].fillna(method='ffill', inplace=True)
        df['VIXHigh'] = df['VIX_Open'] > 20
        df['VIXLow'] = df['VIX_Open'] < 15
        df['VIXMedium'] = (df['VIX_Open'] >= 15) & (df['VIX_Open'] <= 20)
    
    return df

def calculate_profit_loss_long(entry_price, exit_price, position_size):
    """Calculate profit/loss for long position."""
    if entry_price <= 0:
        return 0
    shares = position_size / entry_price
    return shares * (exit_price - entry_price)

def run_strategy(df, filter_func, initial_investment=16000):
    """Run a strategy with given filter function."""
    trades = []
    
    for i in range(len(df)):
        if i < 200:  # Need history for all indicators
            continue
        
        current_day = df.iloc[i]
        avg_drop = current_day['RollingAvgDrop']
        
        if avg_drop <= 0:
            continue
        
        if not filter_func(current_day):
            continue
        
        open_price = current_day['Open']
        desired_entry = open_price - avg_drop
        
        low_price = current_day['Low']
        if low_price <= desired_entry:
            entry_price = desired_entry
            exit_price = current_day['Close']
            profit_loss = calculate_profit_loss_long(entry_price, exit_price, initial_investment)
            
            trades.append({
                'Date': current_day['Date'],
                'Entry': entry_price,
                'Exit': exit_price,
                'Profit/Loss': profit_loss
            })
    
    return trades

def calculate_stats(trades):
    """Calculate strategy statistics."""
    if not trades:
        return {'trades': 0, 'success_rate': 0, 'total_profit': 0, 'avg_profit': 0}
    
    df_trades = pd.DataFrame(trades)
    total_trades = len(trades)
    winning_trades = len(df_trades[df_trades['Profit/Loss'] > 0])
    success_rate = winning_trades / total_trades * 100 if total_trades > 0 else 0
    total_profit = df_trades['Profit/Loss'].sum()
    avg_profit = df_trades['Profit/Loss'].mean()
    
    return {
        'trades': total_trades,
        'success_rate': success_rate,
        'total_profit': total_profit,
        'avg_profit': avg_profit
    }
