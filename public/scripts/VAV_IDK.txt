def load_data(file_name):
    """
    Loads the CSV file into a DataFrame, sorts it by date, and resets the index.
    """
    df = pd.read_csv(file_name, parse_dates=['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def load_vix_data(vix_file):
    """
    Loads the VIX data from a CSV file into a DataFrame, sorts it by date, and resets the index.
    Ensures that the 'Date' column is in datetime format.
    """
    vix_df = pd.read_csv(vix_file, parse_dates=['DATE'], dayfirst=True, on_bad_lines='skip')  # Skip bad lines
    vix_df.rename(columns={'DATE': 'Date'}, inplace=True)
    vix_df['Date'] = pd.to_datetime(vix_df['Date'], dayfirst=True, errors='coerce')
    vix_df.dropna(subset=['Date'], inplace=True)
    vix_df.sort_values('Date', inplace=True)
    vix_df.reset_index(drop=True, inplace=True)
    return vix_df

def load_half_hour_data(half_hour_file):
    """
    Loads the half-hourly trading data and extracts the first and second half-hour's data for each day.
    Returns a DataFrame with 'Date', 'First_Half_Hour_Volume', and 'Second_Half_Hour_Last'.
    """
    df = pd.read_csv(half_hour_file, on_bad_lines='skip')  # Skip malformed lines
    
    df['Time'] = pd.to_datetime(df['Time'], format='%m/%d/%Y %H:%M', errors='coerce')
    
    df = df.dropna(subset=['Time'])
    
    df['Date'] = df['Time'].dt.date
    
    df.sort_values(['Date', 'Time'], inplace=True)
    
    grouped = df.groupby('Date')
    
    first_half_hour = grouped.first().reset_index()
    second_half_hour = grouped.nth(1).reset_index()
    
    half_hour_grouped = pd.merge(
        first_half_hour[['Date', 'Volume']],
        second_half_hour[['Date', 'Last']],
        on='Date',
        how='left'
    )
    
    half_hour_grouped.rename(columns={
        'Volume': 'First_Half_Hour_Volume',
        'Last': 'Second_Half_Hour_Last'
    }, inplace=True)
    
    half_hour_grouped['Date'] = pd.to_datetime(half_hour_grouped['Date'])
    
    return half_hour_grouped

def calculate_day_range(day):
    """
    Calculates the trading range of a day (High - Low).
    """
    return day['High'] - day['Low']

def calculate_gap(prev_close, curr_open):
    """
    Calculates the gap between the previous day's close and the current day's open.
    """
    return curr_open - prev_close

def is_gap_valid(gap, prev_range):
    """
    Checks if the gap meets the specified conditions:
    - At least $0.20
    - Between 15% and 85% of the previous day's range
    """
    gap_abs = abs(gap)
    if gap_abs < 0.20 or prev_range == 0:
        return False
    gap_percentage = gap_abs / prev_range
    return 0.15 <= gap_percentage <= 0.85

def determine_trade_direction(gap):
    """
    Determines the trade direction based on the gap:
    - Gap up: Short position
    - Gap down: Long position
    """
    return 'Short' if gap > 0 else 'Long'

def check_gap_closure(trade_direction, curr_day, prev_close):
    """
    Checks if the gap is closed during the trading day and returns the exit price.
    """
    if trade_direction == 'Short':
        if curr_day['Low'] <= prev_close:
            exit_price = prev_close
            gap_closed = True
        else:
            exit_price = curr_day['Close']
            gap_closed = False
    else:  # Long position
        if curr_day['High'] >= prev_close:
            exit_price = prev_close
            gap_closed = True
        else:
            exit_price = curr_day['Close']
            gap_closed = False
    return exit_price, gap_closed

def calculate_profit_loss(trade_direction, entry_price, exit_price, position_size):
    """
    Calculates the profit or loss of a trade.
    """
    number_of_shares = position_size / entry_price

    if trade_direction == 'Long':
        profit_loss = number_of_shares * (exit_price - entry_price)
    else:  # Short position
        profit_loss = number_of_shares * (entry_price - exit_price)

    return profit_loss

def precompute_trades(df, vix_df, initial_investment):
    """
    Precomputes all valid trades and their associated VIX_open values and profits.
    Returns a DataFrame containing all trades.
    """
    trades = []
    total_days = len(df)

    vix_open_dict = {}
    vix_dates = set(vix_df['Date'])
    min_vix_date = vix_df['Date'].min()

    for current_date in df['Date']:
        prev_date = current_date - pd.Timedelta(days=1)
        while prev_date not in vix_dates and prev_date > min_vix_date:
            prev_date -= pd.Timedelta(days=1)
        if prev_date in vix_dates:
            vix_open = vix_df.loc[vix_df['Date'] == prev_date, 'OPEN'].values[0]
            vix_open_dict[current_date] = vix_open
        else:
            vix_open_dict[current_date] = np.nan  # No VIX data for the previous day

    for i_day in range(1, total_days):
        prev_day = df.iloc[i_day - 1]
        curr_day = df.iloc[i_day]
        current_date = curr_day['Date']
        vix_open = vix_open_dict.get(current_date, np.nan)

        if pd.isna(vix_open):
            continue  # Skip if no VIX data

        prev_range = calculate_day_range(prev_day)
        if prev_range == 0:
            continue  # Avoid division by zero

        gap = calculate_gap(prev_day['Close'], curr_day['Open'])
        if not is_gap_valid(gap, prev_range):
            continue

        trade_direction = determine_trade_direction(gap)
        entry_price_open = curr_day['Open']
        exit_price, gap_closed = check_gap_closure(trade_direction, curr_day, prev_day['Close'])
        profit_loss_open = calculate_profit_loss(trade_direction, entry_price_open, exit_price, initial_investment)

        trades.append({
            'Date': current_date,
            'Trade Direction': trade_direction,
            'Entry Price_Open': entry_price_open,
            'Exit Price': exit_price,
            'Gap Closed': gap_closed,
            'VIX_Open': vix_open,
            'Profit/Loss_Open': profit_loss_open
        })

    trades_df = pd.DataFrame(trades)
    return trades_df

def precompute_trades_with_volume(df, vix_df, half_hour_df, initial_investment):
    """
    Precomputes all valid trades, integrates volume and delayed entry data, and returns a DataFrame containing all trades.
    """
    trades_df = precompute_trades(df, vix_df, initial_investment)
    if trades_df.empty:
        return trades_df

    trades_df = trades_df.merge(half_hour_df, on='Date', how='left')
    
    trades_df.dropna(subset=['First_Half_Hour_Volume', 'Second_Half_Hour_Last'], inplace=True)

    trades_df['Entry Price_Delayed'] = trades_df['Second_Half_Hour_Last']
    trades_df['Profit/Loss_Delayed'] = trades_df.apply(
        lambda row: calculate_profit_loss(
            row['Trade Direction'],
            row['Entry Price_Delayed'],
            row['Exit Price'],
            initial_investment
        ),
        axis=1
    )

    return trades_df

def find_best_vix_conditions_optimized(trades_df, i_min=10.0, i_max=30.0, j_min=10.0, j_max=30.0, step=0.1):
    """
    Optimized function to find the best (i, j) VIX conditions that maximize profit.
    Utilizes vectorized operations for efficiency.
    Returns the best (i, j) pair and a DataFrame of all results.
    """
    i_values = np.round(np.arange(i_min, j_max + step, step), 1)
    j_values = np.round(np.arange(j_min, j_max + step, step), 1)

    i_grid, j_grid = np.meshgrid(i_values, j_values)
    i_flat = i_grid.flatten()
    j_flat = j_grid.flatten()
    valid_pairs_mask = i_flat < j_flat
    i_valid = i_flat[valid_pairs_mask]
    j_valid = j_flat[valid_pairs_mask]

    total_profits = np.zeros(len(i_valid))

    vix_open_array = trades_df['VIX_Open'].values
    profit_open_array = trades_df['Profit/Loss_Open'].values
    profit_delayed_array = trades_df['Profit/Loss_Delayed'].values

    sorted_indices = np.argsort(vix_open_array)
    sorted_vix = vix_open_array[sorted_indices]
    sorted_profit_open = profit_open_array[sorted_indices]
    sorted_profit_delayed = profit_delayed_array[sorted_indices]

    cumulative_profit_open = np.cumsum(sorted_profit_open)
    cumulative_profit_delayed = np.cumsum(sorted_profit_delayed)

    def compute_profit(i, j):
        left = np.searchsorted(sorted_vix, i, side='right')
        right = np.searchsorted(sorted_vix, j, side='left')
        if right > 0:
            profit_open = cumulative_profit_open[right - 1] - (cumulative_profit_open[left - 1] if left > 0 else 0)
            profit_delayed = cumulative_profit_delayed[right - 1] - (cumulative_profit_delayed[left - 1] if left > 0 else 0)
            return profit_open + profit_delayed
        else:
            return 0.0

    vectorized_compute_profit = np.vectorize(compute_profit)
    total_profits = vectorized_compute_profit(i_valid, j_valid)

    results_df = pd.DataFrame({
        'i': i_valid,
        'j': j_valid,
        'Total Profit': total_profits
    })

    best_idx = results_df['Total Profit'].idxmax()
    best_i = results_df.loc[best_idx, 'i']
    best_j = results_df.loc[best_idx, 'j']
    best_profit = results_df.loc[best_idx, 'Total Profit']

    results_df_sorted = results_df.sort_values(by='Total Profit', ascending=False).reset_index(drop=True)

    print("\n=== VIX Optimization Complete ===")
    print(f"Best VIX Conditions: i={best_i}, j={best_j} with Total Profit: ${best_profit:,.2f}\n")

    print("Top 10 VIX Conditions by Total Profit:")
    print(results_df_sorted.head(10).to_string(index=False))

    return best_i, best_j, results_df_sorted

def calculate_atr(df, n):
    """
    Calculates the Average True Range (ATR) over 'n' consecutive days.
    Ensures that 'n' is an integer.
    """
    n = int(n)  # Ensure n is an integer
    df['Previous Close'] = df['Close'].shift(1)
    df['High-Low'] = df['High'] - df['Low']
    df['High-PrevClose'] = abs(df['High'] - df['Previous Close'])
    df['Low-PrevClose'] = abs(df['Low'] - df['Previous Close'])
    df['TR'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=n).mean()
    return df

def compute_atr_total_profit_sequential(args):
    """
    Computes the total profit for a given combination of n and T.
    This function must be defined at the top level to be picklable by multiprocessing.
    """
    trades_df, df, n, T, initial_investment = args
    df_with_atr = calculate_atr(df.copy(), n)
    trades_with_atr = trades_df.merge(df_with_atr[['Date', 'ATR']], on='Date', how='left')
    trades_with_atr = trades_with_atr.dropna(subset=['ATR'])
    trades_filtered = trades_with_atr[trades_with_atr['ATR'] <= T]
    total_profit = trades_filtered['Profit/Loss_Open'].sum() + trades_filtered['Profit/Loss_Delayed'].sum()
    return (n, T, total_profit)

def optimize_atr_parameters(trades_df, df, initial_investment, n_values, T_values):
    """
    Optimizes both ATR period 'n' and ATR threshold 'T' to maximize total profit.
    """
    param_combinations = list(product(n_values, T_values))

    args_list = [(trades_df.copy(), df.copy(), n, T, initial_investment) for n, T in param_combinations]

    with Pool(processes=cpu_count()) as pool:
        results = pool.map(compute_atr_total_profit_sequential, args_list)

    results_df = pd.DataFrame(results, columns=['n', 'T', 'Total Profit'])

    best_result = results_df.loc[results_df['Total Profit'].idxmax()]
    best_n = int(best_result['n'])  # Ensure n is an integer
    best_T = best_result['T']
    best_total_profit_atr = best_result['Total Profit']

    print("\n=== ATR Optimization Complete ===")
    print(f"Best ATR Parameters: n={best_n}, T={best_T} with Total Profit: ${best_total_profit_atr:,.2f}\n")

    top_results = results_df.sort_values(by='Total Profit', ascending=False).reset_index(drop=True)

    print("Top 100 ATR Parameter Combinations by Total Profit:")
    if not top_results.empty:
        print(top_results.head(100).to_string(index=False))
    else:
        print("No ATR parameter combinations yielded any profit.")

    return best_n, best_T, top_results

def find_best_volume_thresholds(trades_df, volume_min, volume_max, step):
    """
    Finds the best volume thresholds that maximize total profit.
    Returns the top 10 thresholds and their corresponding profits.
    """
    thresholds = np.arange(volume_min, volume_max + step, step)
    total_profits = []

    volume_array = trades_df['First_Half_Hour_Volume'].values
    profit_open_array = trades_df['Profit/Loss_Open'].values
    profit_delayed_array = trades_df['Profit/Loss_Delayed'].values

    sorted_indices = np.argsort(volume_array)
    sorted_volume = volume_array[sorted_indices]
    sorted_profit_open = profit_open_array[sorted_indices]
    sorted_profit_delayed = profit_delayed_array[sorted_indices]

    cumulative_profit_open = np.cumsum(sorted_profit_open)
    cumulative_profit_delayed = np.cumsum(sorted_profit_delayed)

    def compute_total_profit(threshold):
        idx = np.searchsorted(sorted_volume, threshold, side='right')
        if idx > 0:
            profit_open = cumulative_profit_open[idx - 1]
            profit_delayed = cumulative_profit_delayed[-1] - (cumulative_profit_delayed[idx - 1] if idx > 0 else 0)
            return profit_open + profit_delayed
        else:
            return cumulative_profit_delayed[-1]

    vectorized_compute_total_profit = np.vectorize(compute_total_profit)
    total_profits = vectorized_compute_total_profit(thresholds)

    results_df = pd.DataFrame({
        'Volume Threshold': thresholds,
        'Total Profit': total_profits
    })

    results_df_sorted = results_df.sort_values(by='Total Profit', ascending=False).reset_index(drop=True)

    top_10 = results_df_sorted.head(10)

    print("\n=== Volume Threshold Optimization Complete ===")
    print("Top 10 Volume Thresholds by Total Profit:")
    print(top_10.to_string(index=False))

    return top_10

def analyze_spy_gaps_with_volume(main_file, vix_file, half_hour_file, initial_investment, 
                                 vix_i_min=10.0, vix_i_max=30.0, 
                                 vix_j_min=10.0, vix_j_max=30.0, vix_step=0.1,
                                 atr_n_min=2, atr_n_max=20, atr_T_min=10.0, atr_T_max=20.0, atr_T_step=0.1,
                                 volume_step=100000):
    """
    Main function that orchestrates the combined VIX, ATR, and Volume optimization.
    """
    print("Loading stock data...")
    df = load_data(main_file)
    print("Loading VIX data...")
    vix_df = load_vix_data(vix_file)
    print("Loading Half-Hour Volume data...")
    half_hour_df = load_half_hour_data(half_hour_file)

    print("Precomputing trades based on gap and VIX conditions...")
    trades_df = precompute_trades_with_volume(df, vix_df, half_hour_df, initial_investment)

    if trades_df.empty:
        print("No valid trades found based on initial criteria.")
        return None, None, None, None, None, None

    print(f"Total Valid Trades: {len(trades_df)}")

    best_i, best_j, all_vix_results = find_best_vix_conditions_optimized(
        trades_df=trades_df,
        i_min=vix_i_min,
        i_max=vix_i_max,
        j_min=vix_j_min,
        j_max=vix_j_max,
        step=vix_step
    )

    trades_best_vix = trades_df[(trades_df['VIX_Open'] > best_i) & (trades_df['VIX_Open'] < best_j)]

    if trades_best_vix.empty:
        print("No trades found with the best VIX conditions.")
        return best_i, best_j, None, None, None, None

    n_values = range(atr_n_min, atr_n_max + 1)          # ATR periods from atr_n_min to atr_n_max
    T_values = np.round(np.arange(atr_T_min, atr_T_max + atr_T_step, atr_T_step), 1)  # ATR thresholds

    best_n, best_T, all_atr_results = optimize_atr_parameters(
        trades_df=trades_best_vix,
        df=df,
        initial_investment=initial_investment,
        n_values=n_values,
        T_values=T_values
    )


    volume_min = trades_df['First_Half_Hour_Volume'].min()
    volume_max = trades_df['First_Half_Hour_Volume'].max()

    print(f"\nOptimizing Volume Thresholds from {volume_min} to {volume_max} with step {volume_step}...")

    top_10_volumes = find_best_volume_thresholds(
        trades_df=trades_df,
        volume_min=volume_min,
        volume_max=volume_max,
        step=volume_step
    )

    best_volume_threshold = top_10_volumes.iloc[0]['Volume Threshold']
    best_volume_profit = top_10_volumes.iloc[0]['Total Profit']

    print(f"\nBest Volume Threshold: {best_volume_threshold} with Total Profit: ${best_volume_profit:,.2f}")

    print(f"\nAnalyzing trades with best VIX, ATR, and Volume parameters: i={best_i}, j={best_j}, n={best_n}, T={best_T}, Volume Threshold={best_volume_threshold}")

    trades_final = trades_df[
        (trades_df['VIX_Open'] > best_i) & 
        (trades_df['VIX_Open'] < best_j)
    ]

    df_with_atr = calculate_atr(df.copy(), best_n)

    trades_final = trades_final.merge(df_with_atr[['Date', 'ATR']], on='Date', how='left')

    trades_final = trades_final[trades_final['ATR'] <= best_T]

    trades_final_filtered = trades_final[trades_final['First_Half_Hour_Volume'] <= best_volume_threshold]

    trades_final['Trade Entry'] = np.where(
        trades_final['First_Half_Hour_Volume'] <= best_volume_threshold,
        'Open',
        'Delayed'
    )

    trades_final['Selected_Profit'] = np.where(
        trades_final['Trade Entry'] == 'Open',
        trades_final['Profit/Loss_Open'],
        trades_final['Profit/Loss_Delayed']
    )

    total_profit = trades_final['Selected_Profit'].sum()

    all_vix_results.to_csv('combined_vix_atr_volume_vix_optimization_results.csv', index=False)
    all_atr_results.to_csv('combined_vix_atr_volume_atr_optimization_results.csv', index=False)
    top_10_volumes.to_csv('combined_vix_atr_volume_volume_threshold_results.csv', index=False)

    num_winning_trades = trades_final[trades_final['Selected_Profit'] > 0].shape[0]
    num_losing_trades = trades_final[trades_final['Selected_Profit'] < 0].shape[0]

    print(f"\nNumber of Winning Trades: {num_winning_trades}")
    print(f"Number of Losing Trades: {num_losing_trades}")

    print("\nTrade Details:")
    if not trades_final.empty:
        print(trades_final.to_string(index=False))
    else:
        print("No trades meet VIX, ATR, and Volume conditions.")

    print("\n=== Combined Optimization Summary ===")
    print(f"Optimal VIX Conditions: i={best_i}, j={best_j}")
    print(f"Optimal ATR Parameters: n={best_n}, T={best_T}")
    print(f"Optimal Volume Threshold: {best_volume_threshold}")
    print(f"Total Profit: ${total_profit:,.2f}")

    return best_i, best_j, best_n, best_T, best_volume_threshold, total_profit
