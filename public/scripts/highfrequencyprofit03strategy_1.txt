import numpy as np
from pathlib import Path

project_root = current_file.parent.parent.parent

from utils.base_strategy_fixed import BaseStrategy

class HighFrequencyProfit03Strategy(BaseStrategy):
    """
    High Frequency Profit #03 Implementation
    
    High frequency trading optimized for profit using Ma 5 10 Cross And Vix High
    """
    
    def __init__(self, data_path=None):
        super().__init__(
            name="High Frequency Profit #03",
            description="HF - Ma 5 10 Cross And Vix High",
            required_data=['Date', 'Open', 'High', 'Low', 'Close', 'Volume']
        )
        
        self.min_gap_percent = 0.3  # Smaller gaps for HF
        self.holding_period = 'intraday'  # Always close same day
        self.max_trades_per_day = 3  # Risk management
        self.vix_lookback = 5  # Shorter lookback
        
        if data_path:
            self.load_data(data_path)
    
    def calculate_indicators(self):
        """Calculate required indicators"""
        df = self.data.copy()
        
        df['Gap_Percent'] = ((df['Open'] - df['Close'].shift(1)) / df['Close'].shift(1)) * 100
        df['Intraday_Range'] = ((df['High'] - df['Low']) / df['Open']) * 100
        
        df['MA_Fast'] = df['Close'].rolling(window=self.ma_fast).mean()
        df['MA_Slow'] = df['Close'].rolling(window=self.ma_slow).mean()
        df['MA_Diff'] = df['MA_Fast'] - df['MA_Slow']
        df['MA_Cross'] = np.sign(df['MA_Diff']).diff()
        
        df['DayOfWeek'] = df['Date'].dt.dayofweek
        df['Tuesday'] = df['DayOfWeek'] == 1
        df['EarlyWeek'] = df['DayOfWeek'] <= 1
        
        return df
    
    def calculate_signals(self):
        """Generate trading signals based on strategy rules"""
        if self.data is None:
            raise ValueError("No data loaded. Please load data first.")
        
        df = self.calculate_indicators()
        signals = []
        
        daily_trades = {}  # Track trades per day
        
        for i in range(20, len(df)):
            curr_day = df.iloc[i]
            curr_date = curr_day['Date'].date()
            
            if curr_date in daily_trades and daily_trades[curr_date] >= self.max_trades_per_day:
                continue
            
            if abs(curr_day['Gap_Percent']) < self.min_gap_percent:
                continue
            
            signal = False
            
            signal = True
            direction = 'SHORT' if curr_day['Gap_Percent'] > 0 else 'LONG'
            
            if signal:
                entry_price = curr_day['Open']
                
                gap_size = abs(curr_day['Gap_Percent'])
                
                if direction == 'LONG':
                    target = entry_price * (1 + gap_size * 0.005)  # 50% of gap
                    stop = entry_price * (1 - gap_size * 0.003)    # 30% stop
                    
                    if curr_day['High'] >= target:
                        exit_price = target
                    elif curr_day['Low'] <= stop:
                        exit_price = stop
                    else:
                        exit_price = curr_day['Close']
                else:  # SHORT
                    target = entry_price * (1 - gap_size * 0.005)  # 50% of gap
                    stop = entry_price * (1 + gap_size * 0.003)    # 30% stop
                    
                    if curr_day['Low'] <= target:
                        exit_price = target
                    elif curr_day['High'] >= stop:
                        exit_price = stop
                    else:
                        exit_price = curr_day['Close']
                
                trade_signal = {
                    'date': curr_day['Date'],
                    'direction': direction,
                    'entry_price': entry_price,
                    'exit_price': exit_price,
                    'gap_percent': curr_day['Gap_Percent'],
                    'signal_strength': 1.0
                }
                signals.append(trade_signal)
                
                if curr_date not in daily_trades:
                    daily_trades[curr_date] = 0
                daily_trades[curr_date] += 1
        
        self.signals = pd.DataFrame(signals)
        return self.signals
    
    def run_backtest(self, initial_capital=10000, position_size=1.0):
        """Run backtest simulation"""
        return super().run_backtest(initial_capital, position_size)
    
    def get_summary_metrics(self):
        """Get performance metrics with strategy-specific additions"""
        if self.performance_metrics is None:
            return None
        
        metrics = self.performance_metrics.copy()
        
        
        
        return metrics
