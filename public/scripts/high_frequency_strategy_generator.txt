        self.df['RSI'] = calculate_rsi(self.df, 14)
        self.df['RSI_5'] = calculate_rsi(self.df, 5)
        self.df['RSI_21'] = calculate_rsi(self.df, 21)
        
        self.df['ATR'] = calculate_atr(self.df, 14)
        self.df['ATR_5'] = calculate_atr(self.df, 5)
        self.df['ATR_21'] = calculate_atr(self.df, 21)
        
        upper_bb, middle_bb, lower_bb = calculate_bollinger_bands(self.df)
        self.df['BB_Upper'] = upper_bb
        self.df['BB_Middle'] = middle_bb
        self.df['BB_Lower'] = lower_bb
        
        macd_line, signal_line = calculate_macd(self.df)
        self.df['MACD'] = macd_line
        self.df['MACD_Signal'] = signal_line
        
        k_percent, d_percent = calculate_stochastic(self.df)
        self.df['Stoch_K'] = k_percent
        self.df['Stoch_D'] = d_percent
        
        self.df['OBV'] = calculate_obv(self.df)
        
        self.df['Volume_SMA'] = self.df['Volume'].rolling(window=20).mean().shift(1)
        self.df['High_Low_Range'] = (self.df['High'] - self.df['Low']).shift(1)
        self.df['Price_Change'] = self.df['Close'].pct_change().shift(1)
        self.df['Volume_Ratio'] = (self.df['Volume'] / self.df['Volume_SMA']).shift(1)

        self.df['Prev2DaysRed'] = (self.df['PrevDayRed'] & 
                                   self.df['PrevDayRed'].shift(1)).fillna(False)
        self.df['Prev3DaysRed'] = (self.df['Prev2DaysRed'] & 
                                   self.df['PrevDayRed'].shift(2)).fillna(False)
        
        self.df['PrevGap'] = (self.df['Open'] - self.df['Close'].shift(1)).shift(1)
        self.df['GapUp'] = (self.df['PrevGap'] > 0).fillna(False)
        self.df['GapDown'] = (self.df['PrevGap'] < 0).fillna(False)

    def test_two_indicator_combination(self, cond1_name, cond1, cond2_name, cond2):
        """Test a specific two-indicator combination"""
        combined_condition = cond1 & cond2
        
        trades = []
        for i in range(1, len(self.df)):
            if not combined_condition.iloc[i]:
                continue
                
            prev_day = self.df.iloc[i - 1]
            curr_day = self.df.iloc[i]
            
            gap = calculate_gap(prev_day['Close'], curr_day['Open'])
            if not is_gap_valid(gap):
                continue
            
            trade_direction = determine_trade_direction(gap)
            entry_price = curr_day['Open']
            exit_price, gap_closed = check_gap_closure(trade_direction, curr_day, prev_day['Close'])
            
            profit_loss = calculate_profit_loss(
                trade_direction, entry_price, exit_price, self.initial_investment
            )
            
            trades.append({
                'Date': curr_day['Date'],
                'Profit/Loss': profit_loss,
                'Gap Closed': gap_closed
            })
        
        if not trades:
            return None
            
        trades_df = pd.DataFrame(trades)
        total_trades = len(trades_df)
        
        if total_trades < 100:  # HIGH FREQUENCY: Require minimum 100 trades
            return None
            
        winning_trades = len(trades_df[trades_df['Profit/Loss'] > 0])
        success_rate = (winning_trades / total_trades * 100) if total_trades > 0 else 0
        total_profit = trades_df['Profit/Loss'].sum()
        avg_profit = trades_df['Profit/Loss'].mean()
        roi = (total_profit / self.initial_investment * 100)
        
        return {
            'Strategy': f"{cond1_name} + {cond2_name}",
            'Indicator1': cond1_name,
            'Indicator2': cond2_name,
            'Total Trades': total_trades,
            'Success Rate': round(success_rate, 2),
            'Total Profit': round(total_profit, 2),
            'Average Profit': round(avg_profit, 2),
            'ROI': round(roi, 2),
            'Final Equity': round(self.initial_investment + total_profit, 2)
        }
    
    def run_all_combinations(self):
        """Run all two-indicator combinations"""
        conditions = self.get_all_indicator_conditions()
        total_combinations = len(list(itertools.combinations(range(len(conditions)), 2)))
        
        print(f"\nTesting {total_combinations} two-indicator combinations...")
        print("HIGH FREQUENCY FILTER: Only strategies with 100+ trades will be included")
        print("This will take several minutes. Progress updates every 100 combinations.")
        
        results = []
        count = 0
        
        for (i, j) in itertools.combinations(range(len(conditions)), 2):
            cond1_name, cond1 = conditions[i]
            cond2_name, cond2 = conditions[j]
            
            result = self.test_two_indicator_combination(cond1_name, cond1, cond2_name, cond2)
            
            if result:
                results.append(result)
            
            count += 1
            if count % 100 == 0:
                print(f"Progress: {count}/{total_combinations} ({count/total_combinations*100:.1f}%) - Valid strategies found: {len(results)}")
        
        print(f"\nCompleted! High-frequency strategies found: {len(results)}")
        self.results = results
        
    def save_results(self):
        """Save results to CSV files"""
        if not self.results:
            print("No high-frequency results to save!")
            return
            
        results_df = pd.DataFrame(self.results)
        
        all_results_file = 'high_frequency_all_results.csv'
        results_df.to_csv(all_results_file, index=False)
        print(f"\nAll high-frequency results saved to: {all_results_file}")
        
        top_success = results_df.nlargest(100, 'Success Rate')
        top_success.to_csv('high_frequency_top100_success_rate.csv', index=False)
        
        top_profit = results_df.nlargest(100, 'Total Profit')
        top_profit.to_csv('high_frequency_top100_profit.csv', index=False)
        
        top_roi = results_df.nlargest(100, 'ROI')
        top_roi.to_csv('high_frequency_top100_roi.csv', index=False)
        
        return results_df
