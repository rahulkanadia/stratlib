"""
Master Script to Run All Trading Strategies

This script automatically finds and runs all trading strategies in the project.
It provides a comprehensive summary of performance across all strategies.

Perfect for beginners to see all strategies at once!
"""

import os
import sys
import pandas as pd
import importlib.util
from pathlib import Path
from datetime import datetime
import json
import warnings
warnings.filterwarnings('ignore')

sys.path.append(str(Path(__file__).parent))

class StrategyRunner:
    """Runs all strategies and collects results"""
    
    def __init__(self, strategies_dir="strategies", data_dir="data"):
        self.strategies_dir = Path(strategies_dir)
        self.data_dir = Path(data_dir)
        self.results = []
        self.errors = []
        
    def find_strategy_files(self):
        """Find all Python files in strategies directory"""
        strategy_files = []
        
        for root, dirs, files in os.walk(self.strategies_dir):
            for file in files:
                if file.endswith('.py') and not file.startswith('__'):
                    full_path = Path(root) / file
                    strategy_files.append(full_path)
        
        return sorted(strategy_files)
    
    def load_and_run_strategy(self, strategy_path):
        """Load and execute a single strategy file"""
        try:
            module_name = strategy_path.stem
            
            spec = importlib.util.spec_from_file_location(module_name, strategy_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            strategy_class = None
            for attr_name in dir(module):
                attr = getattr(module, attr_name)
                if (isinstance(attr, type) and 
                    attr_name.endswith('Strategy') and 
                    attr_name != 'BaseStrategy'):
                    strategy_class = attr
                    break
            
            if strategy_class is None:
                return None
            
            print(f"\nRunning {strategy_class.__name__}...")
            
            data_path = self.data_dir / "7J0OK1VR.csv"
            
            if 'vix' in module_name.lower() or 'VIX' in strategy_class.__name__:
                vix_path = self.data_dir / "VIX_History.csv"
                if hasattr(strategy_class, '__init__') and 'vix' in str(strategy_class.__init__):
                    strategy = strategy_class(data_path, vix_path)
                else:
                    strategy = strategy_class(data_path)
            else:
                strategy = strategy_class(data_path)
            
            signals = strategy.calculate_signals()
            
            strategy.run_backtest(initial_capital=10000)
            
            metrics = strategy.get_summary_metrics()
            
            if metrics:
                result = {
                    'strategy_name': strategy.name,
                    'file_path': str(strategy_path.relative_to(self.strategies_dir)),
                    'total_trades': metrics.get('total_trades', 0),
                    'win_rate': metrics.get('win_rate', 0),
                    'total_profit': metrics.get('total_profit', 0),
                    'sharpe_ratio': metrics.get('sharpe_ratio', 0),
                    'max_drawdown': metrics.get('max_drawdown', 0),
                    'profit_factor': metrics.get('profit_factor', 0)
                }
                
                return result
            
        except Exception as e:
            error_msg = f"Error in {strategy_path.name}: {str(e)}"
            print(f"  ❌ {error_msg}")
            self.errors.append(error_msg)
            return None
    
    def run_all_strategies(self):
        """Run all discovered strategies"""
        print("=" * 80)
        print("RUNNING ALL TRADING STRATEGIES")
        print("=" * 80)
        
        strategy_files = self.find_strategy_files()
        print(f"\nFound {len(strategy_files)} strategy files")
        
        for strategy_path in strategy_files:
            result = self.load_and_run_strategy(strategy_path)
            if result:
                self.results.append(result)
                print(f"  ✓ Success: {result['total_trades']} trades, "
                      f"{result['win_rate']:.1f}% win rate")
        
        print(f"\nSuccessfully ran {len(self.results)} strategies")
        if self.errors:
            print(f"Encountered {len(self.errors)} errors")
    
    def create_summary_report(self):
        """Create comprehensive summary of all strategies"""
        if not self.results:
            print("\nNo results to summarize")
            return
        
        print("\n" + "=" * 80)
        print("STRATEGY PERFORMANCE SUMMARY")
        print("=" * 80)
        
        df = pd.DataFrame(self.results)
        
        print("\n📊 TOP 5 BY WIN RATE:")
        print("-" * 60)
        top_win_rate = df.nlargest(5, 'win_rate')[['strategy_name', 'win_rate', 'total_trades', 'total_profit']]
        for _, row in top_win_rate.iterrows():
            print(f"{row['strategy_name']:30} | Win: {row['win_rate']:5.1f}% | "
                  f"Trades: {row['total_trades']:4} | Profit: ${row['total_profit']:8.2f}")
        
        print("\n💰 TOP 5 BY TOTAL PROFIT:")
        print("-" * 60)
        top_profit = df.nlargest(5, 'total_profit')[['strategy_name', 'total_profit', 'win_rate', 'total_trades']]
        for _, row in top_profit.iterrows():
            print(f"{row['strategy_name']:30} | Profit: ${row['total_profit']:8.2f} | "
                  f"Win: {row['win_rate']:5.1f}% | Trades: {row['total_trades']:4}")
        
        print("\n📈 TOP 5 BY SHARPE RATIO:")
        print("-" * 60)
        valid_sharpe = df[df['sharpe_ratio'].notna() & (df['sharpe_ratio'] != 0)]
        if len(valid_sharpe) > 0:
            top_sharpe = valid_sharpe.nlargest(5, 'sharpe_ratio')[['strategy_name', 'sharpe_ratio', 'total_profit', 'max_drawdown']]
            for _, row in top_sharpe.iterrows():
                print(f"{row['strategy_name']:30} | Sharpe: {row['sharpe_ratio']:6.2f} | "
                      f"Profit: ${row['total_profit']:8.2f} | DD: {row['max_drawdown']:5.1f}%")
        
        print("\n📉 LOWEST DRAWDOWNS:")
        print("-" * 60)
        low_dd = df.nsmallest(5, 'max_drawdown', keep='all')[['strategy_name', 'max_drawdown', 'total_profit', 'win_rate']]
        for _, row in low_dd.iterrows():
            print(f"{row['strategy_name']:30} | DD: {row['max_drawdown']:6.1f}% | "
                  f"Profit: ${row['total_profit']:8.2f} | Win: {row['win_rate']:5.1f}%")
        
        print("\n" + "=" * 80)
        print("OVERALL STATISTICS")
        print("=" * 80)
        print(f"Total Strategies Tested: {len(df)}")
        print(f"Average Win Rate: {df['win_rate'].mean():.1f}%")
        print(f"Total Combined Profit: ${df['total_profit'].sum():.2f}")
        print(f"Average Profit per Strategy: ${df['total_profit'].mean():.2f}")
        print(f"Best Strategy: {df.loc[df['total_profit'].idxmax(), 'strategy_name']}")
        print(f"Most Active Strategy: {df.loc[df['total_trades'].idxmax(), 'strategy_name']} "
              f"({df['total_trades'].max()} trades)")
        
        self.save_results(df)
    
    def save_results(self, df):
        """Save results to CSV and JSON"""
        results_dir = Path("results")
        results_dir.mkdir(exist_ok=True)
        
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        
        csv_path = results_dir / f"all_strategies_results_{timestamp}.csv"
        df.to_csv(csv_path, index=False)
        
        json_data = {
            'run_date': datetime.now().isoformat(),
            'total_strategies': len(df),
            'results': df.to_dict('records'),
            'errors': self.errors
        }
        
        json_path = results_dir / f"all_strategies_results_{timestamp}.json"
        with open(json_path, 'w') as f:
            json.dump(json_data, f, indent=2)
        
        print(f"\n📁 Results saved to:")
        print(f"   - {csv_path}")
        print(f"   - {json_path}")

def main():
    """Main execution function"""
    print("\n🚀 AUTOMATED STRATEGY TESTING SYSTEM")
    print("=" * 80)
    print("This system will automatically find and test all trading strategies")
    print("Each strategy will be backtested with $10,000 initial capital")
    print("=" * 80)
    
    runner = StrategyRunner()
    
    runner.run_all_strategies()
    
    runner.create_summary_report()
    
    print("\n✅ All strategies have been tested!")
    print("\n💡 TIP: Check individual strategy files to understand how each one works")
    print("💡 TIP: Modify strategy parameters in their files to optimize performance")

if __name__ == "__main__":
    main()
