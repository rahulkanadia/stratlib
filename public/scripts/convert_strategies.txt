def identify_strategy_pattern(file_content):
    """Identify the pattern/type of strategy from file content."""
    patterns = {
        'dip': ['rolling.*drop', 'dip', 'avg.*drop', 'rolling.*average.*drop'],
        'gap_and_go': ['gap.*go', 'gap.*direction', 'profit.*target', 'stop.*loss'],
        'gap_fade': ['gap.*fade', 'opposite.*gap', 'fade.*gap'],
        'vix': ['vix', 'volatility.*index'],
        'ml': ['sklearn', 'train_test_split', 'classifier', 'random.*forest', 'gradient.*boost'],
        'atr': ['atr', 'average.*true.*range'],
        'ma': ['moving.*average', 'sma', 'ema', 'crossover']
    }
    
    content_lower = file_content.lower()
    
    for strategy_type, keywords in patterns.items():
        for keyword in keywords:
            if re.search(keyword, content_lower):
                return strategy_type
    
    return 'other'


def extract_strategy_info(file_path):
    """Extract key information from existing strategy file."""
    info = {
        'file_name': os.path.basename(file_path),
        'strategy_type': 'unknown',
        'has_multiple_strategies': False,
        'uses_vix': False,
        'uses_ml': False,
        'main_indicators': [],
        'entry_logic': '',
        'exit_logic': ''
    }
    
    try:
        with open(file_path, 'r') as f:
            content = f.read()
        
        info['strategy_type'] = identify_strategy_pattern(content)
        
        info['uses_vix'] = 'vix' in content.lower()
        
        info['uses_ml'] = 'sklearn' in content or 'train_test_split' in content
        
        class_count = len(re.findall(r'class\s+\w+.*:', content))
        info['has_multiple_strategies'] = class_count > 1
        
        indicator_patterns = [
            'rsi', 'macd', 'bollinger', 'atr', 'ema', 'sma', 
            'volume', 'stochastic', 'momentum', 'adx'
        ]
        for indicator in indicator_patterns:
            if indicator in content.lower():
                info['main_indicators'].append(indicator.upper())
        
    except Exception as e:
        print(f"Error reading {file_path}: {e}")
    
    return info


def create_conversion_report(source_dir):
    """Create a report of all strategies that need conversion."""
    report = []
    
    for root, dirs, files in os.walk(source_dir):
        if 'clean_strategies' in root:
            continue
            
        for file in files:
            if file.endswith('.py') and not file.startswith('__'):
                file_path = os.path.join(root, file)
                info = extract_strategy_info(file_path)
                info['full_path'] = file_path
                info['relative_path'] = os.path.relpath(file_path, source_dir)
                report.append(info)
    
    summary = {
        'total_files': len(report),
        'by_type': {},
        'multiple_strategies': [],
        'uses_vix': [],
        'uses_ml': []
    }
    
    for item in report:
        strategy_type = item['strategy_type']
        summary['by_type'][strategy_type] = summary['by_type'].get(strategy_type, 0) + 1
        
        if item['has_multiple_strategies']:
            summary['multiple_strategies'].append(item['relative_path'])
        if item['uses_vix']:
            summary['uses_vix'].append(item['relative_path'])
        if item['uses_ml']:
            summary['uses_ml'].append(item['relative_path'])
    
    return report, summary
