def fix_file(filepath, file_info):
    """Fix a single file with the correct conditions"""
    
    with open(filepath, 'r') as f:
        content = f.read()
    
    pattern = r"(def get_filter_conditions\(self, df\):.*?# Condition 1:.*?\n\s*condition1 = .*?\n\s*# Condition 2:.*?\n\s*condition2 = .*?\n\s*# Combine conditions.*?\n\s*return condition1 & condition2)"
    
    replacement = f'''def get_filter_conditions(self, df):
        """Get the two indicator conditions"""
        condition1 = {file_info['condition1']}
        
        condition2 = {file_info['condition2']}
        
        return condition1 & condition2'''
    
    content = re.sub(pattern, replacement, content, flags=re.DOTALL)
    
    if 'GapUp' in content and 'df[\'GapUp\']' not in content:
        calc_pattern = r"(def calculate_indicators\(self, df\):.*?)(return df)"
        calc_replacement = r'''\1
        df['PrevGap'] = (df['Open'] - df['Close'].shift(1)).shift(1)
        df['GapUp'] = (df['PrevGap'] > 0).fillna(False)
        df['GapDown'] = (df['PrevGap'] < 0).fillna(False)
        
        \2'''
        content = re.sub(calc_pattern, calc_replacement, content, flags=re.DOTALL)
    
    with open(filepath, 'w') as f:
        f.write(content)
    
    return True


def main():
    """Fix all strategy files"""
    fixes = get_comprehensive_fixes()
    
    fixed_count = 0
    for filename, file_info in fixes.items():
        filepath = os.path.join("/mnt/c/Users/danie/OneDrive/Desktop/claude/2020 to 2024/GAP_CLOSE_OPTIMIZATIONS/HIGH_FREQUENCY_TWO_INDICATORS", filename)
        
        if os.path.exists(filepath):
            print(f"Fixing {filename}...")
            if fix_file(filepath, file_info):
                fixed_count += 1
            else:
                print(f"  Failed to fix {filename}")
        else:
            print(f"  File not found: {filename}")
    
    print(f"\nFixed {fixed_count}/{len(fixes)} files")


if __name__ == "__main__":
    main()
