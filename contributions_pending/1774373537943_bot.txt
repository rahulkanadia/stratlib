//+------------------------------------------------------------------+
//| EMA + DMI Pro Trading Bot                                        |
//+------------------------------------------------------------------+
#property strict
#include <Trade/Trade.mqh>
CTrade trade;

// INPUTS
input int FastEMA = 20;
input int SlowEMA = 50;
input int ADXPeriod = 14;

input double RiskPercent = 1.0;
input double StopLossPoints = 300;
input double TakeProfitPoints = 600;

input bool AllowBuy = true;
input bool AllowSell = true;

input double MaxSpread = 30;

// INDICATOR HANDLES
int emaFastHandle;
int emaSlowHandle;
int adxHandle;

//+------------------------------------------------------------------+
int OnInit()
{
   emaFastHandle = iMA(_Symbol,_Period,FastEMA,0,MODE_EMA,PRICE_CLOSE);
   emaSlowHandle = iMA(_Symbol,_Period,SlowEMA,0,MODE_EMA,PRICE_CLOSE);
   adxHandle = iADX(_Symbol,_Period,ADXPeriod);

   return(INIT_SUCCEEDED);
}
//+------------------------------------------------------------------+

void OnTick()
{
   if(PositionSelect(_Symbol)) return;

   double spread = (SymbolInfoDouble(_Symbol,SYMBOL_ASK) -
                   SymbolInfoDouble(_Symbol,SYMBOL_BID))/_Point;

   if(spread > MaxSpread) return;

   double emaFast[1];
   double emaSlow[1];
   double adx[1];
   double plusDI[1];
   double minusDI[1];

   CopyBuffer(emaFastHandle,0,0,1,emaFast);
   CopyBuffer(emaSlowHandle,0,0,1,emaSlow);

   CopyBuffer(adxHandle,0,0,1,adx);
   CopyBuffer(adxHandle,1,0,1,plusDI);
   CopyBuffer(adxHandle,2,0,1,minusDI);

   double ask = SymbolInfoDouble(_Symbol,SYMBOL_ASK);
   double bid = SymbolInfoDouble(_Symbol,SYMBOL_BID);

   double lot = CalculateLot();

   // BUY SIGNAL
   if(AllowBuy &&
      emaFast[0] > emaSlow[0] &&
      plusDI[0] > minusDI[0] &&
      adx[0] > 20)
   {
      double sl = ask - StopLossPoints*_Point;
      double tp = ask + TakeProfitPoints*_Point;

      trade.Buy(lot,_Symbol,ask,sl,tp);
   }

   // SELL SIGNAL
   if(AllowSell &&
      emaFast[0] < emaSlow[0] &&
      minusDI[0] > plusDI[0] &&
      adx[0] > 20)
   {
      double sl = bid + StopLossPoints*_Point;
      double tp = bid - TakeProfitPoints*_Point;

      trade.Sell(lot,_Symbol,bid,sl,tp);
   }
}

//+------------------------------------------------------------------+

double CalculateLot()
{
   double balance = AccountInfoDouble(ACCOUNT_BALANCE);
   double riskMoney = balance * RiskPercent / 100.0;

   double tickValue = SymbolInfoDouble(_Symbol,SYMBOL_TRADE_TICK_VALUE);

   double lot = riskMoney / (StopLossPoints * tickValue);

   return NormalizeDouble(lot,2);
}
